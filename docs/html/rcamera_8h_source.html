227// Returns the cameras forward vector (normalized)
228Vector3 GetCameraForward(Camera* camera)
229{
230 return Vector3Normalize(Vector3Subtract(camera->target, camera->position));
231}
232
233// Returns the cameras up vector (normalized)
234// Note: The up vector might not be perpendicular to the forward vector
235Vector3 GetCameraUp(Camera* camera)
236{
237 return Vector3Normalize(camera->up);
238}
239
240// Returns the cameras right vector (normalized)
241Vector3 GetCameraRight(Camera* camera)
242{
243 Vector3 forward = GetCameraForward(camera);
244 Vector3 up = GetCameraUp(camera);
245
246 return Vector3Normalize(Vector3CrossProduct(forward, up));
247}

249// Moves the camera in its forward direction
250void CameraMoveForward(Camera* camera, float distance, bool moveInWorldPlane)
251{
252 Vector3 forward = GetCameraForward(camera);
253
254 if (moveInWorldPlane)
255 {
256 // Project vector onto world plane
257 forward.y = 0;
258 forward = Vector3Normalize(forward);
259 }
260
261 // Scale by distance
262 forward = Vector3Scale(forward, distance);
263
264 // Move position and target
265 camera->position = Vector3Add(camera->position, forward);
266 camera->target = Vector3Add(camera->target, forward);
267}
268
269// Moves the camera in its up direction
270void CameraMoveUp(Camera* camera, float distance)
271{
272 Vector3 up = GetCameraUp(camera);
273
274 // Scale by distance
275 up = Vector3Scale(up, distance);
276
277 // Move position and target
278 camera->position = Vector3Add(camera->position, up);
279 camera->target = Vector3Add(camera->target, up);
280}

282// Moves the camera target in its current right direction
283void CameraMoveRight(Camera* camera, float distance, bool moveInWorldPlane)
284{

288 {
289 // Project vector onto world plane
290 right.y = 0;
291 right = Vector3Normalize(right);
292 }
293
294 // Scale by distance
295 right = Vector3Scale(right, distance);
296
297 // Move position and target
298 camera->position = Vector3Add(camera->position, right);
299 camera->target = Vector3Add(camera->target, right);
300}

302// Moves the camera position closer/farther to/from the camera target
303void CameraMoveToTarget(Camera* camera, float delta)
304{
305 float distance = Vector3Distance(camera->position, camera->target);
306
307 // Apply delta
308 distance += delta;
309
310 // Distance must be greater than 0
311 if (distance <= 0) distance = 0.001f;
312
313 // Set new distance by moving the position along the forward vector
314 Vector3 forward = GetCameraForward(camera);
315 camera->position = Vector3Add(camera->target, Vector3Scale(forward, -distance));
316}

318// Rotates the camera around its up vector
319// Yaw is "looking left and right"
320// If rotateAroundTarget is false, the camera rotates around its position
321// Note: angle must be provided in radians
322void CameraYaw(Camera* camera, float angle, bool rotateAroundTarget)
323{
324 // Rotation axis
325 Vector3 up = GetCameraUp(camera);
326
327 // View vector
328 Vector3 targetPosition = Vector3Subtract(camera->target, camera->position);
329
330 // Rotate view vector around up axis
331 targetPosition = Vector3RotateByAxisAngle(targetPosition, up, angle);
332
333 if (rotateAroundTarget)
334 {
335 // Move position relative to target
336 camera->position = Vector3Subtract(camera->target, targetPosition);
337 }
338 else // rotate around camera.position
339 {
340 // Move target relative to position
341 camera->target = Vector3Add(camera->position, targetPosition);
342 }
343}
344
345// Rotates the camera around its right vector, pitch is "looking up and down"
346// - lockView prevents camera overrotation (aka "somersaults")
347// - rotateAroundTarget defines if rotation is around target or around its position
348// - rotateUp rotates the up direction as well (typically only usefull in CAMERA_FREE)
349// NOTE: angle must be provided in radians
350void CameraPitch(Camera* camera, float angle, bool lockView, bool rotateAroundTarget, bool rotateUp)
351{
352 // Up direction
353 Vector3 up = GetCameraUp(camera);
354
355 // View vector
356 Vector3 targetPosition = Vector3Subtract(camera->target, camera->position);
357
358 if (lockView)
359 {
360 // In these camera modes we clamp the Pitch angle
361 // to allow only viewing straight up or down.
362
363 // Clamp view up
364 float maxAngleUp = Vector3Angle(up, targetPosition);
365 maxAngleUp -= 0.001f; // avoid numerical errors
366 if (angle > maxAngleUp) angle = maxAngleUp;
367
368 // Clamp view down
369 float maxAngleDown = Vector3Angle(Vector3Negate(up), targetPosition);
370 maxAngleDown *= -1.0f; // downwards angle is negative
371 maxAngleDown += 0.001f; // avoid numerical errors
372 if (angle < maxAngleDown) angle = maxAngleDown;
373 }
374
375 // Rotation axis
376 Vector3 right = GetCameraRight(camera);
377
378 // Rotate view vector around right axis
379 targetPosition = Vector3RotateByAxisAngle(targetPosition, right, angle);
380
381 if (rotateAroundTarget)
382 {
383 // Move position relative to target
384 camera->position = Vector3Subtract(camera->target, targetPosition);
385 }
386 else // rotate around camera.position
387 {
388 // Move target relative to position
389 camera->target = Vector3Add(camera->position, targetPosition);
390 }
391
392 if (rotateUp)
393 {
394 // Rotate up direction around right axis
395 camera->up = Vector3RotateByAxisAngle(camera->up, right, angle);
396 }
397}

399// Rotates the camera around its forward vector
400// Roll is "turning your head sideways to the left or right"
401// Note: angle must be provided in radians
402void CameraRoll(Camera* camera, float angle)
403{
404 // Rotation axis
405 Vector3 forward = GetCameraForward(camera);
406
407 // Rotate up direction around forward axis
408 camera->up = Vector3RotateByAxisAngle(camera->up, forward, angle);
409}

411// Returns the camera view matrix
412Matrix GetCameraViewMatrix(Camera* camera)
413{
414 return MatrixLookAt(camera->position, camera->target, camera->up);
415}

417// Returns the camera projection matrix
418Matrix GetCameraProjectionMatrix(Camera* camera, float aspect)
419{
420 if (camera->projection == CAMERA_PERSPECTIVE)
421 {
422 return MatrixPerspective(camera->fovy * DEG2RAD, aspect, CAMERA_CULL_DISTANCE_NEAR, CAMERA_CULL_DISTANCE_FAR);
423 }
424 else if (camera->projection == CAMERA_ORTHOGRAPHIC)
425 {
426 double top = camera->fovy / 2.0;
427 double right = top * aspect;
428
429 return MatrixOrtho(-right, right, -top, top, CAMERA_CULL_DISTANCE_NEAR, CAMERA_CULL_DISTANCE_FAR);
430 }
431
432 return MatrixIdentity();
433}

436// Update camera position for selected mode
437// Camera mode: CAMERA_FREE, CAMERA_FIRST_PERSON, CAMERA_THIRD_PERSON, CAMERA_ORBITAL or CUSTOM
438void UpdateCamera(Camera* camera, int mode)
439{
440 Vector2 mousePositionDelta = GetMouseDelta();
441
442 bool moveInWorldPlane = ((mode == CAMERA_FIRST_PERSON) || (mode == CAMERA_THIRD_PERSON));
443 bool rotateAroundTarget = ((mode == CAMERA_THIRD_PERSON) || (mode == CAMERA_ORBITAL));
444 bool lockView = ((mode == CAMERA_FREE) || (mode == CAMERA_FIRST_PERSON) || (mode == CAMERA_THIRD_PERSON) || (mode == CAMERA_ORBITAL));
445 bool rotateUp = false;
446
447 if (mode == CAMERA_ORBITAL)
448 {
449 // Orbital can just orbit
450 Matrix rotation = MatrixRotate(GetCameraUp(camera), CAMERA_ORBITAL_SPEED * GetFrameTime());
451 Vector3 view = Vector3Subtract(camera->position, camera->target);
452 view = Vector3Transform(view, rotation);
453 camera->position = Vector3Add(camera->target, view);
454 }
455 else
456 {
457 // Camera rotation
458 if (IsKeyDown(KEY_DOWN)) CameraPitch(camera, -CAMERA_ROTATION_SPEED, lockView, rotateAroundTarget, rotateUp);
459 if (IsKeyDown(KEY_UP)) CameraPitch(camera, CAMERA_ROTATION_SPEED, lockView, rotateAroundTarget, rotateUp);
460 if (IsKeyDown(KEY_RIGHT)) CameraYaw(camera, -CAMERA_ROTATION_SPEED, rotateAroundTarget);
461 if (IsKeyDown(KEY_LEFT)) CameraYaw(camera, CAMERA_ROTATION_SPEED, rotateAroundTarget);
462 if (IsKeyDown(KEY_Q)) CameraRoll(camera, -CAMERA_ROTATION_SPEED);
463 if (IsKeyDown(KEY_E)) CameraRoll(camera, CAMERA_ROTATION_SPEED);
464
465 // Camera movement
466 // Camera pan (for CAMERA_FREE)
467 if ((mode == CAMERA_FREE) && (IsMouseButtonDown(MOUSE_BUTTON_MIDDLE)))
468 {
469 const Vector2 mouseDelta = GetMouseDelta();
470 if (mouseDelta.x > 0.0f) CameraMoveRight(camera, CAMERA_PAN_SPEED, moveInWorldPlane);
471 if (mouseDelta.x < 0.0f) CameraMoveRight(camera, -CAMERA_PAN_SPEED, moveInWorldPlane);
472 if (mouseDelta.y > 0.0f) CameraMoveUp(camera, -CAMERA_PAN_SPEED);
473 if (mouseDelta.y < 0.0f) CameraMoveUp(camera, CAMERA_PAN_SPEED);
474 }
475 else
476 {
477 // Mouse support
478 CameraYaw(camera, -mousePositionDelta.x * CAMERA_MOUSE_MOVE_SENSITIVITY, rotateAroundTarget);
479 CameraPitch(camera, -mousePositionDelta.y * CAMERA_MOUSE_MOVE_SENSITIVITY, lockView, rotateAroundTarget, rotateUp);
480 }
481
482 // Keyboard support
483 if (IsKeyDown(KEY_W)) CameraMoveForward(camera, CAMERA_MOVE_SPEED, moveInWorldPlane);
484 if (IsKeyDown(KEY_A)) CameraMoveRight(camera, -CAMERA_MOVE_SPEED, moveInWorldPlane);
485 if (IsKeyDown(KEY_S)) CameraMoveForward(camera, -CAMERA_MOVE_SPEED, moveInWorldPlane);
486 if (IsKeyDown(KEY_D)) CameraMoveRight(camera, CAMERA_MOVE_SPEED, moveInWorldPlane);
487
488 // Gamepad movement
489 if (IsGamepadAvailable(0))
490 {
491 // Gamepad controller support
492 CameraYaw(camera, -(GetGamepadAxisMovement(0, GAMEPAD_AXIS_RIGHT_X) * 2) * CAMERA_MOUSE_MOVE_SENSITIVITY, rotateAroundTarget);
493 CameraPitch(camera, -(GetGamepadAxisMovement(0, GAMEPAD_AXIS_RIGHT_Y) * 2) * CAMERA_MOUSE_MOVE_SENSITIVITY, lockView, rotateAroundTarget, rotateUp);
494
495 if (GetGamepadAxisMovement(0, GAMEPAD_AXIS_LEFT_Y) <= -0.25f) CameraMoveForward(camera, CAMERA_MOVE_SPEED, moveInWorldPlane);
496 if (GetGamepadAxisMovement(0, GAMEPAD_AXIS_LEFT_X) <= -0.25f) CameraMoveRight(camera, -CAMERA_MOVE_SPEED, moveInWorldPlane);
497 if (GetGamepadAxisMovement(0, GAMEPAD_AXIS_LEFT_Y) >= 0.25f) CameraMoveForward(camera, -CAMERA_MOVE_SPEED, moveInWorldPlane);
498 if (GetGamepadAxisMovement(0, GAMEPAD_AXIS_LEFT_X) >= 0.25f) CameraMoveRight(camera, CAMERA_MOVE_SPEED, moveInWorldPlane);
499 }
500
501 if (mode == CAMERA_FREE)
502 {
503 if (IsKeyDown(KEY_SPACE)) CameraMoveUp(camera, CAMERA_MOVE_SPEED);
504 if (IsKeyDown(KEY_LEFT_CONTROL)) CameraMoveUp(camera, -CAMERA_MOVE_SPEED);
505 }
506 }
507
508 if ((mode == CAMERA_THIRD_PERSON) || (mode == CAMERA_ORBITAL) || (mode == CAMERA_FREE))
509 {
510 // Zoom target distance
511 CameraMoveToTarget(camera, -GetMouseWheelMove());
512 if (IsKeyPressed(KEY_KP_SUBTRACT)) CameraMoveToTarget(camera, 2.0f);
513 if (IsKeyPressed(KEY_KP_ADD)) CameraMoveToTarget(camera, -2.0f);
514 }
515}

518// Update camera movement, movement/rotation values should be provided by user
519void UpdateCameraPro(Camera* camera, Vector3 movement, Vector3 rotation, float zoom)
520{
521 // Required values
522 // movement.x - Move forward/backward
523 // movement.y - Move right/left
524 // movement.z - Move up/down
525 // rotation.x - yaw
526 // rotation.y - pitch
527 // rotation.z - roll
528 // zoom - Move towards target
529
530 bool lockView = true;
531 bool rotateAroundTarget = false;
532 bool rotateUp = false;
533 bool moveInWorldPlane = true;
534
535 // Camera rotation
536 CameraPitch(camera, -rotation.y * DEG2RAD, lockView, rotateAroundTarget, rotateUp);
537 CameraYaw(camera, -rotation.x * DEG2RAD, rotateAroundTarget);
538 CameraRoll(camera, rotation.z * DEG2RAD);
539
540 // Camera movement
541 CameraMoveForward(camera, movement.x, moveInWorldPlane);
542 CameraMoveRight(camera, movement.y, moveInWorldPlane);
543 CameraMoveUp(camera, movement.z);
544
545 // Zoom target distance
546 CameraMoveToTarget(camera, zoom);
547}

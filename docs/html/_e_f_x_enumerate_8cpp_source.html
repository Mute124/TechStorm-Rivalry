27int main()
28{
29 ALCdevice * pDevice;
30 ALuint uiEffectSlots[128] = { 0 };
31 ALuint uiEffects[1] = { 0 };
32 ALuint uiFilters[1] = { 0 };
33 ALint iEffectSlotsGenerated;
34 ALint iSends;
35
36 // Initialize Framework
37 ALFWInit();
38
39 ALFWprintf("Enumerate EFX Application\n");
40
41 if (!ALFWInitOpenAL())
42 {
43 ALFWprintf("Failed to initialize OpenAL\n");
44 ALFWShutdown();
45 return 0;
46 }
47
48 // Get the current AL Device
49 pDevice = alcGetContextsDevice(alcGetCurrentContext());
50
51 // Check for EFX Extension
52 if (ALFWIsEFXSupported())
53 {
54 // To determine how many Auxiliary Effects Slots are available, create as many as possible (up to 128)
55 // until the call fails.
56 for (iEffectSlotsGenerated = 0; iEffectSlotsGenerated < 128; iEffectSlotsGenerated++)
57 {
58 alGenAuxiliaryEffectSlots(1, &uiEffectSlots[iEffectSlotsGenerated]);
59 if (alGetError() != AL_NO_ERROR)
60 break;
61 }
62
63 ALFWprintf("%d Auxiliary Effect Slot%s\n", iEffectSlotsGenerated, (iEffectSlotsGenerated == 1) ? "" : "s");
64
65 // Retrieve the number of Auxiliary Effect Slots Sends available on each Source
66 alcGetIntegerv(pDevice, ALC_MAX_AUXILIARY_SENDS, 1, &iSends);
67 printf("%d Auxiliary Send%s per Source\n", iSends, (iSends == 1) ? "" : "s");
68
69 // To determine which Effects are supported, generate an Effect Object, and try to set its type to
70 // the various Effect enum values
71 printf("\nEffects Supported: -\n");
72 alGenEffects(1, &uiEffects[0]);
73 if (alGetError() == AL_NO_ERROR)
74 {
75 // Try setting Effect Type to known Effects
76 alEffecti(uiEffects[0], AL_EFFECT_TYPE, AL_EFFECT_REVERB);
77 printf("'Reverb' Support %s\n", (alGetError() == AL_NO_ERROR) ? "YES" : "NO");
78
79 alEffecti(uiEffects[0], AL_EFFECT_TYPE, AL_EFFECT_EAXREVERB);
80 printf("'EAX Reverb' Support %s\n", (alGetError() == AL_NO_ERROR) ? "YES" : "NO");
81
82 alEffecti(uiEffects[0], AL_EFFECT_TYPE, AL_EFFECT_CHORUS);
83 printf("'Chorus' Support %s\n", (alGetError() == AL_NO_ERROR) ? "YES" : "NO");
84
85 alEffecti(uiEffects[0], AL_EFFECT_TYPE, AL_EFFECT_DISTORTION);
86 printf("'Distortion' Support %s\n", (alGetError() == AL_NO_ERROR) ? "YES" : "NO");
87
88 alEffecti(uiEffects[0], AL_EFFECT_TYPE, AL_EFFECT_ECHO);
89 printf("'Echo' Support %s\n", (alGetError() == AL_NO_ERROR) ? "YES" : "NO");
90
91 alEffecti(uiEffects[0], AL_EFFECT_TYPE, AL_EFFECT_FLANGER);
92 printf("'Flanger' Support %s\n", (alGetError() == AL_NO_ERROR) ? "YES" : "NO");
93
94 alEffecti(uiEffects[0], AL_EFFECT_TYPE, AL_EFFECT_FREQUENCY_SHIFTER);
95 printf("'Frequency Shifter' Support %s\n", (alGetError() == AL_NO_ERROR) ? "YES" : "NO");
96
97 alEffecti(uiEffects[0], AL_EFFECT_TYPE, AL_EFFECT_VOCAL_MORPHER);
98 printf("'Vocal Morpher' Support %s\n", (alGetError() == AL_NO_ERROR) ? "YES" : "NO");
99
100 alEffecti(uiEffects[0], AL_EFFECT_TYPE, AL_EFFECT_PITCH_SHIFTER);
101 printf("'Pitch Shifter' Support %s\n", (alGetError() == AL_NO_ERROR) ? "YES" : "NO");
102
103 alEffecti(uiEffects[0], AL_EFFECT_TYPE, AL_EFFECT_RING_MODULATOR);
104 printf("'Ring Modulator' Support %s\n", (alGetError() == AL_NO_ERROR) ? "YES" : "NO");
105
106 alEffecti(uiEffects[0], AL_EFFECT_TYPE, AL_EFFECT_AUTOWAH);
107 printf("'Autowah' Support %s\n", (alGetError() == AL_NO_ERROR) ? "YES" : "NO");
108
109 alEffecti(uiEffects[0], AL_EFFECT_TYPE, AL_EFFECT_COMPRESSOR);
110 printf("'Compressor' Support %s\n", (alGetError() == AL_NO_ERROR) ? "YES" : "NO");
111
112 alEffecti(uiEffects[0], AL_EFFECT_TYPE, AL_EFFECT_EQUALIZER);
113 printf("'Equalizer' Support %s\n", (alGetError() == AL_NO_ERROR) ? "YES" : "NO");
114 }
115
116 // To determine which Filters are supported, generate a Filter Object, and try to set its type to
117 // the various Filter enum values
118 printf("\nFilters Supported: -\n");
119 // Generate a Filter to use to determine what Filter Types are supported
120 alGenFilters(1, &uiFilters[0]);
121 if (alGetError() == AL_NO_ERROR)
122 {
123 // Try setting the Filter type to known Filters
124 alFilteri(uiFilters[0], AL_FILTER_TYPE, AL_FILTER_LOWPASS);
125 printf("'Low Pass' Support %s\n", (alGetError() == AL_NO_ERROR) ? "YES" : "NO");
126
127 alFilteri(uiFilters[0], AL_FILTER_TYPE, AL_FILTER_HIGHPASS);
128 printf("'High Pass' Support %s\n", (alGetError() == AL_NO_ERROR) ? "YES" : "NO");
129
130 alFilteri(uiFilters[0], AL_FILTER_TYPE, AL_FILTER_BANDPASS);
131 printf("'Band Pass' Support %s\n", (alGetError() == AL_NO_ERROR) ? "YES" : "NO");
132 }
133
134 // Clean-up ...
135
136 // Delete Filter
137 alDeleteFilters(1, &uiFilters[0]);
138
139 // Delete Effect
140 alDeleteEffects(1, &uiEffects[0]);
141
142 // Delete Auxiliary Effect Slots
143 alDeleteAuxiliaryEffectSlots(iEffectSlotsGenerated, uiEffectSlots);
144 }
145 else
146 {
147 ALFWprintf("EFX support not found\n");
148 }
149
150 ALFWShutdownOpenAL();
151
152 ALFWShutdown();
153
154 return 0;
155}

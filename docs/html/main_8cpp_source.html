4 PURPOSE : Use OpenAL to render pre-mixed audio data from a S/W mixer. Optionally

193// Hypothetical Software Mixer
194// ulNumSamples : Number of samples requested
195// ulOutputChannels : Number of output channels (dry-mix)
196// pOutput : Destination address for dry-mix (expects 'ulNumSamples' 16bit Samples for 'ulOutputChannels' Channels
197// ulNumEffects : Number of effect outputs (wet-mixes)
198// pEffectOutput : Array of desintation address for the wet-mixes (expect 'ulNumSamples' 16bit Samples for 1 output channel (i.e. mono))
199void MixData(unsigned long ulNumSamples, unsigned long ulOutputChannels, void *pOutput, unsigned long ulNumEffects, void **pEffectOutput)
200{
201 unsigned long ulRequestedBytes;
202 unsigned long ulBytesWritten = 0;
203 unsigned long ulSample;
204 unsigned long ulChannel;
205
206 // Fill Dry Buffer with audio
207 ulRequestedBytes = ulNumSamples * ulOutputChannels * 2;
208 while (ulRequestedBytes)
209 {
210 if (g_bPlayTestWaveFile)
211 {
212 // Data for Dry Buffer
213 g_pWaveLoader->ReadWaveData(g_WaveID, (char*)pOutput + ulBytesWritten, ulRequestedBytes, &ulBytesWritten);
214 ulRequestedBytes -= ulBytesWritten;
215 if (ulRequestedBytes)
216 g_pWaveLoader->SetWaveDataOffset(g_WaveID, 0);
217 }
218 else
219 {
220 // Silence for Dry Buffer
221 memset(pOutput, 0, ulRequestedBytes);
222 ulRequestedBytes = 0;
223 }
224 }
225
226 // Fill Effect Buffers with audio
227 ulRequestedBytes = ulNumSamples * 2;
228 for (unsigned long ulEffectBuffers = 0; ulEffectBuffers < ulNumEffects; ulEffectBuffers++)
229 {
230 if (g_bPlayTestWaveFile && (g_flEffectSends[ulEffectBuffers] != 0.0f))
231 {
232 for (ulSample = 0; ulSample < ulNumSamples; ulSample++)
233 {
234 float flMixSample = 0.0f;
235 for (ulChannel = 0; ulChannel < ulOutputChannels; ulChannel++)
236 flMixSample += ((short*)(pOutput))[(ulSample * ulOutputChannels) + ulChannel];
237
238 if (flMixSample < -32768.0f)
239 flMixSample = -32768.0f;
240 else if (flMixSample > 32767.0f)
241 flMixSample = 32767.0f;
242
243 ((short*)(pEffectOutput[ulEffectBuffers]))[ulSample] = (short)(flMixSample * g_flEffectSends[ulEffectBuffers]);
244 }
245 }
246 else
247 {
248 // Silence for Effect Buffers
249 memset(pEffectOutput[ulEffectBuffers], 0, ulRequestedBytes);
250 }
251 }

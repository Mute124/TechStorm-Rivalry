25#include "Framework.h"
26#include "CWaves.h"

28#define NUMBUFFERS (4)
29#define SERVICE_UPDATE_PERIOD (20)
30
31#define TEST_WAVE_FILE "stereo.wav"
32
33int main()
34{
35 ALuint uiBuffers[NUMBUFFERS];
36 ALuint uiSource;
37 ALuint uiBuffer;
38 ALint iState;
39 CWaves * pWaveLoader = NULL;
40 WAVEID WaveID;
41 ALint iLoop;
42 ALint iBuffersProcessed, iTotalBuffersProcessed, iQueuedBuffers;
43 WAVEFORMATEX wfex;
44 unsigned long ulDataSize = 0;
45 unsigned long ulFrequency = 0;
46 unsigned long ulFormat = 0;
47 unsigned long ulBufferSize;
48 unsigned long ulBytesWritten;
49 void * pData = NULL;
50
51 // Initialize Framework
52 ALFWInit();
53
54 ALFWprintf("PlayStream Test Application\n");
55
56 if (!ALFWInitOpenAL())
57 {
58 ALFWprintf("Failed to initialize OpenAL\n");
59 ALFWShutdown();
60 return 0;
61 }
62
63 // Generate some AL Buffers for streaming
64 alGenBuffers( NUMBUFFERS, uiBuffers );
65
66 // Generate a Source to playback the Buffers
67 alGenSources( 1, &uiSource );
68
69 // Create instance of WaveLoader class
70 pWaveLoader = new CWaves();
71 if ((pWaveLoader) && (SUCCEEDED(pWaveLoader->OpenWaveFile(ALFWaddMediaPath(TEST_WAVE_FILE), &WaveID))))
72 {
73 pWaveLoader->GetWaveSize(WaveID, &ulDataSize);
74 pWaveLoader->GetWaveFrequency(WaveID, &ulFrequency);
75 pWaveLoader->GetWaveALBufferFormat(WaveID, &alGetEnumValue, &ulFormat);
76
77 // Queue 250ms of audio data
78 pWaveLoader->GetWaveFormatExHeader(WaveID, &wfex);
79 ulBufferSize = wfex.nAvgBytesPerSec >> 2;
80
81 // IMPORTANT : The Buffer Size must be an exact multiple of the BlockAlignment ...
82 ulBufferSize -= (ulBufferSize % wfex.nBlockAlign);
83
84 if (ulFormat != 0)
85 {
86 pData = malloc(ulBufferSize);
87 if (pData)
88 {
89 // Set read position to start of audio data
90 pWaveLoader->SetWaveDataOffset(WaveID, 0);
91
92 // Fill all the Buffers with audio data from the wavefile
93 for (iLoop = 0; iLoop < 4; iLoop++)
94 {
95 if (SUCCEEDED(pWaveLoader->ReadWaveData(WaveID, pData, ulBufferSize, &ulBytesWritten)))
96 {
97 alBufferData(uiBuffers[iLoop], ulFormat, pData, ulBytesWritten, ulFrequency);
98 alSourceQueueBuffers(uiSource, 1, &uiBuffers[iLoop]);
99 }
100 }
101
102 // Start playing source
103 alSourcePlay(uiSource);
104
105 iTotalBuffersProcessed = 0;
106
107 while (!ALFWKeyPress())
108 {
109 Sleep( SERVICE_UPDATE_PERIOD );
110
111 // Request the number of OpenAL Buffers have been processed (played) on the Source
112 iBuffersProcessed = 0;
113 alGetSourcei(uiSource, AL_BUFFERS_PROCESSED, &iBuffersProcessed);
114
115 // Keep a running count of number of buffers processed (for logging purposes only)
116 iTotalBuffersProcessed += iBuffersProcessed;
117 ALFWprintf("Buffers Processed %d\r", iTotalBuffersProcessed);
118
119 // For each processed buffer, remove it from the Source Queue, read next chunk of audio
120 // data from disk, fill buffer with new data, and add it to the Source Queue
121 while (iBuffersProcessed)
122 {
123 // Remove the Buffer from the Queue. (uiBuffer contains the Buffer ID for the unqueued Buffer)
124 uiBuffer = 0;
125 alSourceUnqueueBuffers(uiSource, 1, &uiBuffer);
126
127 // Read more audio data (if there is any)
128 pWaveLoader->ReadWaveData(WaveID, pData, ulBufferSize, &ulBytesWritten);
129 if (ulBytesWritten)
130 {
131 // Copy audio data to Buffer
132 alBufferData(uiBuffer, ulFormat, pData, ulBytesWritten, ulFrequency);
133 // Queue Buffer on the Source
134 alSourceQueueBuffers(uiSource, 1, &uiBuffer);
135 }
136
137 iBuffersProcessed--;
138 }
139
140 // Check the status of the Source. If it is not playing, then playback was completed,
141 // or the Source was starved of audio data, and needs to be restarted.
142 alGetSourcei(uiSource, AL_SOURCE_STATE, &iState);
143 if (iState != AL_PLAYING)
144 {
145 // If there are Buffers in the Source Queue then the Source was starved of audio
146 // data, so needs to be restarted (because there is more audio data to play)
147 alGetSourcei(uiSource, AL_BUFFERS_QUEUED, &iQueuedBuffers);
148 if (iQueuedBuffers)
149 {
150 alSourcePlay(uiSource);
151 }
152 else
153 {
154 // Finished playing
155 break;
156 }
157 }
158 }
159
160 // Stop the Source and clear the Queue
161 alSourceStop(uiSource);
162 alSourcei(uiSource, AL_BUFFER, 0);
163
164 // Release temporary storage
165 free(pData);
166 pData = NULL;
167 }
168 else
169 {
170 ALFWprintf("Out of memory\n");
171 }
172 }
173 else
174 {
175 ALFWprintf("Unknown Audio Buffer format\n");
176 }
177
178 // Close Wave Handle
179 pWaveLoader->DeleteWaveFile(WaveID);
180 }
181 else
182 {
183 ALFWprintf("Failed to load %s\n", ALFWaddMediaPath(TEST_WAVE_FILE));
184 }
185
186 // Clean up buffers and sources
187 alDeleteSources( 1, &uiSource );
188 alDeleteBuffers( NUMBUFFERS, uiBuffers );
189
190 if (pWaveLoader)
191 delete pWaveLoader;
192
193 ALFWShutdownOpenAL();
194
195 ALFWShutdown();
196
197 return 0;
198}

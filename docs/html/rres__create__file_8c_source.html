238// Load a continuous data buffer from rresResourceChunkData struct
239static unsigned char *LoadDataBuffer(rresResourceChunkData data, unsigned int rawSize)
240{
241 unsigned char *buffer = (unsigned char *)RRES_CALLOC((data.propCount + 1)*sizeof(unsigned int) + rawSize, 1);
242
243 memcpy(buffer, &data.propCount, sizeof(unsigned int));
244 for (int i = 0; i < data.propCount; i++) memcpy(buffer + (i + 1)*sizeof(unsigned int), &data.props[i], sizeof(unsigned int));
245 memcpy(buffer + (data.propCount + 1)*sizeof(unsigned int), data.raw, rawSize);
246
247 return buffer;
248}

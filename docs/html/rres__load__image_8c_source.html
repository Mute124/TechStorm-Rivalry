44//------------------------------------------------------------------------------------
45// Program main entry point
46//------------------------------------------------------------------------------------
47int main(void)
48{
49 // Initialization
50 //--------------------------------------------------------------------------------------
51 const int screenWidth = 384;
52 const int screenHeight = 512;
53
54 InitWindow(screenWidth, screenHeight, "rres example - rres load image");
55
56 Texture2D texture = { 0 }; // Texture to load our image data
57
58 // Load central directory from .rres file (if available)
59 rresCentralDir dir = rresLoadCentralDirectory("resources.rres");
60
61 // Get resource id from original fileName (stored in centra directory)
62 unsigned int id = rresGetResourceId(dir, "resources/images/fudesumi.png");
63
64 // Setup password to load encrypted data (if required)
65 rresSetCipherPassword("password12345");
66
67 // Load resource chunk from file providing the id
68 rresResourceChunk chunk = rresLoadResourceChunk("resources.rres", id);
69
70 // Decompress/decipher resource data (if required)
71 int result = UnpackResourceChunk(&chunk);
72
73 if (result == RRES_SUCCESS) // Data decompressed/decrypted successfully
74 {
75 // Load image data from resource chunk
76 Image image = LoadImageFromResource(chunk);
77 texture = LoadTextureFromImage(image);
78 UnloadImage(image);
79 }
80
81 rresUnloadResourceChunk(chunk); // Unload resource chunk
82 rresUnloadCentralDirectory(dir); // Unload central directory, not required any more
83
84 SetTargetFPS(60); // Set our game to run at 60 frames-per-second
85 //--------------------------------------------------------------------------------------
86
87 // Main game loop
88 while (!WindowShouldClose()) // Detect window close button or ESC key
89 {
90 // Draw
91 //----------------------------------------------------------------------------------
92 BeginDrawing();
93
94 ClearBackground(RAYWHITE);
95 DrawTexture(texture, 0, 0, WHITE); // Draw loaded texture
96
97 EndDrawing();
98 //----------------------------------------------------------------------------------
99 }
100
101 // De-Initialization
102 //--------------------------------------------------------------------------------------
103 UnloadTexture(texture); // Unload texture (VRAM)
104 CloseWindow(); // Close window and OpenGL context
105 //--------------------------------------------------------------------------------------
106
107 return 0;
108}

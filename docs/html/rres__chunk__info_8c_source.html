41//------------------------------------------------------------------------------------
42// Program main entry point
43//------------------------------------------------------------------------------------
44int main(void)
45{
46 // Load central directory from .rres file (if available)
47 rresCentralDir dir = rresLoadCentralDirectory("resources.rres");
48
49 if (dir.count == 0) printf("WARNING: Central Directory not available\n"); // No central directory
50
51 // NOTE: With no CDIR we can still load the contained resources info,
52 // but we can't know the original input files that generated the resource chunks
53
54 // Load ALL resource chunks info from .rres file
55 unsigned int chunkCount = 0;
56 rresResourceChunkInfo *infos = rresLoadResourceChunkInfoAll("resources.rres", &chunkCount);
57
58 unsigned int prevId = 0;
59
60 // Display resource chunks info
61 // NOTE: Central Directory relates input files to rres resource chunks,
62 // some input files could generate multiple rres resource chunks (Font files)
63 for (unsigned int i = 0; i < chunkCount; i++)
64 {
65 for (unsigned int j = 0; j < dir.count; j++)
66 {
67 if ((infos[i].id == dir.entries[j].id) && (infos[i].id != prevId))
68 {
69 printf("Input File: %s\n", dir.entries[j].fileName);
70 printf("Resource(s) Offset: 0x%08x\n", dir.entries[j].offset);
71 prevId = dir.entries[j].id;
72 break;
73 }
74 }
75
76 printf(" Resource Chunk: %c%c%c%c\n", infos[i].type[0], infos[i].type[1], infos[i].type[2], infos[i].type[3]);
77 printf(" > id: 0x%08x\n", infos[i].id);
78 printf(" > compType: %s (%i)\n", GetCompressionName((int)infos[i].compType), (int)infos[i].compType);
79 printf(" > cipherType: %s (%i)\n", GetCipherName((int)infos[i].cipherType), (int)infos[i].cipherType);
80 printf(" > baseSize: %i\n", infos[i].baseSize);
81 printf(" > packedSize: %i (%i%%)\n", infos[i].packedSize, (int)(((float)infos[i].packedSize/infos[i].baseSize)*100)); // Get compression ratio
82 printf(" > nextOffset: %i\n", infos[i].nextOffset);
83 printf(" > CRC32: 0x%08x\n", infos[i].crc32);
84 }
85
86 // Free allocated memory for chunks info
87 RRES_FREE(infos);
88
89 return 0;
90}

92// Get compression name as a text string
93static const char *GetCompressionName(int compType)
94{
95 // Map compression type: NONE;DEFLATE;LZ4;QOI
96 if (compType == RRES_COMP_NONE) return "none";
97 else if (compType == RRES_COMP_DEFLATE) return "DEFLATE";
98 else if (compType == RRES_COMP_LZ4) return "LZ4";
99 else if (compType == RRES_COMP_QOI) return "QOI";
100 else return "Undefined";
101}
102
103// Get cipher name as a text string
104static const char *GetCipherName(int cipherType)
105{
106 // Map cipher type: NONE;AES;XCHACHA20
107 if (cipherType == RRES_CIPHER_NONE) return "none";
108 else if (cipherType == RRES_CIPHER_AES) return "AES-256";
109 else if (cipherType == RRES_CIPHER_XCHACHA20_POLY1305) return "XChaCha20";
110 else return "Undefined";
111}

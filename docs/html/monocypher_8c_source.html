101{
102 return
103 ((u32)s[0] << 0) |

109static u64 load64_le(const u8 s[8])
110{
111 return load32_le(s) | ((u64)load32_le(s+4) << 32);

114static void store32_le(u8 out[4], u32 in)
115{
116 out[0] = in & 0xff;
117 out[1] = (in >> 8) & 0xff;
118 out[2] = (in >> 16) & 0xff;
119 out[3] = (in >> 24) & 0xff;
120}

122static void store64_le(u8 out[8], u64 in)
123{
124 store32_le(out , (u32)in );
125 store32_le(out + 4, in >> 32);

144static int neq0(u64 diff)
145{
146 // constant time comparison to zero
147 // return diff != 0 ? -1 : 0
148 u64 half = (diff >> 32) | ((u32)diff);
149 return (1 & ((half - 1) >> 32)) - 1;
150}

178static void chacha20_rounds(u32 out[16], const u32 in[16])
179{
180 // The temporary variables make Chacha20 10% faster.
181 u32 t0 = in[ 0]; u32 t1 = in[ 1]; u32 t2 = in[ 2]; u32 t3 = in[ 3];
182 u32 t4 = in[ 4]; u32 t5 = in[ 5]; u32 t6 = in[ 6]; u32 t7 = in[ 7];
183 u32 t8 = in[ 8]; u32 t9 = in[ 9]; u32 t10 = in[10]; u32 t11 = in[11];
184 u32 t12 = in[12]; u32 t13 = in[13]; u32 t14 = in[14]; u32 t15 = in[15];
185
186 FOR (i, 0, 10) { // 20 rounds, 2 rounds per loop.
187 QUARTERROUND(t0, t4, t8 , t12); // column 0
188 QUARTERROUND(t1, t5, t9 , t13); // column 1
189 QUARTERROUND(t2, t6, t10, t14); // column 2
190 QUARTERROUND(t3, t7, t11, t15); // column 3
191 QUARTERROUND(t0, t5, t10, t15); // diagonal 0
192 QUARTERROUND(t1, t6, t11, t12); // diagonal 1
193 QUARTERROUND(t2, t7, t8 , t13); // diagonal 2
194 QUARTERROUND(t3, t4, t9 , t14); // diagonal 3
195 }
196 out[ 0] = t0; out[ 1] = t1; out[ 2] = t2; out[ 3] = t3;
197 out[ 4] = t4; out[ 5] = t5; out[ 6] = t6; out[ 7] = t7;
198 out[ 8] = t8; out[ 9] = t9; out[10] = t10; out[11] = t11;
199 out[12] = t12; out[13] = t13; out[14] = t14; out[15] = t15;
200}

204void crypto_chacha20_h(u8 out[32], const u8 key[32], const u8 in [16])
205{
206 u32 block[16];
207 load32_le_buf(block , chacha20_constant, 4);
208 load32_le_buf(block + 4, key , 8);
209 load32_le_buf(block + 12, in , 4);
210
211 chacha20_rounds(block, block);
212
213 // prevent reversal of the rounds by revealing only half of the buffer.
214 store32_le_buf(out , block , 4); // constant
215 store32_le_buf(out+16, block+12, 4); // counter and nonce
216 WIPE_BUFFER(block);
217}

219u64 crypto_chacha20_djb(u8 *cipher_text, const u8 *plain_text,
220 size_t text_size, const u8 key[32], const u8 nonce[8],
221 u64 ctr)
222{
223 u32 input[16];
224 load32_le_buf(input , chacha20_constant, 4);
225 load32_le_buf(input + 4, key , 8);
226 load32_le_buf(input + 14, nonce , 2);
227 input[12] = (u32) ctr;
228 input[13] = (u32)(ctr >> 32);
229
230 // Whole blocks
231 u32 pool[16];
232 size_t nb_blocks = text_size >> 6;
233 FOR (i, 0, nb_blocks) {
234 chacha20_rounds(pool, input);
235 if (plain_text != 0) {
236 FOR (j, 0, 16) {
237 u32 p = pool[j] + input[j];
238 store32_le(cipher_text, p ^ load32_le(plain_text));
239 cipher_text += 4;
240 plain_text += 4;
241 }
242 } else {
243 FOR (j, 0, 16) {
244 u32 p = pool[j] + input[j];
245 store32_le(cipher_text, p);
246 cipher_text += 4;
247 }
248 }
249 input[12]++;
250 if (input[12] == 0) {
251 input[13]++;
252 }
253 }
254 text_size &= 63;
255
256 // Last (incomplete) block
257 if (text_size > 0) {
258 if (plain_text == 0) {
259 plain_text = zero;
260 }
261 chacha20_rounds(pool, input);
262 u8 tmp[64];
263 FOR (i, 0, 16) {
264 store32_le(tmp + i*4, pool[i] + input[i]);
265 }
266 FOR (i, 0, text_size) {
267 cipher_text[i] = tmp[i] ^ plain_text[i];
268 }
269 WIPE_BUFFER(tmp);
270 }
271 ctr = input[12] + ((u64)input[13] << 32) + (text_size > 0);
272
273 WIPE_BUFFER(pool);
274 WIPE_BUFFER(input);
275 return ctr;
276}

303// h = (h + c) * r
304// preconditions:
305// ctx->h <= 4_ffffffff_ffffffff_ffffffff_ffffffff
306// ctx->r <= 0ffffffc_0ffffffc_0ffffffc_0fffffff
307// end <= 1
308// Postcondition:
309// ctx->h <= 4_ffffffff_ffffffff_ffffffff_ffffffff
310static void poly_block(crypto_poly1305_ctx *ctx, const u8 in[16], unsigned end)
311{
312 u32 s[4];
313 load32_le_buf(s, in, 4);
314
315 //- PROOF Poly1305
316 //-
317 //- # Inputs & preconditions
318 //- ctx->h[0] = u32()
319 //- ctx->h[1] = u32()
320 //- ctx->h[2] = u32()
321 //- ctx->h[3] = u32()
322 //- ctx->h[4] = u32(limit = 4)
323 //-
324 //- ctx->r[0] = u32(limit = 0x0fffffff)
325 //- ctx->r[1] = u32(limit = 0x0ffffffc)
326 //- ctx->r[2] = u32(limit = 0x0ffffffc)
327 //- ctx->r[3] = u32(limit = 0x0ffffffc)
328 //-
329 //- s[0] = u32()
330 //- s[1] = u32()
331 //- s[2] = u32()
332 //- s[3] = u32()
333 //-
334 //- end = unsigned(limit = 1)
335
336 // s = h + c, without carry propagation
337 const u64 s0 = ctx->h[0] + (u64)s[0]; // s0 <= 1_fffffffe
338 const u64 s1 = ctx->h[1] + (u64)s[1]; // s1 <= 1_fffffffe
339 const u64 s2 = ctx->h[2] + (u64)s[2]; // s2 <= 1_fffffffe
340 const u64 s3 = ctx->h[3] + (u64)s[3]; // s3 <= 1_fffffffe
341 const u32 s4 = ctx->h[4] + end; // s4 <= 5
342
343 // Local all the things!
344 const u32 r0 = ctx->r[0]; // r0 <= 0fffffff
345 const u32 r1 = ctx->r[1]; // r1 <= 0ffffffc
346 const u32 r2 = ctx->r[2]; // r2 <= 0ffffffc
347 const u32 r3 = ctx->r[3]; // r3 <= 0ffffffc
348 const u32 rr0 = (r0 >> 2) * 5; // rr0 <= 13fffffb // lose 2 bits...
349 const u32 rr1 = (r1 >> 2) + r1; // rr1 <= 13fffffb // rr1 == (r1 >> 2) * 5
350 const u32 rr2 = (r2 >> 2) + r2; // rr2 <= 13fffffb // rr1 == (r2 >> 2) * 5
351 const u32 rr3 = (r3 >> 2) + r3; // rr3 <= 13fffffb // rr1 == (r3 >> 2) * 5
352
353 // (h + c) * r, without carry propagation
354 const u64 x0 = s0*r0+ s1*rr3+ s2*rr2+ s3*rr1+ s4*rr0; // <= 97ffffe007fffff8
355 const u64 x1 = s0*r1+ s1*r0 + s2*rr3+ s3*rr2+ s4*rr1; // <= 8fffffe20ffffff6
356 const u64 x2 = s0*r2+ s1*r1 + s2*r0 + s3*rr3+ s4*rr2; // <= 87ffffe417fffff4
357 const u64 x3 = s0*r3+ s1*r2 + s2*r1 + s3*r0 + s4*rr3; // <= 7fffffe61ffffff2
358 const u32 x4 = s4 * (r0 & 3); // ...recover 2 bits // <= f
359
360 // partial reduction modulo 2^130 - 5
361 const u32 u5 = x4 + (x3 >> 32); // u5 <= 7ffffff5
362 const u64 u0 = (u5 >> 2) * 5 + (x0 & 0xffffffff);
363 const u64 u1 = (u0 >> 32) + (x1 & 0xffffffff) + (x0 >> 32);
364 const u64 u2 = (u1 >> 32) + (x2 & 0xffffffff) + (x1 >> 32);
365 const u64 u3 = (u2 >> 32) + (x3 & 0xffffffff) + (x2 >> 32);
366 const u64 u4 = (u3 >> 32) + (u5 & 3);
367
368 // Update the hash
369 ctx->h[0] = u0 & 0xffffffff; // u0 <= 1_9ffffff0
370 ctx->h[1] = u1 & 0xffffffff; // u1 <= 1_97ffffe0
371 ctx->h[2] = u2 & 0xffffffff; // u2 <= 1_8fffffe2
372 ctx->h[3] = u3 & 0xffffffff; // u3 <= 1_87ffffe4
373 ctx->h[4] = u4 & 0xffffffff; // u4 <= 4
374
375 //- # postconditions
376 //- ASSERT(ctx->h[4].limit() <= 4)
377 //- CQFD Poly1305
378}

380void crypto_poly1305_init(crypto_poly1305_ctx *ctx, const u8 key[32])
381{
382 ZERO(ctx->h, 5); // Initial hash is zero
383 ctx->c_idx = 0;
384 // load r and pad (r has some of its bits cleared)
385 load32_le_buf(ctx->r , key , 4);
386 load32_le_buf(ctx->pad, key+16, 4);
387 FOR (i, 0, 1) { ctx->r[i] &= 0x0fffffff; }
388 FOR (i, 1, 4) { ctx->r[i] &= 0x0ffffffc; }
389}

391void crypto_poly1305_update(crypto_poly1305_ctx *ctx,
392 const u8 *message, size_t message_size)
393{
394 // Align ourselves with block boundaries
395 size_t aligned = MIN(align(ctx->c_idx, 16), message_size);
396 FOR (i, 0, aligned) {
397 ctx->c[ctx->c_idx] = *message;
398 ctx->c_idx++;
399 message++;
400 message_size--;
401 }
402
403 // If block is complete, process it
404 if (ctx->c_idx == 16) {
405 poly_block(ctx, ctx->c, 1);
406 ctx->c_idx = 0;
407 }
408
409 // Process the message block by block
410 size_t nb_blocks = message_size >> 4;
411 FOR (i, 0, nb_blocks) {
412 poly_block(ctx, message, 1);
413 message += 16;
414 }
415 message_size &= 15;
416
417 // remaining bytes (we never complete a block here)
418 FOR (i, 0, message_size) {
419 ctx->c[ctx->c_idx] = message[i];
420 ctx->c_idx++;
421 }
422}

424void crypto_poly1305_final(crypto_poly1305_ctx *ctx, u8 mac[16])
425{
426 // Process the last block (if any)
427 // We move the final 1 according to remaining input length
428 // (this will add less than 2^130 to the last input block)
429 if (ctx->c_idx != 0) {
430 ZERO(ctx->c + ctx->c_idx, 16 - ctx->c_idx);
431 ctx->c[ctx->c_idx] = 1;
432 poly_block(ctx, ctx->c, 0);
433 }
434
435 // check if we should subtract 2^130-5 by performing the
436 // corresponding carry propagation.
437 u64 c = 5;
438 FOR (i, 0, 4) {
439 c += ctx->h[i];
440 c >>= 32;
441 }
442 c += ctx->h[4];
443 c = (c >> 2) * 5; // shift the carry back to the beginning
444 // c now indicates how many times we should subtract 2^130-5 (0 or 1)
445 FOR (i, 0, 4) {
446 c += (u64)ctx->h[i] + ctx->pad[i];
447 store32_le(mac + i*4, (u32)c);
448 c = c >> 32;
449 }
450 WIPE_CTX(ctx);
451}

453void crypto_poly1305(u8 mac[16], const u8 *message,
454 size_t message_size, const u8 key[32])
455{
456 crypto_poly1305_ctx ctx;
457 crypto_poly1305_init (&ctx, key);
458 crypto_poly1305_update(&ctx, message, message_size);
459 crypto_poly1305_final (&ctx, mac);
460}

569void crypto_blake2b_update(crypto_blake2b_ctx *ctx,
570 const u8 *message, size_t message_size)
571{
572 // Avoid undefined NULL pointer increments with empty messages
573 if (message_size == 0) {
574 return;
575 }
576
577 // Align with word boundaries
578 if ((ctx->input_idx & 7) != 0) {
579 size_t nb_bytes = MIN(align(ctx->input_idx, 8), message_size);
580 size_t word = ctx->input_idx >> 3;
581 size_t byte = ctx->input_idx & 7;
582 FOR (i, 0, nb_bytes) {
583 ctx->input[word] |= (u64)message[i] << ((byte + i) << 3);
584 }
585 ctx->input_idx += nb_bytes;
586 message += nb_bytes;
587 message_size -= nb_bytes;
588 }
589
590 // Align with block boundaries (faster than byte by byte)
591 if ((ctx->input_idx & 127) != 0) {
592 size_t nb_words = MIN(align(ctx->input_idx, 128), message_size) >> 3;
593 load64_le_buf(ctx->input + (ctx->input_idx >> 3), message, nb_words);
594 ctx->input_idx += nb_words << 3;
595 message += nb_words << 3;
596 message_size -= nb_words << 3;
597 }
598
599 // Process block by block
600 size_t nb_blocks = message_size >> 7;
601 FOR (i, 0, nb_blocks) {
602 if (ctx->input_idx == 128) {
603 blake2b_compress(ctx, 0);
604 }
605 load64_le_buf(ctx->input, message, 16);
606 message += 128;
607 ctx->input_idx = 128;
608 }
609 message_size &= 127;
610
611 if (message_size != 0) {
612 // Compress block & flush input buffer as needed
613 if (ctx->input_idx == 128) {
614 blake2b_compress(ctx, 0);
615 ctx->input_idx = 0;
616 }
617 if (ctx->input_idx == 0) {
618 ZERO(ctx->input, 16);
619 }
620 // Fill remaining words (faster than byte by byte)
621 size_t nb_words = message_size >> 3;
622 load64_le_buf(ctx->input, message, nb_words);
623 ctx->input_idx += nb_words << 3;
624 message += nb_words << 3;
625 message_size -= nb_words << 3;
626
627 // Fill remaining bytes
628 FOR (i, 0, message_size) {
629 size_t word = ctx->input_idx >> 3;
630 size_t byte = ctx->input_idx & 7;
631 ctx->input[word] |= (u64)message[i] << (byte << 3);
632 ctx->input_idx++;
633 }
634 }
635}

637void crypto_blake2b_final(crypto_blake2b_ctx *ctx, u8 *hash)
638{
639 blake2b_compress(ctx, 1); // compress the last block
640 size_t hash_size = MIN(ctx->hash_size, 64);
641 size_t nb_words = hash_size >> 3;
642 store64_le_buf(hash, ctx->hash, nb_words);
643 FOR (i, nb_words << 3, hash_size) {
644 hash[i] = (ctx->hash[i >> 3] >> (8 * (i & 7))) & 0xff;
645 }
646 WIPE_CTX(ctx);
647}

692// Hash with a virtually unlimited digest size.
693// Doesn't extract more entropy than the base hash function.
694// Mainly used for filling a whole kilobyte block with pseudo-random bytes.
695// (One could use a stream cipher with a seed hash as the key, but
696// this would introduce another dependency —and point of failure.)
697static void extended_hash(u8 *digest, u32 digest_size,
698 const u8 *input , u32 input_size)
699{
700 crypto_blake2b_ctx ctx;
701 crypto_blake2b_init (&ctx, MIN(digest_size, 64));
702 blake_update_32 (&ctx, digest_size);
703 crypto_blake2b_update(&ctx, input, input_size);
704 crypto_blake2b_final (&ctx, digest);
705
706 if (digest_size > 64) {
707 // the conversion to u64 avoids integer overflow on
708 // ludicrously big hash sizes.
709 u32 r = (u32)(((u64)digest_size + 31) >> 5) - 2;
710 u32 i = 1;
711 u32 in = 0;
712 u32 out = 32;
713 while (i < r) {
714 // Input and output overlap. This is intentional
715 crypto_blake2b(digest + out, 64, digest + in, 64);
716 i += 1;
717 in += 32;
718 out += 32;
719 }
720 crypto_blake2b(digest + out, digest_size - (32 * r), digest + in , 64);
721 }
722}

725#define G(a, b, c, d) \
726 a += b + 2 * LSB(a) * LSB(b); d ^= a; d = rotr64(d, 32); \
727 c += d + 2 * LSB(c) * LSB(d); b ^= c; b = rotr64(b, 24); \
728 a += b + 2 * LSB(a) * LSB(b); d ^= a; d = rotr64(d, 16); \

730#define ROUND(v0, v1, v2, v3, v4, v5, v6, v7, \
731 v8, v9, v10, v11, v12, v13, v14, v15) \
732 G(v0, v4, v8, v12); G(v1, v5, v9, v13); \
733 G(v2, v6, v10, v14); G(v3, v7, v11, v15); \
734 G(v0, v5, v10, v15); G(v1, v6, v11, v12); \
735 G(v2, v7, v8, v13); G(v3, v4, v9, v14)

737// Core of the compression function G. Computes Z from R in place.
738static void g_rounds(blk *b)
739{
740 // column rounds (work_block = Q)
741 for (int i = 0; i < 128; i += 16) {
742 ROUND(b->a[i ], b->a[i+ 1], b->a[i+ 2], b->a[i+ 3],
743 b->a[i+ 4], b->a[i+ 5], b->a[i+ 6], b->a[i+ 7],
744 b->a[i+ 8], b->a[i+ 9], b->a[i+10], b->a[i+11],
745 b->a[i+12], b->a[i+13], b->a[i+14], b->a[i+15]);
746 }
747 // row rounds (b = Z)
748 for (int i = 0; i < 16; i += 2) {
749 ROUND(b->a[i ], b->a[i+ 1], b->a[i+ 16], b->a[i+ 17],
750 b->a[i+32], b->a[i+33], b->a[i+ 48], b->a[i+ 49],
751 b->a[i+64], b->a[i+65], b->a[i+ 80], b->a[i+ 81],
752 b->a[i+96], b->a[i+97], b->a[i+112], b->a[i+113]);
753 }

758void crypto_argon2(u8 *hash, u32 hash_size, void *work_area,
759 crypto_argon2_config config,
760 crypto_argon2_inputs inputs,
761 crypto_argon2_extras extras)
762{
763 const u32 segment_size = config.nb_blocks / config.nb_lanes / 4;
764 const u32 lane_size = segment_size * 4;
765 const u32 nb_blocks = lane_size * config.nb_lanes; // rounding down
766
767 // work area seen as blocks (must be suitably aligned)
768 blk *blocks = (blk*)work_area;
769 {
770 u8 initial_hash[72]; // 64 bytes plus 2 words for future hashes
771 crypto_blake2b_ctx ctx;
772 crypto_blake2b_init (&ctx, 64);
773 blake_update_32 (&ctx, config.nb_lanes ); // p: number of "threads"
774 blake_update_32 (&ctx, hash_size);
775 blake_update_32 (&ctx, config.nb_blocks);
776 blake_update_32 (&ctx, config.nb_passes);
777 blake_update_32 (&ctx, 0x13); // v: version number
778 blake_update_32 (&ctx, config.algorithm); // y: Argon2i, Argon2d...
779 blake_update_32_buf (&ctx, inputs.pass, inputs.pass_size);
780 blake_update_32_buf (&ctx, inputs.salt, inputs.salt_size);
781 blake_update_32_buf (&ctx, extras.key, extras.key_size);
782 blake_update_32_buf (&ctx, extras.ad, extras.ad_size);
783 crypto_blake2b_final(&ctx, initial_hash); // fill 64 first bytes only
784
785 // fill first 2 blocks of each lane
786 u8 hash_area[1024];
787 FOR_T(u32, l, 0, config.nb_lanes) {
788 FOR_T(u32, i, 0, 2) {
789 store32_le(initial_hash + 64, i); // first additional word
790 store32_le(initial_hash + 68, l); // second additional word
791 extended_hash(hash_area, 1024, initial_hash, 72);
792 load64_le_buf(blocks[l * lane_size + i].a, hash_area, 128);
793 }
794 }
795
796 WIPE_BUFFER(initial_hash);
797 WIPE_BUFFER(hash_area);
798 }
799
800 // Argon2i and Argon2id start with constant time indexing
801 int constant_time = config.algorithm != CRYPTO_ARGON2_D;
802
803 // Fill (and re-fill) the rest of the blocks
804 //
805 // Note: even though each segment within the same slice can be
806 // computed in parallel, (one thread per lane), we are computing
807 // them sequentially, because Monocypher doesn't support threads.
808 //
809 // Yet optimal performance (and therefore security) requires one
810 // thread per lane. The only reason Monocypher supports multiple
811 // lanes is compatibility.
812 blk tmp;
813 FOR_T(u32, pass, 0, config.nb_passes) {
814 FOR_T(u32, slice, 0, 4) {
815 // On the first slice of the first pass,
816 // blocks 0 and 1 are already filled, hence pass_offset.
817 u32 pass_offset = pass == 0 && slice == 0 ? 2 : 0;
818 u32 slice_offset = slice * segment_size;
819
820 // Argon2id switches back to non-constant time indexing
821 // after the first two slices of the first pass
822 if (slice == 2 && config.algorithm == CRYPTO_ARGON2_ID) {
823 constant_time = 0;
824 }
825
826 // Each iteration of the following loop may be performed in
827 // a separate thread. All segments must be fully completed
828 // before we start filling the next slice.
829 FOR_T(u32, segment, 0, config.nb_lanes) {
830 blk index_block;
831 u32 index_ctr = 1;
832 FOR_T (u32, block, pass_offset, segment_size) {
833 // Current and previous blocks
834 u32 lane_offset = segment * lane_size;
835 blk *segment_start = blocks + lane_offset + slice_offset;
836 blk *current = segment_start + block;
837 blk *previous =
838 block == 0 && slice_offset == 0
839 ? segment_start + lane_size - 1
840 : segment_start + block - 1;
841
842 u64 index_seed;
843 if (constant_time) {
844 if (block == pass_offset || (block % 128) == 0) {
845 // Fill or refresh deterministic indices block
846
847 // seed the beginning of the block...
848 ZERO(index_block.a, 128);
849 index_block.a[0] = pass;
850 index_block.a[1] = segment;
851 index_block.a[2] = slice;
852 index_block.a[3] = nb_blocks;
853 index_block.a[4] = config.nb_passes;
854 index_block.a[5] = config.algorithm;
855 index_block.a[6] = index_ctr;
856 index_ctr++;
857
858 // ... then shuffle it
859 copy_block(&tmp, &index_block);
860 g_rounds (&index_block);
861 xor_block (&index_block, &tmp);
862 copy_block(&tmp, &index_block);
863 g_rounds (&index_block);
864 xor_block (&index_block, &tmp);
865 }
866 index_seed = index_block.a[block % 128];
867 } else {
868 index_seed = previous->a[0];
869 }
870
871 // Establish the reference set. *Approximately* comprises:
872 // - The last 3 slices (if they exist yet)
873 // - The already constructed blocks in the current segment
874 u32 next_slice = ((slice + 1) % 4) * segment_size;
875 u32 window_start = pass == 0 ? 0 : next_slice;
876 u32 nb_segments = pass == 0 ? slice : 3;
877 u32 window_size = nb_segments * segment_size + block - 1;
878
879 // Find reference block
880 u64 j1 = index_seed & 0xffffffff; // block selector
881 u64 j2 = index_seed >> 32; // lane selector
882 u64 x = (j1 * j1) >> 32;
883 u64 y = (window_size * x) >> 32;
884 u64 z = (window_size - 1) - y;
885 u64 ref = (window_start + z) % lane_size;
886 u32 index = (j2%config.nb_lanes)*lane_size + (u32)ref;
887 blk *reference = blocks + index;
888
889 // Shuffle the previous & reference block
890 // into the current block
891 copy_block(&tmp, previous);
892 xor_block (&tmp, reference);
893 if (pass == 0) { copy_block(current, &tmp); }
894 else { xor_block (current, &tmp); }
895 g_rounds (&tmp);
896 xor_block (current, &tmp);
897 }
898 }
899 }
900 }
901
902 // Wipe temporary block
903 volatile u64* p = tmp.a;
904 ZERO(p, 128);
905
906 // XOR last blocks of each lane
907 blk *last_block = blocks + lane_size - 1;
908 FOR_T (u32, lane, 1, config.nb_lanes) {
909 blk *next_block = last_block + lane_size;
910 xor_block(next_block, last_block);
911 last_block = next_block;
912 }
913
914 // Serialize last block
915 u8 final_block[1024];
916 store64_le_buf(final_block, last_block->a, 128);
917
918 // Wipe work area
919 p = (u64*)work_area;
920 ZERO(p, 128 * nb_blocks);
921
922 // Hash the very last block with H' into the output hash
923 extended_hash(hash, hash_size, final_block, 1024);
924 WIPE_BUFFER(final_block);
925}

1002// Signed carry propagation
1003// ------------------------
1004//
1005// Let t be a number. It can be uniquely decomposed thus:
1006//
1007// t = h*2^26 + l
1008// such that -2^25 <= l < 2^25
1009//
1010// Let c = (t + 2^25) / 2^26 (rounded down)
1011// c = (h*2^26 + l + 2^25) / 2^26 (rounded down)
1012// c = h + (l + 2^25) / 2^26 (rounded down)
1013// c = h (exactly)
1014// Because 0 <= l + 2^25 < 2^26
1015//
1016// Let u = t - c*2^26
1017// u = h*2^26 + l - h*2^26
1018// u = l
1019// Therefore, -2^25 <= u < 2^25
1020//
1021// Additionally, if |t| < x, then |h| < x/2^26 (rounded down)
1022//
1023// Notations:
1024// - In C, 1<<25 means 2^25.
1025// - In C, x>>25 means floor(x / (2^25)).
1026// - All of the above applies with 25 & 24 as well as 26 & 25.
1027//
1028//
1029// Note on negative right shifts
1030// -----------------------------
1031//
1032// In C, x >> n, where x is a negative integer, is implementation
1033// defined. In practice, all platforms do arithmetic shift, which is
1034// equivalent to division by 2^26, rounded down. Some compilers, like
1035// GCC, even guarantee it.
1036//
1037// If we ever stumble upon a platform that does not propagate the sign
1038// bit (we won't), visible failures will show at the slightest test, and
1039// the signed shifts can be replaced by the following:
1040//
1041// typedef struct { i64 x:39; } s25;
1042// typedef struct { i64 x:38; } s26;
1043// i64 shift25(i64 x) { s25 s; s.x = ((u64)x)>>25; return s.x; }
1044// i64 shift26(i64 x) { s26 s; s.x = ((u64)x)>>26; return s.x; }
1045//
1046// Current compilers cannot optimise this, causing a 30% drop in
1047// performance. Fairly expensive for something that never happens.
1048//
1049//
1050// Precondition
1051// ------------
1052//
1053// |t0| < 2^63
1054// |t1|..|t9| < 2^62
1055//
1056// Algorithm
1057// ---------
1058// c = t0 + 2^25 / 2^26 -- |c| <= 2^36
1059// t0 -= c * 2^26 -- |t0| <= 2^25
1060// t1 += c -- |t1| <= 2^63
1061//
1062// c = t4 + 2^25 / 2^26 -- |c| <= 2^36
1063// t4 -= c * 2^26 -- |t4| <= 2^25
1064// t5 += c -- |t5| <= 2^63
1065//
1066// c = t1 + 2^24 / 2^25 -- |c| <= 2^38
1067// t1 -= c * 2^25 -- |t1| <= 2^24
1068// t2 += c -- |t2| <= 2^63
1069//
1070// c = t5 + 2^24 / 2^25 -- |c| <= 2^38
1071// t5 -= c * 2^25 -- |t5| <= 2^24
1072// t6 += c -- |t6| <= 2^63
1073//
1074// c = t2 + 2^25 / 2^26 -- |c| <= 2^37
1075// t2 -= c * 2^26 -- |t2| <= 2^25 < 1.1 * 2^25 (final t2)
1076// t3 += c -- |t3| <= 2^63
1077//
1078// c = t6 + 2^25 / 2^26 -- |c| <= 2^37
1079// t6 -= c * 2^26 -- |t6| <= 2^25 < 1.1 * 2^25 (final t6)
1080// t7 += c -- |t7| <= 2^63
1081//
1082// c = t3 + 2^24 / 2^25 -- |c| <= 2^38
1083// t3 -= c * 2^25 -- |t3| <= 2^24 < 1.1 * 2^24 (final t3)
1084// t4 += c -- |t4| <= 2^25 + 2^38 < 2^39
1085//
1086// c = t7 + 2^24 / 2^25 -- |c| <= 2^38
1087// t7 -= c * 2^25 -- |t7| <= 2^24 < 1.1 * 2^24 (final t7)
1088// t8 += c -- |t8| <= 2^63
1089//
1090// c = t4 + 2^25 / 2^26 -- |c| <= 2^13
1091// t4 -= c * 2^26 -- |t4| <= 2^25 < 1.1 * 2^25 (final t4)
1092// t5 += c -- |t5| <= 2^24 + 2^13 < 1.1 * 2^24 (final t5)
1093//
1094// c = t8 + 2^25 / 2^26 -- |c| <= 2^37
1095// t8 -= c * 2^26 -- |t8| <= 2^25 < 1.1 * 2^25 (final t8)
1096// t9 += c -- |t9| <= 2^63
1097//
1098// c = t9 + 2^24 / 2^25 -- |c| <= 2^38
1099// t9 -= c * 2^25 -- |t9| <= 2^24 < 1.1 * 2^24 (final t9)

1101//
1102// c = t0 + 2^25 / 2^26 -- |c| <= 2^18
1103// t0 -= c * 2^26 -- |t0| <= 2^25 < 1.1 * 2^25 (final t0)
1104// t1 += c -- |t1| <= 2^24 + 2^18 < 1.1 * 2^24 (final t1)

1107// -------------

1127// Decodes a field element from a byte buffer.
1128// mask specifies how many bits we ignore.
1129// Traditionally we ignore 1. It's useful for EdDSA,
1130// which uses that bit to denote the sign of x.
1131// Elligator however uses positive representatives,
1132// which means ignoring 2 bits instead.
1133static void fe_frombytes_mask(fe h, const u8 s[32], unsigned nb_mask)
1134{
1135 u32 mask = 0xffffff >> nb_mask;
1136 i64 t0 = load32_le(s); // t0 < 2^32
1137 i64 t1 = load24_le(s + 4) << 6; // t1 < 2^30
1138 i64 t2 = load24_le(s + 7) << 5; // t2 < 2^29
1139 i64 t3 = load24_le(s + 10) << 3; // t3 < 2^27
1140 i64 t4 = load24_le(s + 13) << 2; // t4 < 2^26
1141 i64 t5 = load32_le(s + 16); // t5 < 2^32
1142 i64 t6 = load24_le(s + 20) << 7; // t6 < 2^31
1143 i64 t7 = load24_le(s + 23) << 5; // t7 < 2^29
1144 i64 t8 = load24_le(s + 26) << 4; // t8 < 2^28
1145 i64 t9 = (load24_le(s + 29) & mask) << 2; // t9 < 2^25
1146 FE_CARRY; // Carry precondition OK
1147}

1155// Precondition
1156// |h[0]|, |h[2]|, |h[4]|, |h[6]|, |h[8]| < 1.1 * 2^25
1157// |h[1]|, |h[3]|, |h[5]|, |h[7]|, |h[9]| < 1.1 * 2^24
1158//
1159// Therefore, |h| < 2^255-19
1160// There are two possibilities:
1161//
1162// - If h is positive, all we need to do is reduce its individual
1163// limbs down to their tight positive range.
1164// - If h is negative, we also need to add 2^255-19 to it.
1165// Or just remove 19 and chop off any excess bit.
1166static void fe_tobytes(u8 s[32], const fe h)
1167{
1168 i32 t[10];
1169 COPY(t, h, 10);
1170 i32 q = (19 * t[9] + (((i32) 1) << 24)) >> 25;
1171 // |t9| < 1.1 * 2^24
1172 // -1.1 * 2^24 < t9 < 1.1 * 2^24
1173 // -21 * 2^24 < 19 * t9 < 21 * 2^24
1174 // -2^29 < 19 * t9 + 2^24 < 2^29
1175 // -2^29 / 2^25 < (19 * t9 + 2^24) / 2^25 < 2^29 / 2^25
1176 // -16 < (19 * t9 + 2^24) / 2^25 < 16
1177 FOR (i, 0, 5) {
1178 q += t[2*i ]; q >>= 26; // q = 0 or -1
1179 q += t[2*i+1]; q >>= 25; // q = 0 or -1
1180 }
1181 // q = 0 iff h >= 0
1182 // q = -1 iff h < 0
1183 // Adding q * 19 to h reduces h to its proper range.
1184 q *= 19; // Shift carry back to the beginning
1185 FOR (i, 0, 5) {
1186 t[i*2 ] += q; q = t[i*2 ] >> 26; t[i*2 ] -= q * ((i32)1 << 26);
1187 t[i*2+1] += q; q = t[i*2+1] >> 25; t[i*2+1] -= q * ((i32)1 << 25);
1188 }
1189 // h is now fully reduced, and q represents the excess bit.
1190
1191 store32_le(s + 0, ((u32)t[0] >> 0) | ((u32)t[1] << 26));
1192 store32_le(s + 4, ((u32)t[1] >> 6) | ((u32)t[2] << 19));
1193 store32_le(s + 8, ((u32)t[2] >> 13) | ((u32)t[3] << 13));
1194 store32_le(s + 12, ((u32)t[3] >> 19) | ((u32)t[4] << 6));
1195 store32_le(s + 16, ((u32)t[5] >> 0) | ((u32)t[6] << 25));
1196 store32_le(s + 20, ((u32)t[6] >> 7) | ((u32)t[7] << 19));
1197 store32_le(s + 24, ((u32)t[7] >> 13) | ((u32)t[8] << 12));
1198 store32_le(s + 28, ((u32)t[8] >> 20) | ((u32)t[9] << 6));
1199
1200 WIPE_BUFFER(t);
1201}

1355// Inverse square root.
1356// Returns true if x is a square, false otherwise.
1357// After the call:
1358// isr = sqrt(1/x) if x is a non-zero square.
1359// isr = sqrt(sqrt(-1)/x) if x is not a square.
1360// isr = 0 if x is zero.
1361// We do not guarantee the sign of the square root.
1362//
1363// Notes:
1364// Let quartic = x^((p-1)/4)
1365//
1366// x^((p-1)/2) = chi(x)
1367// quartic^2 = chi(x)
1368// quartic = sqrt(chi(x))
1369// quartic = 1 or -1 or sqrt(-1) or -sqrt(-1)
1370//
1371// Note that x is a square if quartic is 1 or -1
1372// There are 4 cases to consider:
1373//
1374// if quartic = 1 (x is a square)
1375// then x^((p-1)/4) = 1
1376// x^((p-5)/4) * x = 1
1377// x^((p-5)/4) = 1/x
1378// x^((p-5)/8) = sqrt(1/x) or -sqrt(1/x)
1379//
1380// if quartic = -1 (x is a square)
1381// then x^((p-1)/4) = -1
1382// x^((p-5)/4) * x = -1
1383// x^((p-5)/4) = -1/x
1384// x^((p-5)/8) = sqrt(-1) / sqrt(x)
1385// x^((p-5)/8) * sqrt(-1) = sqrt(-1)^2 / sqrt(x)
1386// x^((p-5)/8) * sqrt(-1) = -1/sqrt(x)
1387// x^((p-5)/8) * sqrt(-1) = -sqrt(1/x) or sqrt(1/x)
1388//
1389// if quartic = sqrt(-1) (x is not a square)
1390// then x^((p-1)/4) = sqrt(-1)
1391// x^((p-5)/4) * x = sqrt(-1)
1392// x^((p-5)/4) = sqrt(-1)/x
1393// x^((p-5)/8) = sqrt(sqrt(-1)/x) or -sqrt(sqrt(-1)/x)
1394//
1395// Note that the product of two non-squares is always a square:
1396// For any non-squares a and b, chi(a) = -1 and chi(b) = -1.
1397// Since chi(x) = x^((p-1)/2), chi(a)*chi(b) = chi(a*b) = 1.
1398// Therefore a*b is a square.
1399//
1400// Since sqrt(-1) and x are both non-squares, their product is a
1401// square, and we can compute their square root.
1402//
1403// if quartic = -sqrt(-1) (x is not a square)
1404// then x^((p-1)/4) = -sqrt(-1)
1405// x^((p-5)/4) * x = -sqrt(-1)
1406// x^((p-5)/4) = -sqrt(-1)/x
1407// x^((p-5)/8) = sqrt(-sqrt(-1)/x)
1408// x^((p-5)/8) = sqrt( sqrt(-1)/x) * sqrt(-1)
1409// x^((p-5)/8) * sqrt(-1) = sqrt( sqrt(-1)/x) * sqrt(-1)^2
1410// x^((p-5)/8) * sqrt(-1) = sqrt( sqrt(-1)/x) * -1
1411// x^((p-5)/8) * sqrt(-1) = -sqrt(sqrt(-1)/x) or sqrt(sqrt(-1)/x)
1412static int invsqrt(fe isr, const fe x)
1413{
1414 fe t0, t1, t2;
1415
1416 // t0 = x^((p-5)/8)
1417 // Can be achieved with a simple double & add ladder,
1418 // but it would be slower.
1419 fe_sq(t0, x);
1420 fe_sq(t1,t0); fe_sq(t1, t1); fe_mul(t1, x, t1);
1421 fe_mul(t0, t0, t1);
1422 fe_sq(t0, t0); fe_mul(t0, t1, t0);
1423 fe_sq(t1, t0); FOR (i, 1, 5) fe_sq(t1, t1); fe_mul(t0, t1, t0);
1424 fe_sq(t1, t0); FOR (i, 1, 10) fe_sq(t1, t1); fe_mul(t1, t1, t0);
1425 fe_sq(t2, t1); FOR (i, 1, 20) fe_sq(t2, t2); fe_mul(t1, t2, t1);
1426 fe_sq(t1, t1); FOR (i, 1, 10) fe_sq(t1, t1); fe_mul(t0, t1, t0);
1427 fe_sq(t1, t0); FOR (i, 1, 50) fe_sq(t1, t1); fe_mul(t1, t1, t0);
1428 fe_sq(t2, t1); FOR (i, 1, 100) fe_sq(t2, t2); fe_mul(t1, t2, t1);
1429 fe_sq(t1, t1); FOR (i, 1, 50) fe_sq(t1, t1); fe_mul(t0, t1, t0);
1430 fe_sq(t0, t0); FOR (i, 1, 2) fe_sq(t0, t0); fe_mul(t0, t0, x);
1431
1432 // quartic = x^((p-1)/4)
1433 i32 *quartic = t1;
1434 fe_sq (quartic, t0);
1435 fe_mul(quartic, quartic, x);
1436
1437 i32 *check = t2;
1438 fe_0 (check); int z0 = fe_isequal(x , check);
1439 fe_1 (check); int p1 = fe_isequal(quartic, check);
1440 fe_neg(check, check ); int m1 = fe_isequal(quartic, check);
1441 fe_neg(check, sqrtm1); int ms = fe_isequal(quartic, check);
1442
1443 // if quartic == -1 or sqrt(-1)
1444 // then isr = x^((p-1)/4) * sqrt(-1)
1445 // else isr = x^((p-1)/4)
1446 fe_mul(isr, t0, sqrtm1);
1447 fe_ccopy(isr, t0, 1 - (m1 | ms));
1448
1449 WIPE_BUFFER(t0);
1450 WIPE_BUFFER(t1);
1451 WIPE_BUFFER(t2);
1452 return p1 | m1 | z0;
1453}

1455// Inverse in terms of inverse square root.
1456// Requires two additional squarings to get rid of the sign.
1457//
1458// 1/x = x * (+invsqrt(x^2))^2
1459// = x * (-invsqrt(x^2))^2
1460//
1461// A fully optimised exponentiation by p-1 would save 6 field
1462// multiplications, but it would require more code.
1463static void fe_invert(fe out, const fe x)
1464{
1465 fe tmp;
1466 fe_sq(tmp, x);
1467 invsqrt(tmp, tmp);
1468 fe_sq(tmp, tmp);
1469 fe_mul(out, tmp, x);
1470 WIPE_BUFFER(tmp);
1471}

1492static void scalarmult(u8 q[32], const u8 scalar[32], const u8 p[32],
1493 int nb_bits)
1494{
1495 // computes the scalar product
1496 fe x1;
1497 fe_frombytes(x1, p);
1498
1499 // computes the actual scalar product (the result is in x2 and z2)
1500 fe x2, z2, x3, z3, t0, t1;
1501 // Montgomery ladder
1502 // In projective coordinates, to avoid divisions: x = X / Z
1503 // We don't care about the y coordinate, it's only 1 bit of information
1504 fe_1(x2); fe_0(z2); // "zero" point
1505 fe_copy(x3, x1); fe_1(z3); // "one" point
1506 int swap = 0;
1507 for (int pos = nb_bits-1; pos >= 0; --pos) {
1508 // constant time conditional swap before ladder step
1509 int b = scalar_bit(scalar, pos);
1510 swap ^= b; // xor trick avoids swapping at the end of the loop
1511 fe_cswap(x2, x3, swap);
1512 fe_cswap(z2, z3, swap);
1513 swap = b; // anticipates one last swap after the loop
1514
1515 // Montgomery ladder step: replaces (P2, P3) by (P2*2, P2+P3)
1516 // with differential addition
1517 fe_sub(t0, x3, z3);
1518 fe_sub(t1, x2, z2);
1519 fe_add(x2, x2, z2);
1520 fe_add(z2, x3, z3);
1521 fe_mul(z3, t0, x2);
1522 fe_mul(z2, z2, t1);
1523 fe_sq (t0, t1 );
1524 fe_sq (t1, x2 );
1525 fe_add(x3, z3, z2);
1526 fe_sub(z2, z3, z2);
1527 fe_mul(x2, t1, t0);
1528 fe_sub(t1, t1, t0);
1529 fe_sq (z2, z2 );
1530 fe_mul_small(z3, t1, 121666);
1531 fe_sq (x3, x3 );
1532 fe_add(t0, t0, z3);
1533 fe_mul(z3, x1, z2);
1534 fe_mul(z2, t1, t0);
1535 }
1536 // last swap is necessary to compensate for the xor trick
1537 // Note: after this swap, P3 == P2 + P1.
1538 fe_cswap(x2, x3, swap);
1539 fe_cswap(z2, z3, swap);
1540
1541 // normalises the coordinates: x == X / Z
1542 fe_invert(z2, z2);
1543 fe_mul(x2, x2, z2);
1544 fe_tobytes(q, x2);
1545
1546 WIPE_BUFFER(x1);
1547 WIPE_BUFFER(x2); WIPE_BUFFER(z2); WIPE_BUFFER(t0);
1548 WIPE_BUFFER(x3); WIPE_BUFFER(z3); WIPE_BUFFER(t1);
1549}

1577// p = a*b + p
1578static void multiply(u32 p[16], const u32 a[8], const u32 b[8])
1579{
1580 FOR (i, 0, 8) {
1581 u64 carry = 0;
1582 FOR (j, 0, 8) {
1583 carry += p[i+j] + (u64)a[i] * b[j];
1584 p[i+j] = (u32)carry;
1585 carry >>= 32;
1586 }
1587 p[i+8] = (u32)carry;
1588 }
1589}

1591static int is_above_l(const u32 x[8])
1592{
1593 // We work with L directly, in a 2's complement encoding
1594 // (-L == ~L + 1)
1595 u64 carry = 1;
1596 FOR (i, 0, 8) {
1597 carry += (u64)x[i] + (~L[i] & 0xffffffff);
1598 carry >>= 32;
1599 }
1600 return (int)carry; // carry is either 0 or 1
1601}

1603// Final reduction modulo L, by conditionally removing L.
1604// if x < l , then r = x
1605// if l <= x 2*l, then r = x-l
1606// otherwise the result will be wrong
1607static void remove_l(u32 r[8], const u32 x[8])
1608{
1609 u64 carry = (u64)is_above_l(x);
1610 u32 mask = ~(u32)carry + 1; // carry == 0 or 1
1611 FOR (i, 0, 8) {
1612 carry += (u64)x[i] + (~L[i] & mask);
1613 r[i] = (u32)carry;
1614 carry >>= 32;
1615 }
1616}

1618// Full reduction modulo L (Barrett reduction)
1619static void mod_l(u8 reduced[32], const u32 x[16])
1620{
1621 static const u32 r[9] = {
1622 0x0a2c131b,0xed9ce5a3,0x086329a7,0x2106215d,
1623 0xffffffeb,0xffffffff,0xffffffff,0xffffffff,0xf,
1624 };
1625 // xr = x * r
1626 u32 xr[25] = {0};
1627 FOR (i, 0, 9) {
1628 u64 carry = 0;
1629 FOR (j, 0, 16) {
1630 carry += xr[i+j] + (u64)r[i] * x[j];
1631 xr[i+j] = (u32)carry;
1632 carry >>= 32;
1633 }
1634 xr[i+16] = (u32)carry;
1635 }
1636 // xr = floor(xr / 2^512) * L
1637 // Since the result is guaranteed to be below 2*L,
1638 // it is enough to only compute the first 256 bits.
1639 // The division is performed by saying xr[i+16]. (16 * 32 = 512)
1640 ZERO(xr, 8);
1641 FOR (i, 0, 8) {
1642 u64 carry = 0;
1643 FOR (j, 0, 8-i) {
1644 carry += xr[i+j] + (u64)xr[i+16] * L[j];
1645 xr[i+j] = (u32)carry;
1646 carry >>= 32;
1647 }
1648 }
1649 // xr = x - xr
1650 u64 carry = 1;
1651 FOR (i, 0, 8) {
1652 carry += (u64)x[i] + (~xr[i] & 0xffffffff);
1653 xr[i] = (u32)carry;
1654 carry >>= 32;
1655 }
1656 // Final reduction modulo L (conditional subtraction)
1657 remove_l(xr, xr);
1658 store32_le_buf(reduced, xr, 8);
1659
1660 WIPE_BUFFER(xr);
1661}

1664{
1665 u32 x[16];
1666 load32_le_buf(x, expanded, 16);
1667 mod_l(reduced, x);
1668 WIPE_BUFFER(x);
1669}

1720// h = -s, where s is a point encoded in 32 bytes
1721//
1722// Variable time! Inputs must not be secret!
1723// => Use only to *check* signatures.
1724//
1725// From the specifications:
1726// The encoding of s contains y and the sign of x
1727// x = sqrt((y^2 - 1) / (d*y^2 + 1))
1728// In extended coordinates:
1729// X = x, Y = y, Z = 1, T = x*y
1730//
1731// Note that num * den is a square iff num / den is a square
1732// If num * den is not a square, the point was not on the curve.
1733// From the above:
1734// Let num = y^2 - 1
1735// Let den = d*y^2 + 1
1736// x = sqrt((y^2 - 1) / (d*y^2 + 1))
1737// x = sqrt(num / den)
1738// x = sqrt(num^2 / (num * den))
1739// x = num * sqrt(1 / (num * den))
1740//
1741// Therefore, we can just compute:
1742// num = y^2 - 1
1743// den = d*y^2 + 1
1744// isr = invsqrt(num * den) // abort if not square
1745// x = num * isr
1746// Finally, negate x if its sign is not as specified.
1747static int ge_frombytes_neg_vartime(ge *h, const u8 s[32])
1748{
1749 fe_frombytes(h->Y, s);
1750 fe_1(h->Z);
1751 fe_sq (h->T, h->Y); // t = y^2
1752 fe_mul(h->X, h->T, d ); // x = d*y^2
1753 fe_sub(h->T, h->T, h->Z); // t = y^2 - 1
1754 fe_add(h->X, h->X, h->Z); // x = d*y^2 + 1
1755 fe_mul(h->X, h->T, h->X); // x = (y^2 - 1) * (d*y^2 + 1)
1756 int is_square = invsqrt(h->X, h->X);
1757 if (!is_square) {
1758 return -1; // Not on the curve, abort
1759 }
1760 fe_mul(h->X, h->T, h->X); // x = sqrt((y^2 - 1) / (d*y^2 + 1))
1761 if (fe_isodd(h->X) == (s[31] >> 7)) {
1762 fe_neg(h->X, h->X);
1763 }
1764 fe_mul(h->T, h->X, h->Y);
1765 return 0;
1766}

1914// Incremental sliding windows (left to right)
1915// Based on Roberto Maria Avanzi[2005]
1916typedef struct {
1917 i16 next_index; // position of the next signed digit
1918 i8 next_digit; // next signed digit (odd number below 2^window_width)
1919 u8 next_check; // point at which we must check for a new window
1920} slide_ctx;
1921
1922static void slide_init(slide_ctx *ctx, const u8 scalar[32])
1923{
1924 // scalar is guaranteed to be below L, either because we checked (s),
1925 // or because we reduced it modulo L (h_ram). L is under 2^253, so
1926 // so bits 253 to 255 are guaranteed to be zero. No need to test them.
1927 //
1928 // Note however that L is very close to 2^252, so bit 252 is almost
1929 // always zero. If we were to start at bit 251, the tests wouldn't
1930 // catch the off-by-one error (constructing one that does would be
1931 // prohibitively expensive).
1932 //
1933 // We should still check bit 252, though.
1934 int i = 252;
1935 while (i > 0 && scalar_bit(scalar, i) == 0) {
1936 i--;
1937 }
1938 ctx->next_check = (u8)(i + 1);
1939 ctx->next_index = -1;
1940 ctx->next_digit = -1;
1941}
1942
1943static int slide_step(slide_ctx *ctx, int width, int i, const u8 scalar[32])
1944{
1945 if (i == ctx->next_check) {
1946 if (scalar_bit(scalar, i) == scalar_bit(scalar, i - 1)) {
1947 ctx->next_check--;
1948 } else {
1949 // compute digit of next window
1950 int w = MIN(width, i + 1);
1951 int v = -(scalar_bit(scalar, i) << (w-1));
1952 FOR_T (int, j, 0, w-1) {
1953 v += scalar_bit(scalar, i-(w-1)+j) << j;
1954 }
1955 v += scalar_bit(scalar, i-w);
1956 int lsb = v & (~v + 1); // smallest bit of v
1957 int s = // log2(lsb)
1958 (((lsb & 0xAA) != 0) << 0) |
1959 (((lsb & 0xCC) != 0) << 1) |
1960 (((lsb & 0xF0) != 0) << 2);
1961 ctx->next_index = (i16)(i-(w-1)+s);
1962 ctx->next_digit = (i8) (v >> s );
1963 ctx->next_check -= (u8) w;
1964 }
1965 }
1966 return i == ctx->next_index ? ctx->next_digit: 0;
1967}

1969#define P_W_WIDTH 3 // Affects the size of the stack
1970#define B_W_WIDTH 5 // Affects the size of the binary
1971#define P_W_SIZE (1<<(P_W_WIDTH-2))

1973int crypto_eddsa_check_equation(const u8 signature[64], const u8 public_key[32],
1974 const u8 h[32])
1975{
1976 ge minus_A; // -public_key
1977 ge minus_R; // -first_half_of_signature
1978 const u8 *s = signature + 32;
1979
1980 // Check that A and R are on the curve
1981 // Check that 0 <= S < L (prevents malleability)
1982 // *Allow* non-cannonical encoding for A and R
1983 {
1984 u32 s32[8];
1985 load32_le_buf(s32, s, 8);
1986 if (ge_frombytes_neg_vartime(&minus_A, public_key) ||
1987 ge_frombytes_neg_vartime(&minus_R, signature) ||
1988 is_above_l(s32)) {
1989 return -1;
1990 }
1991 }
1992
1993 // look-up table for minus_A
1994 ge_cached lutA[P_W_SIZE];
1995 {
1996 ge minus_A2, tmp;
1997 ge_double(&minus_A2, &minus_A, &tmp);
1998 ge_cache(&lutA[0], &minus_A);
1999 FOR (i, 1, P_W_SIZE) {
2000 ge_add(&tmp, &minus_A2, &lutA[i-1]);
2001 ge_cache(&lutA[i], &tmp);
2002 }
2003 }
2004
2005 // sum = [s]B - [h]A
2006 // Merged double and add ladder, fused with sliding
2007 slide_ctx h_slide; slide_init(&h_slide, h);
2008 slide_ctx s_slide; slide_init(&s_slide, s);
2009 int i = MAX(h_slide.next_check, s_slide.next_check);
2010 ge *sum = &minus_A; // reuse minus_A for the sum
2011 ge_zero(sum);
2012 while (i >= 0) {
2013 ge tmp;
2014 ge_double(sum, sum, &tmp);
2015 int h_digit = slide_step(&h_slide, P_W_WIDTH, i, h);
2016 int s_digit = slide_step(&s_slide, B_W_WIDTH, i, s);
2017 if (h_digit > 0) { ge_add(sum, sum, &lutA[ h_digit / 2]); }
2018 if (h_digit < 0) { ge_sub(sum, sum, &lutA[-h_digit / 2]); }
2019 fe t1, t2;
2020 if (s_digit > 0) { ge_madd(sum, sum, b_window + s_digit/2, t1, t2); }
2021 if (s_digit < 0) { ge_msub(sum, sum, b_window + -s_digit/2, t1, t2); }
2022 i--;
2023 }
2024
2025 // Compare [8](sum-R) and the zero point
2026 // The multiplication by 8 eliminates any low-order component
2027 // and ensures consistency with batched verification.
2028 ge_cached cached;
2029 u8 check[32];
2030 static const u8 zero_point[32] = {1}; // Point of order 1
2031 ge_cache(&cached, &minus_R);
2032 ge_add(sum, sum, &cached);
2033 ge_double(sum, sum, &minus_R); // reuse minus_R as temporary
2034 ge_double(sum, sum, &minus_R); // reuse minus_R as temporary
2035 ge_double(sum, sum, &minus_R); // reuse minus_R as temporary
2036 ge_tobytes(check, sum);
2037 return crypto_verify32(check, zero_point);
2038}

2143static void lookup_add(ge *p, ge_precomp *tmp_c, fe tmp_a, fe tmp_b,
2144 const ge_precomp comb[8], const u8 scalar[32], int i)
2145{
2146 u8 teeth = (u8)((scalar_bit(scalar, i) ) +
2147 (scalar_bit(scalar, i + 32) << 1) +
2148 (scalar_bit(scalar, i + 64) << 2) +
2149 (scalar_bit(scalar, i + 96) << 3));
2150 u8 high = teeth >> 3;
2151 u8 index = (teeth ^ (high - 1)) & 7;
2152 FOR (j, 0, 8) {
2153 i32 select = 1 & (((j ^ index) - 1) >> 8);
2154 fe_ccopy(tmp_c->Yp, comb[j].Yp, select);
2155 fe_ccopy(tmp_c->Ym, comb[j].Ym, select);
2156 fe_ccopy(tmp_c->T2, comb[j].T2, select);
2157 }
2158 fe_neg(tmp_a, tmp_c->T2);
2159 fe_cswap(tmp_c->T2, tmp_a , high ^ 1);
2160 fe_cswap(tmp_c->Yp, tmp_c->Ym, high ^ 1);
2161 ge_madd(p, p, tmp_c, tmp_a, tmp_b);
2162}

2169 // 1 / 2 modulo L

2189 fe_1(tmp_c.Ym);
2190 fe_0(tmp_c.T2);

2253// Digital signature of a message with from a secret key.
2254//
2255// The secret key comprises two parts:
2256// - The seed that generates the key (secret_key[ 0..31])
2257// - The public key (secret_key[32..63])
2258//
2259// The seed and the public key are bundled together to make sure users
2260// don't use mismatched seeds and public keys, which would instantly
2261// leak the secret scalar and allow forgeries (allowing this to happen
2262// has resulted in critical vulnerabilities in the wild).
2263//
2264// The seed is hashed to derive the secret scalar and a secret prefix.
2265// The sole purpose of the prefix is to generate a secret random nonce.
2266// The properties of that nonce must be as follows:
2267// - Unique: we need a different one for each message.
2268// - Secret: third parties must not be able to predict it.
2269// - Random: any detectable bias would break all security.
2270//
2271// There are two ways to achieve these properties. The obvious one is
2272// to simply generate a random number. Here that would be a parameter
2273// (Monocypher doesn't have an RNG). It works, but then users may reuse
2274// the nonce by accident, which _also_ leaks the secret scalar and
2275// allows forgeries. This has happened in the wild too.
2276//
2277// This is no good, so instead we generate that nonce deterministically
2278// by reducing modulo L a hash of the secret prefix and the message.
2279// The secret prefix makes the nonce unpredictable, the message makes it
2280// unique, and the hash/reduce removes all bias.
2281//
2282// The cost of that safety is hashing the message twice. If that cost
2283// is unacceptable, there are two alternatives:
2284//
2285// - Signing a hash of the message instead of the message itself. This
2286// is fine as long as the hash is collision resistant. It is not
2287// compatible with existing "pure" signatures, but at least it's safe.
2288//
2289// - Using a random nonce. Please exercise **EXTREME CAUTION** if you
2290// ever do that. It is absolutely **critical** that the nonce is
2291// really an unbiased random number between 0 and L-1, never reused,
2292// and wiped immediately.
2293//
2294// To lower the likelihood of complete catastrophe if the RNG is
2295// either flawed or misused, you can hash the RNG output together with
2296// the secret prefix and the beginning of the message, and use the
2297// reduction of that hash instead of the RNG output itself. It's not
2298// foolproof (you'd need to hash the whole message) but it helps.
2299//
2300// Signing a message involves the following operations:
2301//
2302// scalar, prefix = HASH(secret_key)
2303// r = HASH(prefix || message) % L
2304// R = [r]B
2305// h = HASH(R || public_key || message) % L
2306// S = ((h * a) + r) % L
2307// signature = R || S
2308void crypto_eddsa_sign(u8 signature [64], const u8 secret_key[64],
2309 const u8 *message, size_t message_size)
2310{
2311 u8 a[64]; // secret scalar and prefix
2312 u8 r[32]; // secret deterministic "random" nonce
2313 u8 h[32]; // publically verifiable hash of the message (not wiped)
2314 u8 R[32]; // first half of the signature (allows overlapping inputs)
2315
2316 crypto_blake2b(a, 64, secret_key, 32);
2317 crypto_eddsa_trim_scalar(a, a);
2318 hash_reduce(r, a + 32, 32, message, message_size, 0, 0);
2319 crypto_eddsa_scalarbase(R, r);
2320 hash_reduce(h, R, 32, secret_key + 32, 32, message, message_size);
2321 COPY(signature, R, 32);
2322 crypto_eddsa_mul_add(signature + 32, h, a, r);
2323
2324 WIPE_BUFFER(a);
2325 WIPE_BUFFER(r);
2326}
2327
2328// To check the signature R, S of the message M with the public key A,
2329// there are 3 steps:
2330//
2331// compute h = HASH(R || A || message) % L
2332// check that A is on the curve.
2333// check that R == [s]B - [h]A
2334//
2335// The last two steps are done in crypto_eddsa_check_equation()
2336int crypto_eddsa_check(const u8 signature[64], const u8 public_key[32],
2337 const u8 *message, size_t message_size)
2338{
2339 u8 h[32];
2340 hash_reduce(h, signature, 32, public_key, 32, message, message_size);
2341 return crypto_eddsa_check_equation(signature, public_key, h);
2342}

2381// Those functions generates a public key, *without* clearing the
2382// cofactor. Sending that key over the network leaks 3 bits of the
2383// private key. Use only to generate ephemeral keys that will be hidden
2384// with crypto_curve_to_hidden().
2385//
2386// The public key is otherwise compatible with crypto_x25519(), which
2387// properly clears the cofactor.
2388//
2389// Note that the distribution of the resulting public keys is almost
2390// uniform. Flipping the sign of the v coordinate (not provided by this
2391// function), covers the entire key space almost perfectly, where
2392// "almost" means a 2^-128 bias (undetectable). This uniformity is
2393// needed to ensure the proper randomness of the resulting
2394// representatives (once we apply crypto_curve_to_hidden()).
2395//
2396// Recall that Curve25519 has order C = 2^255 + e, with e < 2^128 (not
2397// to be confused with the prime order of the main subgroup, L, which is
2398// 8 times less than that).
2399//
2400// Generating all points would require us to multiply a point of order C
2401// (the base point plus any point of order 8) by all scalars from 0 to
2402// C-1. Clamping limits us to scalars between 2^254 and 2^255 - 1. But
2403// by negating the resulting point at random, we also cover scalars from
2404// -2^255 + 1 to -2^254 (which modulo C is congruent to e+1 to 2^254 + e).
2405//
2406// In practice:
2407// - Scalars from 0 to e + 1 are never generated
2408// - Scalars from 2^255 to 2^255 + e are never generated
2409// - Scalars from 2^254 + 1 to 2^254 + e are generated twice
2410//
2411// Since e < 2^128, detecting this bias requires observing over 2^100
2412// representatives from a given source (this will never happen), *and*
2413// recovering enough of the private key to determine that they do, or do
2414// not, belong to the biased set (this practically requires solving
2415// discrete logarithm, which is conjecturally intractable).
2416//
2417// In practice, this means the bias is impossible to detect.
2418
2419// s + (x*L) % 8*L
2420// Guaranteed to fit in 256 bits iff s fits in 255 bits.
2421// L < 2^253
2422// x%8 < 2^3
2423// L * (x%8) < 2^255
2424// s < 2^255
2425// s + L * (x%8) < 2^256
2426static void add_xl(u8 s[32], u8 x)
2427{
2428 u64 mod8 = x & 7;
2429 u64 carry = 0;
2430 FOR (i , 0, 8) {
2431 carry = carry + load32_le(s + 4*i) + L[i] * mod8;
2432 store32_le(s + 4*i, (u32)carry);
2433 carry >>= 32;
2434 }
2435}

2437// "Small" dirty ephemeral key.
2438// Use if you need to shrink the size of the binary, and can afford to
2439// slow down by a factor of two (compared to the fast version)
2440//
2441// This version works by decoupling the cofactor from the main factor.
2442//
2443// - The trimmed scalar determines the main factor
2444// - The clamped bits of the scalar determine the cofactor.
2445//
2446// Cofactor and main factor are combined into a single scalar, which is
2447// then multiplied by a point of order 8*L (unlike the base point, which
2448// has prime order). That "dirty" base point is the addition of the
2449// regular base point (9), and a point of order 8.
2450void crypto_x25519_dirty_small(u8 public_key[32], const u8 secret_key[32])
2451{
2452 // Base point of order 8*L
2453 // Raw scalar multiplication with it does not clear the cofactor,
2454 // and the resulting public key will reveal 3 bits of the scalar.
2455 //
2456 // The low order component of this base point has been chosen
2457 // to yield the same results as crypto_x25519_dirty_fast().
2458 static const u8 dirty_base_point[32] = {
2459 0xd8, 0x86, 0x1a, 0xa2, 0x78, 0x7a, 0xd9, 0x26,
2460 0x8b, 0x74, 0x74, 0xb6, 0x82, 0xe3, 0xbe, 0xc3,
2461 0xce, 0x36, 0x9a, 0x1e, 0x5e, 0x31, 0x47, 0xa2,
2462 0x6d, 0x37, 0x7c, 0xfd, 0x20, 0xb5, 0xdf, 0x75,
2463 };
2464 // separate the main factor & the cofactor of the scalar
2465 u8 scalar[32];
2466 crypto_eddsa_trim_scalar(scalar, secret_key);
2467
2468 // Separate the main factor and the cofactor
2469 //
2470 // The scalar is trimmed, so its cofactor is cleared. The three
2471 // least significant bits however still have a main factor. We must
2472 // remove it for X25519 compatibility.
2473 //
2474 // cofactor = lsb * L (modulo 8*L)
2475 // combined = scalar + cofactor (modulo 8*L)
2476 add_xl(scalar, secret_key[0]);
2477 scalarmult(public_key, scalar, dirty_base_point, 256);
2478 WIPE_BUFFER(scalar);
2479}

2481// Select low order point
2482// We're computing the [cofactor]lop scalar multiplication, where:
2483//
2484// cofactor = tweak & 7.
2485// lop = (lop_x, lop_y)
2486// lop_x = sqrt((sqrt(d + 1) + 1) / d)
2487// lop_y = -lop_x * sqrtm1
2488//
2489// The low order point has order 8. There are 4 such points. We've
2490// chosen the one whose both coordinates are positive (below p/2).
2491// The 8 low order points are as follows:
2492//
2493// [0]lop = ( 0 , 1 )
2494// [1]lop = ( lop_x , lop_y)
2495// [2]lop = ( sqrt(-1), -0 )
2496// [3]lop = ( lop_x , -lop_y)
2497// [4]lop = (-0 , -1 )
2498// [5]lop = (-lop_x , -lop_y)
2499// [6]lop = (-sqrt(-1), 0 )
2500// [7]lop = (-lop_x , lop_y)
2501//
2502// The x coordinate is either 0, sqrt(-1), lop_x, or their opposite.
2503// The y coordinate is either 0, -1 , lop_y, or their opposite.
2504// The pattern for both is the same, except for a rotation of 2 (modulo 8)
2505//
2506// This helper function captures the pattern, and we can use it thus:
2507//
2508// select_lop(x, lop_x, sqrtm1, cofactor);
2509// select_lop(y, lop_y, fe_one, cofactor + 2);
2510//
2511// This is faster than an actual scalar multiplication,
2512// and requires less code than naive constant time look up.
2513static void select_lop(fe out, const fe x, const fe k, u8 cofactor)
2514{
2515 fe tmp;
2516 fe_0(out);
2517 fe_ccopy(out, k , (cofactor >> 1) & 1); // bit 1
2518 fe_ccopy(out, x , (cofactor >> 0) & 1); // bit 0
2519 fe_neg (tmp, out);
2520 fe_ccopy(out, tmp, (cofactor >> 2) & 1); // bit 2
2521 WIPE_BUFFER(tmp);
2522}

2524// "Fast" dirty ephemeral key
2525// We use this one by default.
2526//
2527// This version works by performing a regular scalar multiplication,
2528// then add a low order point. The scalar multiplication is done in
2529// Edwards space for more speed (*2 compared to the "small" version).
2530// The cost is a bigger binary for programs that don't also sign messages.
2531void crypto_x25519_dirty_fast(u8 public_key[32], const u8 secret_key[32])
2532{
2533 // Compute clean scalar multiplication
2534 u8 scalar[32];
2535 ge pk;
2536 crypto_eddsa_trim_scalar(scalar, secret_key);
2537 ge_scalarmult_base(&pk, scalar);
2538
2539 // Compute low order point
2540 fe t1, t2;
2541 select_lop(t1, lop_x, sqrtm1, secret_key[0]);
2542 select_lop(t2, lop_y, fe_one, secret_key[0] + 2);
2543 ge_precomp low_order_point;
2544 fe_add(low_order_point.Yp, t2, t1);
2545 fe_sub(low_order_point.Ym, t2, t1);
2546 fe_mul(low_order_point.T2, t2, t1);
2547 fe_mul(low_order_point.T2, low_order_point.T2, D2);
2548
2549 // Add low order point to the public key
2550 ge_madd(&pk, &pk, &low_order_point, t1, t2);
2551
2552 // Convert to Montgomery u coordinate (we ignore the sign)
2553 fe_add(t1, pk.Z, pk.Y);
2554 fe_sub(t2, pk.Z, pk.Y);
2555 fe_invert(t2, t2);
2556 fe_mul(t1, t1, t2);
2557
2558 fe_tobytes(public_key, t1);
2559
2560 WIPE_BUFFER(t1); WIPE_CTX(&pk);
2561 WIPE_BUFFER(t2); WIPE_CTX(&low_order_point);
2562 WIPE_BUFFER(scalar);
2563}

2570// Elligator direct map
2571//
2572// Computes the point corresponding to a representative, encoded in 32
2573// bytes (little Endian). Since positive representatives fits in 254
2574// bits, The two most significant bits are ignored.
2575//
2576// From the paper:
2577// w = -A / (fe(1) + non_square * r^2)
2578// e = chi(w^3 + A*w^2 + w)
2579// u = e*w - (fe(1)-e)*(A//2)
2580// v = -e * sqrt(u^3 + A*u^2 + u)
2581//
2582// We ignore v because we don't need it for X25519 (the Montgomery
2583// ladder only uses u).
2584//
2585// Note that e is either 0, 1 or -1
2586// if e = 0 u = 0 and v = 0
2587// if e = 1 u = w
2588// if e = -1 u = -w - A = w * non_square * r^2
2589//
2590// Let r1 = non_square * r^2
2591// Let r2 = 1 + r1
2592// Note that r2 cannot be zero, -1/non_square is not a square.
2593// We can (tediously) verify that:
2594// w^3 + A*w^2 + w = (A^2*r1 - r2^2) * A / r2^3
2595// Therefore:
2596// chi(w^3 + A*w^2 + w) = chi((A^2*r1 - r2^2) * (A / r2^3))
2597// chi(w^3 + A*w^2 + w) = chi((A^2*r1 - r2^2) * (A / r2^3)) * 1
2598// chi(w^3 + A*w^2 + w) = chi((A^2*r1 - r2^2) * (A / r2^3)) * chi(r2^6)
2599// chi(w^3 + A*w^2 + w) = chi((A^2*r1 - r2^2) * (A / r2^3) * r2^6)
2600// chi(w^3 + A*w^2 + w) = chi((A^2*r1 - r2^2) * A * r2^3)
2601// Corollary:
2602// e = 1 if (A^2*r1 - r2^2) * A * r2^3) is a non-zero square
2603// e = -1 if (A^2*r1 - r2^2) * A * r2^3) is not a square
2604// Note that w^3 + A*w^2 + w (and therefore e) can never be zero:
2605// w^3 + A*w^2 + w = w * (w^2 + A*w + 1)
2606// w^3 + A*w^2 + w = w * (w^2 + A*w + A^2/4 - A^2/4 + 1)
2607// w^3 + A*w^2 + w = w * (w + A/2)^2 - A^2/4 + 1)
2608// which is zero only if:
2609// w = 0 (impossible)
2610// (w + A/2)^2 = A^2/4 - 1 (impossible, because A^2/4-1 is not a square)
2611//
2612// Let isr = invsqrt((A^2*r1 - r2^2) * A * r2^3)
2613// isr = sqrt(1 / ((A^2*r1 - r2^2) * A * r2^3)) if e = 1
2614// isr = sqrt(sqrt(-1) / ((A^2*r1 - r2^2) * A * r2^3)) if e = -1
2615//
2616// if e = 1
2617// let u1 = -A * (A^2*r1 - r2^2) * A * r2^2 * isr^2
2618// u1 = w
2619// u1 = u
2620//
2621// if e = -1
2622// let ufactor = -non_square * sqrt(-1) * r^2
2623// let vfactor = sqrt(ufactor)
2624// let u2 = -A * (A^2*r1 - r2^2) * A * r2^2 * isr^2 * ufactor
2625// u2 = w * -1 * -non_square * r^2
2626// u2 = w * non_square * r^2
2627// u2 = u
2628void crypto_elligator_map(u8 curve[32], const u8 hidden[32])
2629{
2630 fe r, u, t1, t2, t3;
2631 fe_frombytes_mask(r, hidden, 2); // r is encoded in 254 bits.
2632 fe_sq(r, r);
2633 fe_add(t1, r, r);
2634 fe_add(u, t1, fe_one);
2635 fe_sq (t2, u);
2636 fe_mul(t3, A2, t1);
2637 fe_sub(t3, t3, t2);
2638 fe_mul(t3, t3, A);
2639 fe_mul(t1, t2, u);
2640 fe_mul(t1, t3, t1);
2641 int is_square = invsqrt(t1, t1);
2642 fe_mul(u, r, ufactor);
2643 fe_ccopy(u, fe_one, is_square);
2644 fe_sq (t1, t1);
2645 fe_mul(u, u, A);
2646 fe_mul(u, u, t3);
2647 fe_mul(u, u, t2);
2648 fe_mul(u, u, t1);
2649 fe_neg(u, u);
2650 fe_tobytes(curve, u);
2651
2652 WIPE_BUFFER(t1); WIPE_BUFFER(r);
2653 WIPE_BUFFER(t2); WIPE_BUFFER(u);
2654 WIPE_BUFFER(t3);
2655}

2657// Elligator inverse map
2658//
2659// Computes the representative of a point, if possible. If not, it does
2660// nothing and returns -1. Note that the success of the operation
2661// depends only on the point (more precisely its u coordinate). The
2662// tweak parameter is used only upon success
2663//
2664// The tweak should be a random byte. Beyond that, its contents are an
2665// implementation detail. Currently, the tweak comprises:
2666// - Bit 1 : sign of the v coordinate (0 if positive, 1 if negative)
2667// - Bit 2-5: not used
2668// - Bits 6-7: random padding
2669//
2670// From the paper:
2671// Let sq = -non_square * u * (u+A)
2672// if sq is not a square, or u = -A, there is no mapping
2673// Assuming there is a mapping:
2674// if v is positive: r = sqrt(-u / (non_square * (u+A)))
2675// if v is negative: r = sqrt(-(u+A) / (non_square * u ))
2676//
2677// We compute isr = invsqrt(-non_square * u * (u+A))
2678// if it wasn't a square, abort.
2679// else, isr = sqrt(-1 / (non_square * u * (u+A))
2680//
2681// If v is positive, we return isr * u:
2682// isr * u = sqrt(-1 / (non_square * u * (u+A)) * u
2683// isr * u = sqrt(-u / (non_square * (u+A))
2684//
2685// If v is negative, we return isr * (u+A):
2686// isr * (u+A) = sqrt(-1 / (non_square * u * (u+A)) * (u+A)
2687// isr * (u+A) = sqrt(-(u+A) / (non_square * u)
2688int crypto_elligator_rev(u8 hidden[32], const u8 public_key[32], u8 tweak)
2689{
2690 fe t1, t2, t3;
2691 fe_frombytes(t1, public_key); // t1 = u
2692
2693 fe_add(t2, t1, A); // t2 = u + A
2694 fe_mul(t3, t1, t2);
2695 fe_mul_small(t3, t3, -2);
2696 int is_square = invsqrt(t3, t3); // t3 = sqrt(-1 / non_square * u * (u+A))
2697 if (is_square) {
2698 // The only variable time bit. This ultimately reveals how many
2699 // tries it took us to find a representable key.
2700 // This does not affect security as long as we try keys at random.
2701
2702 fe_ccopy (t1, t2, tweak & 1); // multiply by u if v is positive,
2703 fe_mul (t3, t1, t3); // multiply by u+A otherwise
2704 fe_mul_small(t1, t3, 2);
2705 fe_neg (t2, t3);
2706 fe_ccopy (t3, t2, fe_isodd(t1));
2707 fe_tobytes(hidden, t3);
2708
2709 // Pad with two random bits
2710 hidden[31] |= tweak & 0xc0;
2711 }
2712
2713 WIPE_BUFFER(t1);
2714 WIPE_BUFFER(t2);
2715 WIPE_BUFFER(t3);
2716 return is_square - 1;
2717}

2745// Montgomery reduction.
2746// Divides x by (2^256), and reduces the result modulo L
2747//
2748// Precondition:
2749// x < L * 2^256
2750// Constants:
2751// r = 2^256 (makes division by r trivial)
2752// k = (r * (1/r) - 1) // L (1/r is computed modulo L )
2753// Algorithm:
2754// s = (x * k) % r
2755// t = x + s*L (t is always a multiple of r)
2756// u = (t/r) % L (u is always below 2*L, conditional subtraction is enough)
2757static void redc(u32 u[8], u32 x[16])
2758{
2759 static const u32 k[8] = {
2760 0x12547e1b, 0xd2b51da3, 0xfdba84ff, 0xb1a206f2,
2761 0xffa36bea, 0x14e75438, 0x6fe91836, 0x9db6c6f2,
2762 };
2763
2764 // s = x * k (modulo 2^256)
2765 // This is cheaper than the full multiplication.
2766 u32 s[8] = {0};
2767 FOR (i, 0, 8) {
2768 u64 carry = 0;
2769 FOR (j, 0, 8-i) {
2770 carry += s[i+j] + (u64)x[i] * k[j];
2771 s[i+j] = (u32)carry;
2772 carry >>= 32;
2773 }
2774 }
2775 u32 t[16] = {0};
2776 multiply(t, s, L);
2777
2778 // t = t + x
2779 u64 carry = 0;
2780 FOR (i, 0, 16) {
2781 carry += (u64)t[i] + x[i];
2782 t[i] = (u32)carry;
2783 carry >>= 32;
2784 }
2785
2786 // u = (t / 2^256) % L
2787 // Note that t / 2^256 is always below 2*L,
2788 // So a constant time conditional subtraction is enough
2789 remove_l(u, t+8);
2790
2791 WIPE_BUFFER(s);
2792 WIPE_BUFFER(t);
2793}

2795void crypto_x25519_inverse(u8 blind_salt [32], const u8 private_key[32],
2796 const u8 curve_point[32])
2797{
2798 static const u8 Lm2[32] = { // L - 2
2799 0xeb, 0xd3, 0xf5, 0x5c, 0x1a, 0x63, 0x12, 0x58,
2800 0xd6, 0x9c, 0xf7, 0xa2, 0xde, 0xf9, 0xde, 0x14,
2801 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00,
2802 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x10,
2803 };
2804 // 1 in Montgomery form
2805 u32 m_inv [8] = {
2806 0x8d98951d, 0xd6ec3174, 0x737dcf70, 0xc6ef5bf4,
2807 0xfffffffe, 0xffffffff, 0xffffffff, 0x0fffffff,
2808 };
2809
2810 u8 scalar[32];
2811 crypto_eddsa_trim_scalar(scalar, private_key);
2812
2813 // Convert the scalar in Montgomery form
2814 // m_scl = scalar * 2^256 (modulo L)
2815 u32 m_scl[8];
2816 {
2817 u32 tmp[16];
2818 ZERO(tmp, 8);
2819 load32_le_buf(tmp+8, scalar, 8);
2820 mod_l(scalar, tmp);
2821 load32_le_buf(m_scl, scalar, 8);
2822 WIPE_BUFFER(tmp); // Wipe ASAP to save stack space
2823 }
2824
2825 // Compute the inverse
2826 u32 product[16];
2827 for (int i = 252; i >= 0; i--) {
2828 ZERO(product, 16);
2829 multiply(product, m_inv, m_inv);
2830 redc(m_inv, product);
2831 if (scalar_bit(Lm2, i)) {
2832 ZERO(product, 16);
2833 multiply(product, m_inv, m_scl);
2834 redc(m_inv, product);
2835 }
2836 }
2837 // Convert the inverse *out* of Montgomery form
2838 // scalar = m_inv / 2^256 (modulo L)
2839 COPY(product, m_inv, 8);
2840 ZERO(product + 8, 8);
2841 redc(m_inv, product);
2842 store32_le_buf(scalar, m_inv, 8); // the *inverse* of the scalar
2843
2844 // Clear the cofactor of scalar:
2845 // cleared = scalar * (3*L + 1) (modulo 8*L)
2846 // cleared = scalar + scalar * 3 * L (modulo 8*L)
2847 // Note that (scalar * 3) is reduced modulo 8, so we only need the
2848 // first byte.
2849 add_xl(scalar, scalar[0] * 3);
2850
2851 // Recall that 8*L < 2^256. However it is also very close to
2852 // 2^255. If we spanned the ladder over 255 bits, random tests
2853 // wouldn't catch the off-by-one error.
2854 scalarmult(blind_salt, scalar, curve_point, 256);
2855
2856 WIPE_BUFFER(scalar); WIPE_BUFFER(m_scl);
2857 WIPE_BUFFER(product); WIPE_BUFFER(m_inv);
2858}

239bool Start()
240{
241 ALuint uiLoop, uiInnerLoop;
242 bool bReturn = false;
243
244 if (g_bALInit)
245 {
246 alGetError();
247
248 // We are going to fill all the streaming Buffers of all the Sources
249 // with silence to kick things off
250 memset(g_pData, 0, g_uiNumSamplesPerBuffer * 2 * g_uiOutputChannels);
251
252 // Fill direct path Buffers with silence
253 for (uiLoop = 0; uiLoop < NUM_STREAMING_BUFFERS; uiLoop++)
254 {
255 alBufferData(g_uiBuffers[uiLoop], g_eFormat, g_pData, g_uiNumSamplesPerBuffer * 2 * g_uiOutputChannels, g_uiFrequency);
256 alSourceQueueBuffers(g_uiSources[0], 1, &g_uiBuffers[uiLoop]);
257 }
258
259 // Fill Effect Buffers with silence
260 for (uiLoop = 1; uiLoop < (g_uiNumEffectSlots + 1); uiLoop++)
261 {
262 for (uiInnerLoop = 0; uiInnerLoop < NUM_STREAMING_BUFFERS; uiInnerLoop++)
263 {
264 alBufferData(g_uiBuffers[(uiLoop * NUM_STREAMING_BUFFERS) + uiInnerLoop], AL_FORMAT_MONO16, g_pData, g_uiNumSamplesPerBuffer * 2, g_uiFrequency);
265 alSourceQueueBuffers(g_uiSources[uiLoop], 1, &g_uiBuffers[(uiLoop * NUM_STREAMING_BUFFERS) + uiInnerLoop]);
266 }
267 }
268
269 // Vector Play call
270 alSourcePlayv(g_uiNumEffectSlots + 1, g_uiSources);
271
272 if (alGetError() == AL_NO_ERROR)
273 bReturn = true;
274 }
275
276 return bReturn;

302unsigned long GetBuffersProcessed()
303{
304 ALint iProcessed;
305 ALuint uiMinProcessed = -1;
306 ALuint uiLoop;
307 ALint iState;
308
309 alGetError();
310
311 // Keep things in sync. by returning the minimum number of
312 // Buffers processed on each of the playing Sources
313 for (uiLoop = 0; uiLoop < (g_uiNumEffectSlots + 1); uiLoop++)
314 {
315 alGetSourcei(g_uiSources[uiLoop], AL_BUFFERS_PROCESSED, &iProcessed);
316 uiMinProcessed = __min(uiMinProcessed, (ALuint)iProcessed);
317 if (uiMinProcessed == 0)
318 break;
319 }
320
321 // Check if all the Sources are still Playing
322 for (uiLoop = 0; uiLoop < (g_uiNumEffectSlots + 1); uiLoop++)
323 {
324 alGetSourcei(g_uiSources[uiLoop], AL_SOURCE_STATE, &iState);
325 if (iState != AL_PLAYING)
326 break;
327 }
328
329 // If any of them have stopped, we need to stop them all
330 if (iState != AL_PLAYING)
331 {
332 alSourceStopv(g_uiNumEffectSlots + 1, g_uiSources);
333 // It is possible that a buffer was queued *after* the Source stopped,
334 // so we need to check how many Buffers have been processed
335 for (uiLoop = 0; uiLoop < (g_uiNumEffectSlots + 1); uiLoop++)
336 {
337 alGetSourcei(g_uiSources[uiLoop], AL_BUFFERS_PROCESSED, &iProcessed);
338 uiMinProcessed = __min(uiMinProcessed, (ALuint)iProcessed);
339 }
340 g_uiPlayCounter = uiMinProcessed;
341 OutputDebugMessage("ALOutput : One or more Sources were starved, stopping all Sources (%d Buffers Processed)\n", uiMinProcessed);
342 }
343
344 if (alGetError() != AL_NO_ERROR)
345 OutputDebugMessage("ALOutput : GetBuffersProcessed set an AL error\n");
346
347 return uiMinProcessed;
348}

381
382 // Do we need to re-start the Sources?
383 if (g_uiPlayCounter)

386 if (g_uiPlayCounter == 0)

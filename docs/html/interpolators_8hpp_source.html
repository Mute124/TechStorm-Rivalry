36 // A template class for an object interpolator. The values in the objects are
37 // not interpolated, but rather the current or target object is returned: if
38 // 'timeLeft' becomes smaller, the chance of getting the target object instead
39 // of the current one becomes bigger.
40 template <class TYPE> class objectInterpolator {
41 public:
42 objectInterpolator() : totalTime(0), timeLeft(0), _isSet(false) {}
43
44 // Set a new target object. Be sure it has a valid copy constructor
45 // (unless type is a pointer)
46 objectInterpolator<TYPE> & set(TYPE obj, Flt time = 0) {
47 if (!_isSet) {
48 totalTime = timeLeft = 0;
49 current = obj;
50 _isSet = true;
51 }
52 else {
53 totalTime = timeLeft = time;
54 timeLeft > 0 ? target = obj : current = obj;
55 }
56 return *this;
57 }
58
59 // call this function once for every loop
60 objectInterpolator<TYPE> & update(Flt timeDelta) {
61 if (timeLeft > 0) {
62 timeLeft -= timeDelta;
63 if (timeLeft <= 0) {
64 current = target;
65 timeLeft = 0;
66 }
67 }
68 return *this;
69 }
70
71 // get the object
72 TYPE & operator()() {
73 if (timeLeft > 0) {
74 return RandomF(totalTime) > timeLeft ? target : current;
75 }
76 return current;
77 }
78
79 Bool isSet() { return _isSet; }
80 private:
81 TYPE current, target;
82 Flt totalTime, timeLeft;
83 Bool _isSet;
84 };

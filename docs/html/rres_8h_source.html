1/**********************************************************************************************
2*
3* rres v1.0 - A simple and easy-to-use file-format to package resources
4*
5* CONFIGURATION:
6*
7* #define RRES_IMPLEMENTATION
8* Generates the implementation of the library into the included file.
9* If not defined, the library is in header only mode and can be included in other headers
10* or source files without problems. But only ONE file should hold the implementation.
11*
12* FEATURES:
13*
14* - Multi-resource files: Some files could end-up generating multiple connected resources in
15* the rres output file (i.e TTF files could generate RRES_DATA_FONT_GLYPHS and RRES_DATA_IMAGE).
16* - File packaging as raw resource data: Avoid data processing and just package the file bytes.
17* - Per-file data compression/encryption: Configure compression/encription for every input file.
18* - Externally linked files: Package only the file path, to be loaded from external file when the
19* specific id is requested. WARNING: Be careful with path, it should be relative to application dir.
20* - Central Directory resource (optional): Create a central directory with the input filename relation
21* to the resource(s) id. This is the default option but it can be avoided; in that case, a header
22* file (.h) is generated with the file ids definitions.
23*
24* FILE STRUCTURE:
25*
26* rres files consist of a file header followed by a number of resource chunks.
27*
28* Optionally it can contain a Central Directory resource chunk (usually at the end) with the info
29* of all the files processed into the rres file.
30*
31* NOTE: Chunks count could not match files count, some processed files (i.e Font, Mesh)
32* could generate multiple chunks with the same id related by the rresResourceChunkInfo.nextOffset
33* Those chunks are loaded together when resource is loaded
34*
35* rresFileHeader (16 bytes)
36* Signature Id (4 bytes) // File signature id: 'rres'
37* Version (2 bytes) // Format version
38* Resource Count (2 bytes) // Number of resource chunks contained
39* CD Offset (4 bytes) // Central Directory offset (if available)
40* Reserved (4 bytes) // <reserved>
41*
42* rresResourceChunk[]
43* {
44* rresResourceChunkInfo (32 bytes)
45* Type (4 bytes) // Resource type (FourCC)
46* Id (4 bytes) // Resource identifier (CRC32 filename hash or custom)
47* Compressor (1 byte) // Data compression algorithm
48* Cipher (1 byte) // Data encryption algorithm
49* Flags (2 bytes) // Data flags (if required)
50* Data Packed Size (4 bytes) // Data packed size (compressed/encrypted + custom data appended)
51* Data Base Size (4 bytes) // Data base size (uncompressed/unencrypted)
52* Next Offset (4 bytes) // Next resource chunk offset (if required)
53* Reserved (4 bytes) // <reserved>
54* CRC32 (4 bytes) // Resource Data Chunk CRC32
55*
56* rresResourceChunkData (n bytes) // Packed data
57* Property Count (4 bytes) // Number of properties contained
58* Properties[] (4*i bytes) // Resource data required properties, depend on Type
59* Data (m bytes) // Resource data
60* }
61*
62* rresResourceChunk: RRES_DATA_DIRECTORY // Central directory (special resource chunk)
63* {
64* rresResourceChunkInfo (32 bytes)
65*
66* rresCentralDir (n bytes) // rresResourceChunkData
67* Entries Count (4 bytes) // Central directory entries count (files)
68* rresDirEntry[]
69* {
70* Id (4 bytes) // Resource id
71* Offset (4 bytes) // Resource global offset in file
72* reserved (4 bytes) // <reserved>
73* FileName Size (4 bytes) // Resource fileName size (NULL terminator and 4-bytes align padding considered)
74* FileName (m bytes) // Resource original fileName (NULL terminated and padded to 4-byte alignment)
75* }
76* }
77*
78* DESIGN DECISIONS / LIMITATIONS:
79*
80* - rres file maximum chunks: 65535 (16bit chunk count in rresFileHeader)
81* - rres file maximum size: 4GB (chunk offset and Central Directory Offset is 32bit, so it can not address more than 4GB
82* - Chunk search by ID is done one by one, starting at first chunk and accessed with fread() function
83* - Endianness: rres does not care about endianness, data is stored as desired by the host platform (most probably Little Endian)
84* Endianness won't affect chunk data but it will affect rresFileHeader and rresResourceChunkInfo
85* - CRC32 hash is used to to generate the rres file identifier from filename
86* There is a "small" probability of random collision (1 in 2^32 approx.) but considering
87* the chance of collision is related to the number of data inputs, not the size of the inputs, we assume that risk
88* Also note that CRC32 is not used as a security/cryptographic hash, just an identifier for the input file
89* - CRC32 hash is also used to detect chunk data corruption. CRC32 is smaller and computationally much less complex than MD5 or SHA1.
90* Using a hash function like MD5 is probably overkill for random error detection
91* - Central Directory rresDirEntry.fileName is NULL terminated and padded to 4-byte, rresDirEntry.fileNameSize considers the padding
92* - Compression and Encryption. rres supports chunks data compression and encryption, it provides two fields in the rresResourceChunkInfo to
93* note it, but in those cases is up to the user to implement the desired compressor/uncompressor and encryption/decryption mechanisms
94* In case of data encryption, it's recommended that any additional resource data (i.e. MAC) to be appended to data chunk and properly
95* noted in the packed data size field of rresResourceChunkInfo. Data compression should be applied before encryption.
96*
97* DEPENDENCIES:
98*
99* rres library dependencies has been keep to the minimum. It depends only some libc functionality:
100*
101* - stdlib.h: Required for memory allocation: malloc(), calloc(), free()
102* NOTE: Allocators can be redefined with macros RRES_MALLOC, RRES_CALLOC, RRES_FREE
103* - stdio.h: Required for file access functionality: FILE, fopen(), fseek(), fread(), fclose()
104* - string.h: Required for memory data management: memcpy(), memcmp()
105*
106* VERSION HISTORY:

186// rres file header (16 bytes)
187typedef struct rresFileHeader {
188 unsigned char id[4]; // File identifier: rres
189 unsigned short version; // File version: 100 for version 1.0
190 unsigned short chunkCount; // Number of resource chunks in the file (MAX: 65535)
191 unsigned int cdOffset; // Central Directory offset in file (0 if not available)
192 unsigned int reserved; // <reserved>
193} rresFileHeader;
194
195// rres resource chunk info header (32 bytes)
196typedef struct rresResourceChunkInfo {
197 unsigned char type[4]; // Resource chunk type (FourCC)
198 unsigned int id; // Resource chunk identifier (generated from filename CRC32 hash)
199 unsigned char compType; // Data compression algorithm
200 unsigned char cipherType; // Data encription algorithm
201 unsigned short flags; // Data flags (if required)
202 unsigned int packedSize; // Data chunk size (compressed/encrypted + custom data appended)
203 unsigned int baseSize; // Data base size (uncompressed/unencrypted)
204 unsigned int nextOffset; // Next resource chunk global offset (if resource has multiple chunks)
205 unsigned int reserved; // <reserved>
206 unsigned int crc32; // Data chunk CRC32 (propCount + props[] + data)
207} rresResourceChunkInfo;

211 unsigned int propCount; // Resource chunk properties count

214} rresResourceChunkData;

262// rres resource chunk data type
263// NOTE 1: Data type determines the properties and the data included in every chunk
264// NOTE 2: This enum defines the basic resource data types,
265// some input files could generate multiple resource chunks:
266// Fonts processed could generate (2) resource chunks:
267// - [FNTG] rres[0]: RRES_DATA_FONT_GLYPHS
268// - [IMGE] rres[1]: RRES_DATA_IMAGE
269//
270// Mesh processed could generate (n) resource chunks:
271// - [VRTX] rres[0]: RRES_DATA_VERTEX
272// ...
273// - [VRTX] rres[n]: RRES_DATA_VERTEX
274typedef enum rresResourceDataType {
275 RRES_DATA_NULL = 0, // FourCC: NULL - Reserved for empty chunks, no props/data
276 RRES_DATA_RAW = 1, // FourCC: RAWD - Raw file data, 4 properties
277 // props[0]:size (bytes)
278 // props[1]:extension01 (big-endian: ".png" = 0x2e706e67)
279 // props[2]:extension02 (additional part, extensions with +3 letters)
280 // props[3]:reserved
281 // data: raw bytes
282 RRES_DATA_TEXT = 2, // FourCC: TEXT - Text file data, 4 properties
283 // props[0]:size (bytes)
284 // props[1]:rresTextEncoding
285 // props[2]:rresCodeLang
286 // props[3]:cultureCode
287 // data: text
288 RRES_DATA_IMAGE = 3, // FourCC: IMGE - Image file data, 4 properties
289 // props[0]:width
290 // props[1]:height
291 // props[2]:rresPixelFormat
292 // props[3]:mipmaps
293 // data: pixels
294 RRES_DATA_WAVE = 4, // FourCC: WAVE - Audio file data, 4 properties
295 // props[0]:frameCount
296 // props[1]:sampleRate
297 // props[2]:sampleSize
298 // props[3]:channels
299 // data: samples
300 RRES_DATA_VERTEX = 5, // FourCC: VRTX - Vertex file data, 4 properties
301 // props[0]:vertexCount
302 // props[1]:rresVertexAttribute
303 // props[2]:componentCount
304 // props[3]:rresVertexFormat

313 // props[0]:size (bytes)
314 // data: filepath (as provided on input)

316 // props[0]:entryCount, 1 property

565// Load one resource chunk for provided id
566rresResourceChunk rresLoadResourceChunk(const char *fileName, int rresId)
567{
568 rresResourceChunk chunk = { 0 };
569
570 FILE *rresFile = fopen(fileName, "rb");
571
572 if (rresFile == NULL) RRES_LOG("RRES: WARNING: [%s] rres file could not be opened\n", fileName);
573 else
574 {
575 RRES_LOG("RRES: INFO: Loading resource from file: %s\n", fileName);
576
577 rresFileHeader header = { 0 };
578
579 // Read rres file header
580 fread(&header, sizeof(rresFileHeader), 1, rresFile);
581
582 // Verify file signature: "rres" and file version: 100
583 if (((header.id[0] == 'r') && (header.id[1] == 'r') && (header.id[2] == 'e') && (header.id[3] == 's')) && (header.version == 100))
584 {
585 bool found = false;
586
587 // Check all available chunks looking for the requested id
588 for (int i = 0; i < header.chunkCount; i++)
589 {
590 rresResourceChunkInfo info = { 0 };
591
592 // Read resource info header
593 fread(&info, sizeof(rresResourceChunkInfo), 1, rresFile);
594
595 // Check if resource id is the requested one
596 if (info.id == rresId)
597 {
598 found = true;
599
600 RRES_LOG("RRES: INFO: Found requested resource id: 0x%08x\n", info.id);
601 RRES_LOG("RRES: %c%c%c%c: Id: 0x%08x | Base size: %i | Packed size: %i\n", info.type[0], info.type[1], info.type[2], info.type[3], info.id, info.baseSize, info.packedSize);
602
603 // NOTE: We only load first matching id resource chunk found but
604 // we show a message if additional chunks are detected
605 if (info.nextOffset != 0) RRES_LOG("RRES: WARNING: Multiple linked resource chunks available for the provided id");
606
607 /*
608 // Variables required to check multiple chunks
609 int chunkCount = 0;
610 long currentFileOffset = ftell(rresFile); // Store current file position
611 rresResourceChunkInfo temp = info; // Temp info header to scan resource chunks
612
613 // Count all linked resource chunks checking temp.nextOffset
614 while (temp.nextOffset != 0)
615 {
616 fseek(rresFile, temp.nextOffset, SEEK_SET); // Jump to next linked resource
617 fread(&temp, sizeof(rresResourceChunkInfo), 1, rresFile); // Read next resource info header
618 chunkCount++;
619 }
620
621 fseek(rresFile, currentFileOffset, SEEK_SET); // Return to first resource chunk position
622 */
623
624 // Read and resource chunk from file data
625 // NOTE: Read data can be compressed/encrypted, it's up to the user library to manage decompression/decryption
626 void *data = RRES_MALLOC(info.packedSize); // Allocate enough memory to store resource data chunk
627 fread(data, info.packedSize, 1, rresFile); // Read data: propsCount + props[] + data (+additional_data)
628
629 // Get chunk.data properly organized (only if uncompressed/unencrypted)
630 chunk.data = rresLoadResourceChunkData(info, data);
631 chunk.info = info;
632
633 RRES_FREE(data);
634
635 break; // Resource id found and loaded, stop checking the file
636 }
637 else
638 {
639 // Skip required data size to read next resource info header
640 fseek(rresFile, info.packedSize, SEEK_CUR);
641 }
642 }
643
644 if (!found) RRES_LOG("RRES: WARNING: Requested resource not found: 0x%08x\n", rresId);
645 }
646 else RRES_LOG("RRES: WARNING: The provided file is not a valid rres file, file signature or version not valid\n");
647
648 fclose(rresFile);
649 }
650
651 return chunk;
652}

855{
856 rresCentralDir dir = { 0 };
857
858 FILE *rresFile = fopen(fileName, "rb");
859
860 if (rresFile != NULL)
861 {

868 {
869 // Check if there is a Central Directory available
870 if (header.cdOffset == 0) RRES_LOG("RRES: WARNING: CDIR: No central directory found\n");
871 else
872 {
873 rresResourceChunkInfo info = { 0 };
874
875 fseek(rresFile, header.cdOffset, SEEK_CUR); // Move to central directory position
876 fread(&info, sizeof(rresResourceChunkInfo), 1, rresFile); // Read resource info
877
878 // Verify resource type is CDIR
879 if ((info.type[0] == 'C') && (info.type[1] == 'D') && (info.type[2] == 'I') && (info.type[3] == 'R'))
880 {
881 RRES_LOG("RRES: CDIR: Central Directory found at offset: 0x%08x\n", header.cdOffset);
882
883 void *data = RRES_MALLOC(info.packedSize);
884 fread(data, info.packedSize, 1, rresFile);
885
886 // Load resource chunk data (central directory), data is uncompressed/unencrypted by default
887 rresResourceChunkData chunkData = rresLoadResourceChunkData(info, data);
888 RRES_FREE(data);
889
890 dir.count = chunkData.props[0]; // File entries count
891
892 RRES_LOG("RRES: CDIR: Central Directory file entries count: %i\n", dir.count);
893
894 unsigned char *ptr = chunkData.raw;
895 dir.entries = (rresDirEntry *)RRES_CALLOC(dir.count, sizeof(rresDirEntry));
896
897 for (unsigned int i = 0; i < dir.count; i++)
898 {
899 dir.entries[i].id = ((int *)ptr)[0]; // Resource id
900 dir.entries[i].offset = ((int *)ptr)[1]; // Resource offset in file
901 // NOTE: There is a reserved integer value before fileNameSize
902 dir.entries[i].fileNameSize = ((int *)ptr)[3]; // Resource fileName size
903
904 // Resource fileName, NULL terminated and 0-padded to 4-byte,
905 // fileNameSize considers NULL and padding
906 memcpy(dir.entries[i].fileName, ptr + 16, dir.entries[i].fileNameSize);
907
908 ptr += (16 + dir.entries[i].fileNameSize); // Move pointer for next entry
909 }
910
911 RRES_FREE(chunkData.props);
912 RRES_FREE(chunkData.raw);
913 }
914 }
915 }
916 else RRES_LOG("RRES: WARNING: The provided file is not a valid rres file, file signature or version not valid\n");
917
918 fclose(rresFile);
919 }
920
921 return dir;
922}

1049// Load user resource chunk from resource packed data (as contained in .rres file)
1050// WARNING: Data can be compressed and/or encrypted, in those cases is up to the user to process it,
1051// and chunk.data.propCount = 0, chunk.data.props = NULL and chunk.data.raw contains all resource packed data
1052static rresResourceChunkData rresLoadResourceChunkData(rresResourceChunkInfo info, void *data)
1053{
1054 rresResourceChunkData chunkData = { 0 };
1055
1056 // CRC32 data validation, verify packed data is not corrupted
1057 unsigned int crc32 = rresComputeCRC32(data, info.packedSize);
1058
1059 if ((rresGetDataType(info.type) != RRES_DATA_NULL) && (crc32 == info.crc32)) // Make sure chunk contains data and data is not corrupted
1060 {
1061 // Check if data chunk is compressed/encrypted to retrieve properties + data
1062 if ((info.compType == RRES_COMP_NONE) && (info.cipherType == RRES_CIPHER_NONE))
1063 {
1064 // Data is not compressed/encrypted (info.packedSize = info.baseSize)
1065 chunkData.propCount = ((unsigned int *)data)[0];
1066
1067 if (chunkData.propCount > 0)
1068 {
1069 chunkData.props = (unsigned int *)RRES_CALLOC(chunkData.propCount, sizeof(unsigned int));
1070 for (unsigned int i = 0; i < chunkData.propCount; i++) chunkData.props[i] = ((unsigned int *)data)[i + 1];
1071 }
1072
1073 chunkData.raw = RRES_MALLOC(info.baseSize);
1074 memcpy(chunkData.raw, ((unsigned char *)data) + sizeof(int) + (chunkData.propCount*sizeof(int)), info.baseSize);
1075 }
1076 else
1077 {
1078 // Data is compressed/encrypted
1079 // We just return the loaded resource packed data from .rres file,
1080 // it's up to the user to manage decompression/decryption on user library
1081 chunkData.raw = RRES_MALLOC(info.packedSize);
1082 memcpy(chunkData.raw, (unsigned char *)data, info.packedSize);
1083 }
1084 }
1085
1086 if (crc32 != info.crc32) RRES_LOG("RRES: WARNING: [ID %i] CRC32 does not match, data can be corrupted\n", info.id);
1087
1088 return chunkData;
1089}

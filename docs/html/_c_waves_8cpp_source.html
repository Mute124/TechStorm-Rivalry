110WAVERESULT CWaves::LoadWaveFile(const char *szFilename, WAVEID *pWaveID)
111{
112 WAVERESULT wr = WR_OUTOFMEMORY;
113 LPWAVEFILEINFO pWaveInfo;
114
115 pWaveInfo = new WAVEFILEINFO;
116 if (pWaveInfo)
117 {
118 if (SUCCEEDED(wr = ParseFile(szFilename, pWaveInfo)))
119 {
120 // Allocate memory for sample data
121 pWaveInfo->pData = new char[pWaveInfo->ulDataSize];
122 if (pWaveInfo->pData)
123 {
124 // Seek to start of audio data
125 fseek(pWaveInfo->pFile, pWaveInfo->ulDataOffset, SEEK_SET);
126
127 // Read Sample Data
128 if (fread(pWaveInfo->pData, 1, pWaveInfo->ulDataSize, pWaveInfo->pFile) == pWaveInfo->ulDataSize)
129 {
130 long lLoop = 0;
131 for (lLoop = 0; lLoop < MAX_NUM_WAVEID; lLoop++)
132 {
133 if (!m_WaveIDs[lLoop])
134 {
135 m_WaveIDs[lLoop] = pWaveInfo;
136 *pWaveID = lLoop;
137 wr = WR_OK;
138 break;
139 }
140 }
141
142 if (lLoop == MAX_NUM_WAVEID)
143 {
144 delete pWaveInfo->pData;
145 wr = WR_OUTOFMEMORY;
146 }
147 }
148 else
149 {
150 delete pWaveInfo->pData;
151 wr = WR_BADWAVEFILE;
152 }
153 }
154 else
155 {
156 wr = WR_OUTOFMEMORY;
157 }
158
159 fclose(pWaveInfo->pFile);
160 pWaveInfo->pFile = 0;
161 }
162
163 if (wr != WR_OK)
164 delete pWaveInfo;
165 }
166
167 return wr;
168}

204WAVERESULT CWaves::ReadWaveData(WAVEID WaveID, void *pData, unsigned long ulDataSize, unsigned long *pulBytesWritten)
205{
206 LPWAVEFILEINFO pWaveInfo;
207 WAVERESULT wr = WR_BADWAVEFILE;
208
209 if (!pData || !pulBytesWritten || ulDataSize == 0)
210 return WR_INVALIDPARAM;
211
212 if (IsWaveID(WaveID))
213 {
214 pWaveInfo = m_WaveIDs[WaveID];
215 if (pWaveInfo->pFile)
216 {
217 unsigned long ulOffset = ftell(pWaveInfo->pFile);
218
219 if ((ulOffset - pWaveInfo->ulDataOffset + ulDataSize) > pWaveInfo->ulDataSize)
220 ulDataSize = pWaveInfo->ulDataSize - (ulOffset - pWaveInfo->ulDataOffset);
221
222 *pulBytesWritten = (unsigned long)fread(pData, 1, ulDataSize, pWaveInfo->pFile);
223
224 wr = WR_OK;
225 }
226 }
227 else
228 {
229 wr = WR_INVALIDWAVEID;
230 }
231
232 return wr;

282WAVERESULT CWaves::ParseFile(const char *szFilename, LPWAVEFILEINFO pWaveInfo)
283{
284 WAVEFILEHEADER waveFileHeader;
285 RIFFCHUNK riffChunk;
286 WAVEFMT waveFmt;
287 WAVERESULT wr = WR_BADWAVEFILE;
288
289 if (!szFilename || !pWaveInfo)
290 return WR_INVALIDPARAM;
291
292 memset(pWaveInfo, 0, sizeof(WAVEFILEINFO));
293
294 // Open the wave file for reading
295 fopen_s(&pWaveInfo->pFile, szFilename, "rb");
296 if (pWaveInfo->pFile)
297 {
298 // Read Wave file header
299 fread(&waveFileHeader, 1, sizeof(WAVEFILEHEADER), pWaveInfo->pFile);
300 if (!_strnicmp(waveFileHeader.szRIFF, "RIFF", 4) && !_strnicmp(waveFileHeader.szWAVE, "WAVE", 4))
301 {
302 while (fread(&riffChunk, 1, sizeof(RIFFCHUNK), pWaveInfo->pFile) == sizeof(RIFFCHUNK))
303 {
304 if (!_strnicmp(riffChunk.szChunkName, "fmt ", 4))
305 {
306 if (riffChunk.ulChunkSize <= sizeof(WAVEFMT))
307 {
308 fread(&waveFmt, 1, riffChunk.ulChunkSize, pWaveInfo->pFile);
309
310 // Determine if this is a WAVEFORMATEX or WAVEFORMATEXTENSIBLE wave file
311 if (waveFmt.usFormatTag == WAVE_FORMAT_PCM)
312 {
313 pWaveInfo->wfType = WF_EX;
314 memcpy(&pWaveInfo->wfEXT.Format, &waveFmt, sizeof(PCMWAVEFORMAT));
315 }
316 else if (waveFmt.usFormatTag == WAVE_FORMAT_EXTENSIBLE)
317 {
318 pWaveInfo->wfType = WF_EXT;
319 memcpy(&pWaveInfo->wfEXT, &waveFmt, sizeof(WAVEFORMATEXTENSIBLE));
320 }
321 }
322 else
323 {
324 fseek(pWaveInfo->pFile, riffChunk.ulChunkSize, SEEK_CUR);
325 }
326 }
327 else if (!_strnicmp(riffChunk.szChunkName, "data", 4))
328 {
329 pWaveInfo->ulDataSize = riffChunk.ulChunkSize;
330 pWaveInfo->ulDataOffset = ftell(pWaveInfo->pFile);
331 fseek(pWaveInfo->pFile, riffChunk.ulChunkSize, SEEK_CUR);
332 }
333 else
334 {
335 fseek(pWaveInfo->pFile, riffChunk.ulChunkSize, SEEK_CUR);
336 }
337
338 // Ensure that we are correctly aligned for next chunk
339 if (riffChunk.ulChunkSize & 1)
340 fseek(pWaveInfo->pFile, 1, SEEK_CUR);
341 }
342
343 if (pWaveInfo->ulDataSize && pWaveInfo->ulDataOffset && ((pWaveInfo->wfType == WF_EX) || (pWaveInfo->wfType == WF_EXT)))
344 wr = WR_OK;
345 else
346 fclose(pWaveInfo->pFile);
347 }
348 }
349 else
350 {
351 wr = WR_INVALIDFILENAME;
352 }
353
354 return wr;
355}

473WAVERESULT CWaves::GetWaveALBufferFormat(WAVEID WaveID, PFNALGETENUMVALUE pfnGetEnumValue, unsigned long *pulFormat)
474{
475 WAVERESULT wr = WR_OK;
476
477 if (IsWaveID(WaveID))
478 {
479 if (pfnGetEnumValue && pulFormat)
480 {
481 *pulFormat = 0;
482
483 if (m_WaveIDs[WaveID]->wfType == WF_EX)
484 {
485 if (m_WaveIDs[WaveID]->wfEXT.Format.nChannels == 1)
486 {
487 switch (m_WaveIDs[WaveID]->wfEXT.Format.wBitsPerSample)
488 {
489 case 4:
490 *pulFormat = pfnGetEnumValue("AL_FORMAT_MONO_IMA4");
491 break;
492 case 8:
493 *pulFormat = pfnGetEnumValue("AL_FORMAT_MONO8");
494 break;
495 case 16:
496 *pulFormat = pfnGetEnumValue("AL_FORMAT_MONO16");
497 break;
498 }
499 }
500 else if (m_WaveIDs[WaveID]->wfEXT.Format.nChannels == 2)
501 {
502 switch (m_WaveIDs[WaveID]->wfEXT.Format.wBitsPerSample)
503 {
504 case 4:
505 *pulFormat = pfnGetEnumValue("AL_FORMAT_STEREO_IMA4");
506 break;
507 case 8:
508 *pulFormat = pfnGetEnumValue("AL_FORMAT_STEREO8");
509 break;
510 case 16:
511 *pulFormat = pfnGetEnumValue("AL_FORMAT_STEREO16");
512 break;
513 }
514 }
515 else if ((m_WaveIDs[WaveID]->wfEXT.Format.nChannels == 4) && (m_WaveIDs[WaveID]->wfEXT.Format.wBitsPerSample == 16))
516 *pulFormat = pfnGetEnumValue("AL_FORMAT_QUAD16");
517 }
518 else if (m_WaveIDs[WaveID]->wfType == WF_EXT)
519 {
520 if ((m_WaveIDs[WaveID]->wfEXT.Format.nChannels == 1) &&
521 ((m_WaveIDs[WaveID]->wfEXT.dwChannelMask == SPEAKER_FRONT_CENTER) ||
522 (m_WaveIDs[WaveID]->wfEXT.dwChannelMask == (SPEAKER_FRONT_LEFT|SPEAKER_FRONT_RIGHT)) ||
523 (m_WaveIDs[WaveID]->wfEXT.dwChannelMask == 0)))
524 {
525 switch (m_WaveIDs[WaveID]->wfEXT.Format.wBitsPerSample)
526 {
527 case 4:
528 *pulFormat = pfnGetEnumValue("AL_FORMAT_MONO_IMA4");
529 break;
530 case 8:
531 *pulFormat = pfnGetEnumValue("AL_FORMAT_MONO8");
532 break;
533 case 16:
534 *pulFormat = pfnGetEnumValue("AL_FORMAT_MONO16");
535 break;
536 }
537 }
538 else if ((m_WaveIDs[WaveID]->wfEXT.Format.nChannels == 2) && (m_WaveIDs[WaveID]->wfEXT.dwChannelMask == (SPEAKER_FRONT_LEFT|SPEAKER_FRONT_RIGHT)))
539 {
540 switch (m_WaveIDs[WaveID]->wfEXT.Format.wBitsPerSample)
541 {
542 case 4:
543 *pulFormat = pfnGetEnumValue("AL_FORMAT_STEREO_IMA4");
544 break;
545 case 8:
546 *pulFormat = pfnGetEnumValue("AL_FORMAT_STEREO8");
547 break;
548 case 16:
549 *pulFormat = pfnGetEnumValue("AL_FORMAT_STEREO16");
550 break;
551 }
552 }
553 else if ((m_WaveIDs[WaveID]->wfEXT.Format.nChannels == 2) && (m_WaveIDs[WaveID]->wfEXT.Format.wBitsPerSample == 16) && (m_WaveIDs[WaveID]->wfEXT.dwChannelMask == (SPEAKER_BACK_LEFT|SPEAKER_BACK_RIGHT)))
554 *pulFormat = pfnGetEnumValue("AL_FORMAT_REAR16");
555 else if ((m_WaveIDs[WaveID]->wfEXT.Format.nChannels == 4) && (m_WaveIDs[WaveID]->wfEXT.Format.wBitsPerSample == 16) && (m_WaveIDs[WaveID]->wfEXT.dwChannelMask == (SPEAKER_FRONT_LEFT|SPEAKER_FRONT_RIGHT|SPEAKER_BACK_LEFT|SPEAKER_BACK_RIGHT)))

557 else if ((m_WaveIDs[WaveID]->wfEXT.Format.nChannels == 6) && (m_WaveIDs[WaveID]->wfEXT.Format.wBitsPerSample == 16) && (m_WaveIDs[WaveID]->wfEXT.dwChannelMask == (SPEAKER_FRONT_LEFT|SPEAKER_FRONT_RIGHT|SPEAKER_FRONT_CENTER|SPEAKER_LOW_FREQUENCY|SPEAKER_BACK_LEFT|SPEAKER_BACK_RIGHT)))
558 *pulFormat = pfnGetEnumValue("AL_FORMAT_51CHN16");
559 else if ((m_WaveIDs[WaveID]->wfEXT.Format.nChannels == 7) && (m_WaveIDs[WaveID]->wfEXT.Format.wBitsPerSample == 16) && (m_WaveIDs[WaveID]->wfEXT.dwChannelMask == (SPEAKER_FRONT_LEFT|SPEAKER_FRONT_RIGHT|SPEAKER_FRONT_CENTER|SPEAKER_LOW_FREQUENCY|SPEAKER_BACK_LEFT|SPEAKER_BACK_RIGHT|SPEAKER_BACK_CENTER)))

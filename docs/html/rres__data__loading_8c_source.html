46//------------------------------------------------------------------------------------
47// Program main entry point
48//------------------------------------------------------------------------------------
49int main(void)
50{
51 // Initialization
52 //--------------------------------------------------------------------------------------
53 const int screenWidth = 384;
54 const int screenHeight = 512;
55
56 InitWindow(screenWidth, screenHeight, "rres example - rres data loading");
57
58 void *data = NULL; // Store RRES_DATA_RAW loaded data
59 char *text = NULL; // Store RRES_DATA_TEXT loaded data
60 Texture2D texture = { 0 }; // Store RRES_DATA_IMAGE loaded data -> LoadTextureFromImage()
61 Sound sound = { 0 }; // Store RRES_DATA_WAVE loaded data -> LoadSoundFromWave()
62 Font font = { 0 }; // Store RRES_DATA_FONT_GLYPHS + RRES_DATA_IMAGE
63
64 // Load content from rres file
65 rresResourceChunk chunk = { 0 }; // Single resource chunk
66 rresResourceMulti multi = { 0 }; // Multiple resource chunks
67
68 InitAudioDevice(); // Initialize audio device, useful for audio testing
69
70 SetTargetFPS(60); // Set our game to run at 60 frames-per-second
71 //--------------------------------------------------------------------------------------
72
73 // Main game loop
74 while (!WindowShouldClose()) // Detect window close button or ESC key
75 {
76 // Dropped files logic
77 //----------------------------------------------------------------------------------
78 if (IsFileDropped())
79 {
80 FilePathList droppedFiles = LoadDroppedFiles();
81
82 if (IsFileExtension(droppedFiles.paths[0], ".rres"))
83 {
84 int result = 0; // Result of data unpacking
85
86 // TEST 01: Load rres Central Directory (RRES_DATA_DIRECTORY)
87 //------------------------------------------------------------------------------------------------------
88 rresCentralDir dir = rresLoadCentralDirectory(droppedFiles.paths[0]);
89
90 // NOTE: By default central directory is never compressed/encrypted
91
92 // Check if central directory is available
93 // NOTE: CDIR is not mandatory, resources are referenced by its id
94 if (dir.count == 0) TraceLog(LOG_WARNING, "No central directory available in the file");
95 else
96 {
97 // List all files contained on central directory
98 for (unsigned int i = 0; i < dir.count; i++)
99 {
100 TraceLog(LOG_INFO, "RRES: CDIR: File entry %03i: %s | Resource(s) id: 0x%08x | Offset: 0x%08x", i + 1, dir.entries[i].fileName, dir.entries[i].id, dir.entries[i].offset);
101
102 // TODO: List contained resource chunks info
103 //rresResourceChunkInfo info = rresGetResourceChunkInfo(droppedFiles.paths[0], dir.entries[i]);
104 }
105 }
106 //------------------------------------------------------------------------------------------------------
107 /*
108 // TEST 02: Loading raw data (RRES_DATA_RAW)
109 //------------------------------------------------------------------------------------------------------
110 chunk = rresLoadResourceChunk(droppedFiles.paths[0], rresGetResourceId(dir, "resources/image.png.raw"));
111 result = UnpackResourceChunk(&chunk); // Decompres/decipher resource data (if required)
112
113 if (result == 0) // Data decompressed/decrypted successfully
114 {
115 unsigned int dataSize = 0;
116 data = LoadDataFromResource(chunk, &dataSize); // Load raw data, must be freed at the end
117
118 if ((data != NULL) && (dataSize > 0))
119 {
120 FILE *rawFile = fopen("export_data.raw", "wb");
121 fwrite(data, 1, dataSize, rawFile);
122 fclose(rawFile);
123 }
124 }
125
126 rresUnloadResourceChunk(chunk);
127 //------------------------------------------------------------------------------------------------------
128
129 // TEST 03: Load text data (RRES_DATA_TEXT)
130 //------------------------------------------------------------------------------------------------------
131 chunk = rresLoadResourceChunk(droppedFiles.paths[0], rresGetResourceId(dir, "resources/text_data.txt"));
132 result = UnpackResourceChunk(&chunk); // Decompres/decipher resource data (if required)
133
134 if (result == 0) // Data decompressed/decrypted successfully
135 {
136 text = LoadTextFromResource(chunk); // Load text data, must be freed at the end
137 }
138
139 rresUnloadResourceChunk(chunk);
140 //------------------------------------------------------------------------------------------------------
141 */
142
143 // TEST 04: Load image data (RRES_DATA_IMAGE)
144 //------------------------------------------------------------------------------------------------------
145 chunk = rresLoadResourceChunk(droppedFiles.paths[0], rresGetResourceId(dir, "fudesumi.png"));
146 result = UnpackResourceChunk(&chunk); // Decompres/decipher resource data (if required)
147
148 if (result == 0) // Data decompressed/decrypted successfully
149 {
150 Image image = LoadImageFromResource(chunk);
151 if (image.data != NULL)
152 {
153 texture = LoadTextureFromImage(image);
154 UnloadImage(image);
155 }
156 }
157
158 rresUnloadResourceChunk(chunk);
159 //------------------------------------------------------------------------------------------------------
160
161 // TEST 05: Load wave data (RRES_DATA_WAVE)
162 //------------------------------------------------------------------------------------------------------
163 chunk = rresLoadResourceChunk(droppedFiles.paths[0], rresGetResourceId(dir, "tanatana.ogg"));
164 result = UnpackResourceChunk(&chunk); // Decompres/decipher resource data (if required)
165
166 if (result == 0) // Data decompressed/decrypted successfully
167 {
168 Wave wave = LoadWaveFromResource(chunk);
169 sound = LoadSoundFromWave(wave);
170 UnloadWave(wave);
171 }
172
173 rresUnloadResourceChunk(chunk);
174 //------------------------------------------------------------------------------------------------------
175
176 // TEST 06: Load font data, multiples chunks (RRES_DATA_FONT_GLYPHS + RRE_DATA_IMAGE)
177 //------------------------------------------------------------------------------------------------------
178 multi = rresLoadResourceMulti(droppedFiles.paths[0], rresGetResourceId(dir, "pixantiqua.ttf"));
179 for (unsigned int i = 0; i < multi.count; i++)
180 {
181 result = UnpackResourceChunk(&multi.chunks[i]); // Decompres/decipher resource data (if required)
182 if (result != 0) break;
183 }
184
185 if (result == 0) // All resources data decompressed/decrypted successfully
186 {
187 font = LoadFontFromResource(multi);
188 }
189
190 rresUnloadResourceMulti(multi);
191 //------------------------------------------------------------------------------------------------------
192
193 // Unload central directory info, not required any more
194 rresUnloadCentralDirectory(dir);
195 }
196
197 UnloadDroppedFiles(droppedFiles); // Unload filepaths from memory
198 }
199 //----------------------------------------------------------------------------------
200
201 // Update
202 //----------------------------------------------------------------------------------
203 // Play audio loaded from wave from .rres: RRES_DATA_WAVE
204 if (IsKeyPressed(KEY_SPACE)) PlaySound(sound);
205 //----------------------------------------------------------------------------------
206
207 // Draw
208 //----------------------------------------------------------------------------------
209 BeginDrawing();
210
211 ClearBackground(RAYWHITE);
212
213 DrawText("rres file loading: drag & drop a .rres file", 10, 10, 10, DARKGRAY);
214
215 // Draw texture loaded from image from .rres: RRES_DATA_IMAGE
216 DrawTexture(texture, 0, 0, WHITE);
217
218 // Draw text using font loaded from .rres: RRES_DATA_FONT_GLYPHS + RRES_DATA_IMAGE
219 DrawTextEx(font, "THIS IS a TEST!", (Vector2) { 10, 50 }, (float)font.baseSize, 0, RED);
220
221 EndDrawing();
222 //----------------------------------------------------------------------------------
223 }
224
225 // De-Initialization
226 //--------------------------------------------------------------------------------------
227 MemFree(data); // Unload raw data, using raylib memory allocator (same used by rres-raylib.h)
228 MemFree(text); // Unload text data, using raylib memory allocator (same used by rres-raylib.h)
229 UnloadTexture(texture); // Unload texture (VRAM)
230 UnloadSound(sound); // Unload sound
231 UnloadFont(font); // Unload font
232
233 CloseAudioDevice(); // Close audio device
234
235 CloseWindow(); // Close window and OpenGL context
236 //--------------------------------------------------------------------------------------
237
238 return 0;
239}

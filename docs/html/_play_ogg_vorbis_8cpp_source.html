67int main()
68{
69 ALuint uiBuffers[NUMBUFFERS];
70 ALuint uiSource;
71 ALuint uiBuffer;
72 ALint iState;
73 ALint iLoop;
74 ALint iBuffersProcessed, iTotalBuffersProcessed, iQueuedBuffers;
75 unsigned long ulFrequency = 0;
76 unsigned long ulFormat = 0;
77 unsigned long ulChannels = 0;
78 unsigned long ulBufferSize;
79 unsigned long ulBytesWritten;
80 char *pDecodeBuffer;
81
82 // Initialize Framework
83 ALFWInit();
84
85 ALFWprintf("PlayOggVorbis Test Application\n");
86
87 // Initialize OggVorbis libary
88 InitVorbisFile();
89 if (!g_bVorbisInit)
90 {
91 ALFWprintf("Failed to find OggVorbis DLLs (vorbisfile.dll, ogg.dll, or vorbis.dll)\n");
92 ALFWShutdown();
93 return 0;
94 }
95
96 // Initialise OpenAL
97 if (!ALFWInitOpenAL())
98 {
99 ALFWprintf("Failed to initialize OpenAL\n");
100 ALFWShutdown();
101 return 0;
102 }
103
104 // Open Ogg Stream
105 ov_callbacks sCallbacks;
106 OggVorbis_File sOggVorbisFile;
107 vorbis_info *psVorbisInfo;
108
109 sCallbacks.read_func = ov_read_func;
110 sCallbacks.seek_func = ov_seek_func;
111 sCallbacks.close_func = ov_close_func;
112 sCallbacks.tell_func = ov_tell_func;
113
114 // Open the OggVorbis file
115 FILE *pOggVorbisFile = fopen(ALFWaddMediaPath(TEST_OGGVORBIS_FILE), "rb");
116 if (!pOggVorbisFile)
117 {
118 ALFWprintf("Could not find %s\n", ALFWaddMediaPath(TEST_OGGVORBIS_FILE));
119 ShutdownVorbisFile();
120 ALFWShutdownOpenAL();
121 ALFWShutdown();
122 return 0;
123 }
124
125 // Create an OggVorbis file stream
126 if (fn_ov_open_callbacks(pOggVorbisFile, &sOggVorbisFile, NULL, 0, sCallbacks) == 0)
127 {
128 // Get some information about the file (Channels, Format, and Frequency)
129 psVorbisInfo = fn_ov_info(&sOggVorbisFile, -1);
130 if (psVorbisInfo)
131 {
132 ulFrequency = psVorbisInfo->rate;
133 ulChannels = psVorbisInfo->channels;
134 if (psVorbisInfo->channels == 1)
135 {
136 ulFormat = AL_FORMAT_MONO16;
137 // Set BufferSize to 250ms (Frequency * 2 (16bit) divided by 4 (quarter of a second))
138 ulBufferSize = ulFrequency >> 1;
139 // IMPORTANT : The Buffer Size must be an exact multiple of the BlockAlignment ...
140 ulBufferSize -= (ulBufferSize % 2);
141 }
142 else if (psVorbisInfo->channels == 2)
143 {
144 ulFormat = AL_FORMAT_STEREO16;
145 // Set BufferSize to 250ms (Frequency * 4 (16bit stereo) divided by 4 (quarter of a second))
146 ulBufferSize = ulFrequency;
147 // IMPORTANT : The Buffer Size must be an exact multiple of the BlockAlignment ...
148 ulBufferSize -= (ulBufferSize % 4);
149 }
150 else if (psVorbisInfo->channels == 4)
151 {
152 ulFormat = alGetEnumValue("AL_FORMAT_QUAD16");
153 // Set BufferSize to 250ms (Frequency * 8 (16bit 4-channel) divided by 4 (quarter of a second))
154 ulBufferSize = ulFrequency * 2;
155 // IMPORTANT : The Buffer Size must be an exact multiple of the BlockAlignment ...
156 ulBufferSize -= (ulBufferSize % 8);
157 }
158 else if (psVorbisInfo->channels == 6)
159 {
160 ulFormat = alGetEnumValue("AL_FORMAT_51CHN16");
161 // Set BufferSize to 250ms (Frequency * 12 (16bit 6-channel) divided by 4 (quarter of a second))
162 ulBufferSize = ulFrequency * 3;
163 // IMPORTANT : The Buffer Size must be an exact multiple of the BlockAlignment ...
164 ulBufferSize -= (ulBufferSize % 12);
165 }
166 }
167
168 if (ulFormat != 0)
169 {
170 // Allocate a buffer to be used to store decoded data for all Buffers
171 pDecodeBuffer = (char*)malloc(ulBufferSize);
172 if (!pDecodeBuffer)
173 {
174 ALFWprintf("Failed to allocate memory for decoded OggVorbis data\n");
175 fn_ov_clear(&sOggVorbisFile);
176 ShutdownVorbisFile();
177 ALFWShutdownOpenAL();
178 ALFWShutdown();
179 }
180
181 // Generate some AL Buffers for streaming
182 alGenBuffers( NUMBUFFERS, uiBuffers );
183
184 // Generate a Source to playback the Buffers
185 alGenSources( 1, &uiSource );
186
187 // Fill all the Buffers with decoded audio data from the OggVorbis file
188 for (iLoop = 0; iLoop < NUMBUFFERS; iLoop++)
189 {
190 ulBytesWritten = DecodeOggVorbis(&sOggVorbisFile, pDecodeBuffer, ulBufferSize, ulChannels);
191 if (ulBytesWritten)
192 {
193 alBufferData(uiBuffers[iLoop], ulFormat, pDecodeBuffer, ulBytesWritten, ulFrequency);
194 alSourceQueueBuffers(uiSource, 1, &uiBuffers[iLoop]);
195 }
196 }
197
198 // Start playing source
199 alSourcePlay(uiSource);
200
201 iTotalBuffersProcessed = 0;
202
203 while (!ALFWKeyPress())
204 {
205 Sleep( SERVICE_UPDATE_PERIOD );
206
207 // Request the number of OpenAL Buffers have been processed (played) on the Source
208 iBuffersProcessed = 0;
209 alGetSourcei(uiSource, AL_BUFFERS_PROCESSED, &iBuffersProcessed);
210
211 // Keep a running count of number of buffers processed (for logging purposes only)
212 iTotalBuffersProcessed += iBuffersProcessed;
213 ALFWprintf("Buffers Processed %d\r", iTotalBuffersProcessed);
214
215 // For each processed buffer, remove it from the Source Queue, read next chunk of audio
216 // data from disk, fill buffer with new data, and add it to the Source Queue
217 while (iBuffersProcessed)
218 {
219 // Remove the Buffer from the Queue. (uiBuffer contains the Buffer ID for the unqueued Buffer)
220 uiBuffer = 0;
221 alSourceUnqueueBuffers(uiSource, 1, &uiBuffer);
222
223 // Read more audio data (if there is any)
224 ulBytesWritten = DecodeOggVorbis(&sOggVorbisFile, pDecodeBuffer, ulBufferSize, ulChannels);
225 if (ulBytesWritten)
226 {
227 alBufferData(uiBuffer, ulFormat, pDecodeBuffer, ulBytesWritten, ulFrequency);
228 alSourceQueueBuffers(uiSource, 1, &uiBuffer);
229 }
230
231 iBuffersProcessed--;
232 }
233
234 // Check the status of the Source. If it is not playing, then playback was completed,
235 // or the Source was starved of audio data, and needs to be restarted.
236 alGetSourcei(uiSource, AL_SOURCE_STATE, &iState);
237 if (iState != AL_PLAYING)
238 {
239 // If there are Buffers in the Source Queue then the Source was starved of audio
240 // data, so needs to be restarted (because there is more audio data to play)
241 alGetSourcei(uiSource, AL_BUFFERS_QUEUED, &iQueuedBuffers);
242 if (iQueuedBuffers)
243 {
244 alSourcePlay(uiSource);
245 }
246 else
247 {
248 // Finished playing
249 break;
250 }
251 }
252 }
253
254 // Stop the Source and clear the Queue
255 alSourceStop(uiSource);
256 alSourcei(uiSource, AL_BUFFER, 0);
257
258 if (pDecodeBuffer)
259 {
260 free(pDecodeBuffer);
261 pDecodeBuffer = NULL;
262 }
263
264 // Clean up buffers and sources
265 alDeleteSources( 1, &uiSource );
266 alDeleteBuffers( NUMBUFFERS, uiBuffers );
267 }
268 else
269 {
270 ALFWprintf("Failed to find format information, or unsupported format\n");
271 }
272
273 // Close OggVorbis stream
274 fn_ov_clear(&sOggVorbisFile);
275 }
276
277 // Shutdown VorbisFile Library
278 ShutdownVorbisFile();
279
280 // Shutdown AL
281 ALFWShutdownOpenAL();
282
283 // Shutdown Framework
284 ALFWShutdown();
285
286 return 0;
287}

289void InitVorbisFile()
290{
291 if (g_bVorbisInit)
292 return;
293
294 // Try and load Vorbis DLLs (VorbisFile.dll will load ogg.dll and vorbis.dll)
295 g_hVorbisFileDLL = LoadLibrary("vorbisfile.dll");
296 if (g_hVorbisFileDLL)
297 {
298 fn_ov_clear = (LPOVCLEAR)GetProcAddress(g_hVorbisFileDLL, "ov_clear");
299 fn_ov_read = (LPOVREAD)GetProcAddress(g_hVorbisFileDLL, "ov_read");
300 fn_ov_pcm_total = (LPOVPCMTOTAL)GetProcAddress(g_hVorbisFileDLL, "ov_pcm_total");
301 fn_ov_info = (LPOVINFO)GetProcAddress(g_hVorbisFileDLL, "ov_info");
302 fn_ov_comment = (LPOVCOMMENT)GetProcAddress(g_hVorbisFileDLL, "ov_comment");
303 fn_ov_open_callbacks = (LPOVOPENCALLBACKS)GetProcAddress(g_hVorbisFileDLL, "ov_open_callbacks");
304
305 if (fn_ov_clear && fn_ov_read && fn_ov_pcm_total && fn_ov_info &&
306 fn_ov_comment && fn_ov_open_callbacks)
307 {
308 g_bVorbisInit = true;
309 }
310 }
311}

323unsigned long DecodeOggVorbis(OggVorbis_File *psOggVorbisFile, char *pDecodeBuffer, unsigned long ulBufferSize, unsigned long ulChannels)
324{
325 int current_section;
326 long lDecodeSize;
327 unsigned long ulSamples;
328 short *pSamples;
329
330 unsigned long ulBytesDone = 0;
331 while (1)
332 {
333 lDecodeSize = fn_ov_read(psOggVorbisFile, pDecodeBuffer + ulBytesDone, ulBufferSize - ulBytesDone, 0, 2, 1, &current_section);
334 if (lDecodeSize > 0)
335 {
336 ulBytesDone += lDecodeSize;
337
338 if (ulBytesDone >= ulBufferSize)
339 break;
340 }
341 else
342 {
343 break;
344 }
345 }
346
347 // Mono, Stereo and 4-Channel files decode into the same channel order as WAVEFORMATEXTENSIBLE,
348 // however 6-Channels files need to be re-ordered
349 if (ulChannels == 6)
350 {
351 pSamples = (short*)pDecodeBuffer;
352 for (ulSamples = 0; ulSamples < (ulBufferSize>>1); ulSamples+=6)
353 {
354 // WAVEFORMATEXTENSIBLE Order : FL, FR, FC, LFE, RL, RR
355 // OggVorbis Order : FL, FC, FR, RL, RR, LFE
356 Swap(pSamples[ulSamples+1], pSamples[ulSamples+2]);
357 Swap(pSamples[ulSamples+3], pSamples[ulSamples+5]);
358 Swap(pSamples[ulSamples+4], pSamples[ulSamples+5]);
359 }
360 }
361
362 return ulBytesDone;
363}

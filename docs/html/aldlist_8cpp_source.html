33ALDeviceList::ALDeviceList()
34{
35 ALDEVICEINFO ALDeviceInfo;
36 char *devices;
37 int index;
38 const char *defaultDeviceName;
39 const char *actualDeviceName;
40
41 // DeviceInfo vector stores, for each enumerated device, it's device name, selection status, spec version #, and extension support
42 vDeviceInfo.empty();
43 vDeviceInfo.reserve(10);
44
45 defaultDeviceIndex = 0;
46
47 // grab function pointers for 1.0-API functions, and if successful proceed to enumerate all devices
48 if (LoadOAL10Library(NULL, &ALFunction) == TRUE) {
49 if (ALFunction.alcIsExtensionPresent(NULL, "ALC_ENUMERATION_EXT")) {
50 devices = (char *)ALFunction.alcGetString(NULL, ALC_DEVICE_SPECIFIER);
51 defaultDeviceName = (char *)ALFunction.alcGetString(NULL, ALC_DEFAULT_DEVICE_SPECIFIER);
52 index = 0;
53 // go through device list (each device terminated with a single NULL, list terminated with double NULL)
54 while (*devices != NULL) {
55 if (strcmp(defaultDeviceName, devices) == 0) {
56 defaultDeviceIndex = index;
57 }
58 ALCdevice *device = ALFunction.alcOpenDevice(devices);
59 if (device) {
60 ALCcontext *context = ALFunction.alcCreateContext(device, NULL);
61 if (context) {
62 ALFunction.alcMakeContextCurrent(context);
63 // if new actual device name isn't already in the list, then add it...
64 actualDeviceName = ALFunction.alcGetString(device, ALC_DEVICE_SPECIFIER);
65 bool bNewName = true;
66 for (int i = 0; i < GetNumDevices(); i++) {
67 if (strcmp(GetDeviceName(i), actualDeviceName) == 0) {
68 bNewName = false;
69 }
70 }
71 if ((bNewName) && (actualDeviceName != NULL) && (strlen(actualDeviceName) > 0)) {
72 memset(&ALDeviceInfo, 0, sizeof(ALDEVICEINFO));
73 ALDeviceInfo.bSelected = true;
74 ALDeviceInfo.strDeviceName = actualDeviceName;
75 ALFunction.alcGetIntegerv(device, ALC_MAJOR_VERSION, sizeof(int), &ALDeviceInfo.iMajorVersion);
76 ALFunction.alcGetIntegerv(device, ALC_MINOR_VERSION, sizeof(int), &ALDeviceInfo.iMinorVersion);
77
78 ALDeviceInfo.pvstrExtensions = new vector<string>;
79
80 // Check for ALC Extensions
81 if (ALFunction.alcIsExtensionPresent(device, "ALC_EXT_CAPTURE") == AL_TRUE)
82 ALDeviceInfo.pvstrExtensions->push_back("ALC_EXT_CAPTURE");
83 if (ALFunction.alcIsExtensionPresent(device, "ALC_EXT_EFX") == AL_TRUE)
84 ALDeviceInfo.pvstrExtensions->push_back("ALC_EXT_EFX");
85
86 // Check for AL Extensions
87 if (ALFunction.alIsExtensionPresent("AL_EXT_OFFSET") == AL_TRUE)
88 ALDeviceInfo.pvstrExtensions->push_back("AL_EXT_OFFSET");
89
90 if (ALFunction.alIsExtensionPresent("AL_EXT_LINEAR_DISTANCE") == AL_TRUE)
91 ALDeviceInfo.pvstrExtensions->push_back("AL_EXT_LINEAR_DISTANCE");
92 if (ALFunction.alIsExtensionPresent("AL_EXT_EXPONENT_DISTANCE") == AL_TRUE)
93 ALDeviceInfo.pvstrExtensions->push_back("AL_EXT_EXPONENT_DISTANCE");
94
95 if (ALFunction.alIsExtensionPresent("EAX2.0") == AL_TRUE)
96 ALDeviceInfo.pvstrExtensions->push_back("EAX2.0");
97 if (ALFunction.alIsExtensionPresent("EAX3.0") == AL_TRUE)
98 ALDeviceInfo.pvstrExtensions->push_back("EAX3.0");
99 if (ALFunction.alIsExtensionPresent("EAX4.0") == AL_TRUE)
100 ALDeviceInfo.pvstrExtensions->push_back("EAX4.0");
101 if (ALFunction.alIsExtensionPresent("EAX5.0") == AL_TRUE)
102 ALDeviceInfo.pvstrExtensions->push_back("EAX5.0");
103
104 if (ALFunction.alIsExtensionPresent("EAX-RAM") == AL_TRUE)
105 ALDeviceInfo.pvstrExtensions->push_back("EAX-RAM");
106
107 // Get Source Count
108 ALDeviceInfo.uiSourceCount = GetMaxNumSources();
109
110 vDeviceInfo.push_back(ALDeviceInfo);
111 }
112 ALFunction.alcMakeContextCurrent(NULL);
113 ALFunction.alcDestroyContext(context);
114 }
115 ALFunction.alcCloseDevice(device);
116 }
117 devices += strlen(devices) + 1;
118 index += 1;
119 }
120 }
121 }
122
123 ResetFilters();
124}

129ALDeviceList::~ALDeviceList()
130{
131 for (unsigned int i = 0; i < vDeviceInfo.size(); i++) {
132 if (vDeviceInfo[i].pvstrExtensions) {
133 vDeviceInfo[i].pvstrExtensions->empty();
134 delete vDeviceInfo[i].pvstrExtensions;
135 }
136 }
137
138 vDeviceInfo.empty();
139
140 UnloadOAL10Library();

187/*
188 * Checks if the extension is supported on the given device
189 */
190bool ALDeviceList::IsExtensionSupported(int index, char *szExtName)
191{
192 bool bReturn = false;
193
194 if (index < GetNumDevices()) {
195 for (unsigned int i = 0; i < vDeviceInfo[index].pvstrExtensions->size(); i++) {
196 if (!_stricmp(vDeviceInfo[index].pvstrExtensions->at(i).c_str(), szExtName)) {
197 bReturn = true;
198 break;
199 }
200 }
201 }
202
203 return bReturn;
204}

242/*
243 * Deselects device which don't support the given extension name
244 */
245void ALDeviceList::FilterDevicesExtension(char *szExtName)
246{
247 bool bFound;
248
249 for (unsigned int i = 0; i < vDeviceInfo.size(); i++) {
250 bFound = false;
251 for (unsigned int j = 0; j < vDeviceInfo[i].pvstrExtensions->size(); j++) {
252 if (!_stricmp(vDeviceInfo[i].pvstrExtensions->at(j).c_str(), szExtName)) {
253 bFound = true;
254 break;
255 }
256 }
257 if (!bFound)
258 vDeviceInfo[i].bSelected = false;
259 }
260}

305/*
306 * Internal function to detemine max number of Sources that can be generated
307 */
308unsigned int ALDeviceList::GetMaxNumSources()
309{
310 ALuint uiSources[256];
311 unsigned int iSourceCount = 0;
312
313 // Clear AL Error Code
314 ALFunction.alGetError();
315
316 // Generate up to 256 Sources, checking for any errors
317 for (iSourceCount = 0; iSourceCount < 256; iSourceCount++)
318 {
319 ALFunction.alGenSources(1, &uiSources[iSourceCount]);
320 if (ALFunction.alGetError() != AL_NO_ERROR)
321 break;
322 }
323
324 // Release the Sources
325 ALFunction.alDeleteSources(iSourceCount, uiSources);
326 if (ALFunction.alGetError() != AL_NO_ERROR)
327 {
328 for (unsigned int i = 0; i < 256; i++)
329 {
330 ALFunction.alDeleteSources(1, &uiSources[i]);
331 }
332 }
333
334 return iSourceCount;
335}

29ALboolean CreateFilter( ALuint *filter, ALenum filterType, ALfloat gain, ALfloat gainHF)
30{
31 // Clear AL Error
32 alGetError();
33
34 alGenFilters( 1, filter);
35 if (alGetError() != AL_NO_ERROR)
36 return AL_FALSE;
37
38 if( alIsFilter( (*filter) ) )
39 {
40 alFilteri( (*filter), AL_FILTER_TYPE, filterType );
41 if (alGetError() != AL_NO_ERROR)
42 return AL_FALSE;
43 }
44
45 alFilterf((*filter), AL_LOWPASS_GAIN, gain);
46 alFilterf((*filter), AL_LOWPASS_GAINHF, gainHF);
47
48 return AL_TRUE;
49}

51void Play(ALuint uiSource)
52{
53 ALint state;
54
55 // Begin playback
56 alSourcePlay( uiSource );
57
58 while( 1 )
59 {
60 // Get state
61 alGetSourcei( uiSource, AL_SOURCE_STATE, &state);
62 if( state != AL_PLAYING )
63 break;
64 }
65}

75ALboolean PlayDirectFilter(ALuint uiSource)
76{
77 ALuint uiFilter;
78
79 ALFWprintf("Source played through a direct lowpass filter\n");
80
81 if(AL_FALSE == CreateFilter( &uiFilter, AL_FILTER_LOWPASS, 1.0f, 0.5f))
82 {
83 return AL_FALSE;
84 }
85
86 // Assign filter to the source
87 alSourcei(uiSource, AL_DIRECT_FILTER, uiFilter);
88 if (alGetError() != AL_NO_ERROR)
89 return AL_FALSE;
90
91 Play(uiSource);
92
93 // Cleanup
94 alSourcei(uiSource, AL_DIRECT_FILTER, AL_FILTER_NULL);
95 alDeleteFilters( 1, &uiFilter );
96
97 return AL_TRUE;
98}

100ALboolean PlayAuxiliaryNoFilter(ALuint uiSource)
101{
102 ALuint uiEffectSlot;
103 ALuint uiEffect;
104
105 ALFWprintf("Source played through an auxiliary reverb without filtering\n");
106 alGenAuxiliaryEffectSlots( 1, &uiEffectSlot );
107
108 // Create an effect to be loaded in the auxiliary effect slot
109 alGenEffects( 1, &uiEffect );
110
111 // Configure effect to be Reverb
112 if( alIsEffect( uiEffect ) )
113 {
114 alEffecti( uiEffect, AL_EFFECT_TYPE, AL_EFFECT_REVERB );
115
116 // Load effect into slot
117 if( alIsAuxiliaryEffectSlot( uiEffectSlot ) )
118 {
119 alAuxiliaryEffectSloti( uiEffectSlot, AL_EFFECTSLOT_EFFECT, uiEffect );
120 }
121 }
122
123 // Enable Send 0 from the Source to the Auxiliary Effect Slot without filtering
124 alSource3i(uiSource, AL_AUXILIARY_SEND_FILTER, uiEffectSlot, 0, AL_FILTER_NULL);
125 if (alGetError() != AL_NO_ERROR)
126 return AL_FALSE;
127
128 Play(uiSource);
129
130 // Cleanup
131 alSource3i(uiSource, AL_AUXILIARY_SEND_FILTER, AL_EFFECTSLOT_NULL, 0, AL_FILTER_NULL);
132 alDeleteAuxiliaryEffectSlots( 1, &uiEffectSlot );
133 alDeleteEffects( 1, &uiEffect );
134
135 return AL_TRUE;
136}

184ALboolean playStaticWaveFile( const char *filename )
185{
186 ALuint uiSource;
187 ALuint uiBuffer;
188
189 // Check for EFX Extension
190 if (!ALFWIsEFXSupported())
191 {
192 ALFWprintf("Unable to run PlayEFX example without the EFX extension\n");
193 return AL_FALSE;
194 }
195
196 alGenBuffers( 1, &uiBuffer );
197 if (alGetError() != AL_NO_ERROR)
198 return AL_FALSE;
199
200 if (!ALFWLoadWaveToBuffer((char*)ALFWaddMediaPath(filename), uiBuffer))
201 {
202 ALFWprintf("Failed to load or attach %s\n", ALFWaddMediaPath(filename));
203 alDeleteBuffers(1, &uiBuffer);
204 return AL_FALSE;
205 }
206
207 // Generate source object
208 alGenSources( 1, &uiSource );
209
210 alSourcei( uiSource, AL_BUFFER, uiBuffer );
211 if (alGetError() != AL_NO_ERROR)
212 return AL_FALSE;
213
214 // Play the source with no filter
215 PlayDry(uiSource);
216
217 // Play the source with a direct filter
218 PlayDirectFilter(uiSource);
219
220 // Play the source through an auxiliary reverb with no filter
221 PlayAuxiliaryNoFilter(uiSource);
222
223 // Play the source through an auxiliary reverb with an auxiliary filter
224 PlayAuxiliaryFilter(uiSource);
225
226 return AL_TRUE;
227}

44// Reset layout to default values
45static void ResetLayout(GuiLayout* layout)
46{
47 // Set anchor points to default values
48 for (int i = 0; i < MAX_ANCHOR_POINTS; i++)
49 {
50 layout->anchors[i].id = i;
51 layout->anchors[i].x = 0;
52 layout->anchors[i].y = 0;
53 layout->anchors[i].enabled = false;
54 layout->anchors[i].hidding = false;
55 memset(layout->anchors[i].name, 0, MAX_ANCHOR_NAME_LENGTH);
56
57 if (i == 0) strcpy(static_cast<char*>(static_cast<void*>(layout->anchors[i].name)), "refPoint");
58 else strcpy(static_cast<char*>(static_cast<void*>(layout->anchors[layout->anchorCount].name)), TextFormat("anchor%02i", i));
59 }
60
61 // Initialize layout controls data
62 for (int i = 0; i < MAX_GUI_CONTROLS; i++)

64 layout->controls[i].id = 0;
65 layout->controls[i].type = 0;

68 memset(layout->controls[i].name, 0, MAX_CONTROL_NAME_LENGTH);
69 layout->controls[i].ap = &layout->anchors[0]; // By default, set parent anchor
70 }

72 layout->refWindow = Rectangle{ 0, 40, -1, -1 };
73 layout->anchorCount = 0;
74 layout->controlCount = 0;

77// Load layout from text file
78// NOTE: If NULL provided, an empty layout is initialized
79static GuiLayout* LoadLayout(const char* fileName)
80{
81 GuiLayout* layout = (GuiLayout*)RL_CALLOC(1, sizeof(GuiLayout));
82
83 ResetLayout(layout); // Init some default values
84
85 if (fileName != NULL) // Try loading text layout file
86 {
87 FILE* rglFile = fopen(fileName, "rt");
88
89 if (rglFile != NULL)
90 {
91 char buffer[256] = { 0 };
92 char anchorName[MAX_ANCHOR_NAME_LENGTH] = { 0 };
93
94 fgets(buffer, 256, rglFile);
95
96 while (!feof(rglFile))
97 {
98 switch (buffer[0])
99 {
100 case 'r':
101 {
102 sscanf(buffer, "r %f %f %f %f", &layout->refWindow.x, &layout->refWindow.y, &layout->refWindow.width, &layout->refWindow.height);
103
104 // NOTE: Reference window defines anchor[0]
105 layout->anchors[0].id = 0;
106 layout->anchors[0].ap = NULL;
107 layout->anchors[0].x = layout->refWindow.x;
108 layout->anchors[0].y = layout->refWindow.y;
109 layout->anchors[0].enabled = true;
110
111 strcpy(static_cast<char*>(static_cast<void*>(layout->anchors[0].name)), "refPoint");
112
113 layout->anchorCount++;
114 } break;
115 case 'a':
116 {
117 int enabled = 0;
118 sscanf(buffer, "a %d %s %d %d %d",
119 &layout->anchors[layout->anchorCount].id,
120 anchorName,
121 &layout->anchors[layout->anchorCount].x,
122 &layout->anchors[layout->anchorCount].y,
123 &enabled);
124
125 if (layout->anchors[layout->anchorCount].id > 0)
126 {
127 layout->anchors[layout->anchorCount].enabled = (enabled ? true : false);
128
129 strcpy(static_cast<char*>(static_cast<void*>(layout->anchors[layout->anchorCount].name)), anchorName);
130
131 layout->anchorCount++;
132 }
133 } break;
134 case 'c':
135 {
136 int anchorId = 0;
137 sscanf(buffer, "c %d %d %s %f %f %f %f %d %[^\n]s",
138 &layout->controls[layout->controlCount].id,
139 &layout->controls[layout->controlCount].type,
140 layout->controls[layout->controlCount].name,
141 &layout->controls[layout->controlCount].rec.x,
142 &layout->controls[layout->controlCount].rec.y,
143 &layout->controls[layout->controlCount].rec.width,
144 &layout->controls[layout->controlCount].rec.height,
145 &anchorId,
146 layout->controls[layout->controlCount].text);
147
148 layout->controls[layout->controlCount].ap = &layout->anchors[anchorId];
149
150 // NOTE: refWindow offset (anchor[0]) must be added to controls with no anchor
151 if (anchorId == 0)
152 {
153 layout->controls[layout->controlCount].rec.x += layout->refWindow.x;
154 layout->controls[layout->controlCount].rec.y += layout->refWindow.y;
155 }
156
157 layout->controlCount++;
158 } break;
159 default: break;
160 }
161
162 fgets(buffer, 256, rglFile);
163 }
164
165 fclose(rglFile);
166
167 // NOTE: refWindow offset (anchor[0]) must be added to all anchors
168 for (int i = 1; i < layout->anchorCount; i++)
169 {
170 layout->anchors[i].x += layout->refWindow.x;
171 layout->anchors[i].y += layout->refWindow.y;
172 }
173 }
174 }
175 else
176 {
177 // Define reference anchor
178 layout->anchors[0].x = 0;
179 layout->anchors[0].y = 40;
180 layout->anchors[0].enabled = true;
181 layout->refWindow.x = 0;
182 layout->refWindow.y = 40;
183 layout->anchorCount = 1;
184 }
185
186 return layout;
187}

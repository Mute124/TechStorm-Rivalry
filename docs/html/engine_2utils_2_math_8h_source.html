46inline float ArchAlgorithm(float amplitude, float frequency, float time, float steepness, float offset, float rangeScaleValue, float scaleFactor) {
47 float sinValue = sin(time * (frequency / scaleFactor));
48 float powValue = powf(sinValue, steepness);
49 float result = sqrtf(amplitude * powValue * rangeScaleValue) + offset;
50 return result;
51}

59* Get the average value in a dataset;
60*
61* System of equations equation:
62*
63* x = !dataset[];
64* y = x/length
65*
66*/
67template<typename T>
68inline float DataSetAverage(std::vector<T> dataset) {
69 float sum;
70 const int length = DatasetLength<T>(dataset);
71 for (int i = 0; i < length; i++) {
72 sum += dataset[i];
73 }
74
75 return sum / length;

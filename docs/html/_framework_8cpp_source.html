106ALboolean ALFWInitOpenAL()
107{
108 ALDeviceList *pDeviceList = NULL;
109 ALCcontext *pContext = NULL;
110 ALCdevice *pDevice = NULL;
111 ALint i;
112 ALboolean bReturn = AL_FALSE;
113
114 pDeviceList = new ALDeviceList();
115 if ((pDeviceList) && (pDeviceList->GetNumDevices()))
116 {
117 ALFWprintf("\nSelect OpenAL Device:\n");
118 for (i = 0; i < pDeviceList->GetNumDevices(); i++)
119 ALFWprintf("%d. %s%s\n", i + 1, pDeviceList->GetDeviceName(i), i == pDeviceList->GetDefaultDevice() ? "(DEFAULT)" : "");
120
121 do {
122 ALchar ch = _getch();
123 i = atoi(&ch);
124 } while ((i < 1) || (i > pDeviceList->GetNumDevices()));
125
126 pDevice = alcOpenDevice(pDeviceList->GetDeviceName(i - 1));
127 if (pDevice)
128 {
129 pContext = alcCreateContext(pDevice, NULL);
130 if (pContext)
131 {
132 ALFWprintf("\nOpened %s Device\n", alcGetString(pDevice, ALC_DEVICE_SPECIFIER));
133 alcMakeContextCurrent(pContext);
134 bReturn = AL_TRUE;
135 }
136 else
137 {
138 alcCloseDevice(pDevice);
139 }
140 }
141
142 delete pDeviceList;
143 }
144
145 return bReturn;
146}

163ALboolean ALFWLoadWaveToBuffer(const char *szWaveFile, ALuint uiBufferID, ALenum eXRAMBufferMode)
164{
165 WAVEID WaveID;
166 ALint iDataSize, iFrequency;
167 ALenum eBufferFormat;
168 ALchar *pData;
169 ALboolean bReturn;
170
171 bReturn = AL_FALSE;
172 if (g_pWaveLoader)
173 {
174 if (SUCCEEDED(g_pWaveLoader->LoadWaveFile(szWaveFile, &WaveID)))
175 {
176 if ((SUCCEEDED(g_pWaveLoader->GetWaveSize(WaveID, (unsigned long*)&iDataSize))) &&
177 (SUCCEEDED(g_pWaveLoader->GetWaveData(WaveID, (void**)&pData))) &&
178 (SUCCEEDED(g_pWaveLoader->GetWaveFrequency(WaveID, (unsigned long*)&iFrequency))) &&
179 (SUCCEEDED(g_pWaveLoader->GetWaveALBufferFormat(WaveID, &alGetEnumValue, (unsigned long*)&eBufferFormat))))
180 {
181 // Set XRAM Mode (if application)
182 if (eaxSetBufferMode && eXRAMBufferMode)
183 eaxSetBufferMode(1, &uiBufferID, eXRAMBufferMode);
184
185 alGetError();
186 alBufferData(uiBufferID, eBufferFormat, pData, iDataSize, iFrequency);
187 if (alGetError() == AL_NO_ERROR)
188 bReturn = AL_TRUE;
189
190 g_pWaveLoader->DeleteWaveFile(WaveID);
191 }
192 }
193 }
194
195 return bReturn;
196}

220ALboolean ALFWIsXRAMSupported()
221{
222 ALboolean bXRAM = AL_FALSE;
223
224 if (alIsExtensionPresent("EAX-RAM") == AL_TRUE)
225 {
226 // Get X-RAM Function pointers
227 eaxSetBufferMode = (EAXSetBufferMode)alGetProcAddress("EAXSetBufferMode");
228 eaxGetBufferMode = (EAXGetBufferMode)alGetProcAddress("EAXGetBufferMode");
229
230 if (eaxSetBufferMode && eaxGetBufferMode)
231 {
232 eXRAMSize = alGetEnumValue("AL_EAX_RAM_SIZE");
233 eXRAMFree = alGetEnumValue("AL_EAX_RAM_FREE");
234 eXRAMAuto = alGetEnumValue("AL_STORAGE_AUTOMATIC");
235 eXRAMHardware = alGetEnumValue("AL_STORAGE_HARDWARE");
236 eXRAMAccessible = alGetEnumValue("AL_STORAGE_ACCESSIBLE");
237
238 if (eXRAMSize && eXRAMFree && eXRAMAuto && eXRAMHardware && eXRAMAccessible)
239 bXRAM = AL_TRUE;
240 }
241 }
242
243 return bXRAM;
244}

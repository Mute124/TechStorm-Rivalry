117 // an properly processed on shader
118 pbrShader.locs[SHADER_LOC_MAP_EMISSION] = GetShaderLocation(pbrShader, "emissiveMap");
119 pbrShader.locs[SHADER_LOC_COLOR_DIFFUSE] = GetShaderLocation(pbrShader, "albedoColor");
120
121 // Setup additional required shader locations, including lights data
122 pbrShader.locs[SHADER_LOC_VECTOR_VIEW] = GetShaderLocation(pbrShader, "viewPos");
123 lightCountLoc = GetShaderLocation(pbrShader, "numOfLights");
124
125 // set it to the value of the MAX_LIGHTS macro and pass it to the shader

134 SetShaderValue(pbrShader, GetShaderLocation(pbrShader, "ambient"), &ambientIntensity, SHADER_UNIFORM_FLOAT);

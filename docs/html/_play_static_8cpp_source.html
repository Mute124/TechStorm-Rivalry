49
50 // Load Wave file into OpenAL Buffer
51 if (!ALFWLoadWaveToBuffer((char*)ALFWaddMediaPath(TEST_WAVE_FILE), uiBuffer))
52 {
53 ALFWprintf("Failed to load %s\n", ALFWaddMediaPath(TEST_WAVE_FILE));
54 }
55
56 // Generate a Source to playback the Buffer
57 alGenSources( 1, &uiSource );
58
59 // Attach Source to Buffer
60 alSourcei( uiSource, AL_BUFFER, uiBuffer );
61
62 // Play Source
63 alSourcePlay( uiSource );
64 ALFWprintf("Playing Source ");
65
66 do
67 {
68 Sleep(100);

70 // Get Source State

73
74 ALFWprintf("\n");
75
76 // Clean up by deleting Source(s) and Buffer(s)

78 alDeleteSources(1, &uiSource);
79 alDeleteBuffers(1, &uiBuffer);

81 ALFWShutdownOpenAL();
82
83 ALFWShutdown();

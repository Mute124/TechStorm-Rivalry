61* CPU Feature Detection
62**************************************/
63/* LZ4_FORCE_MEMORY_ACCESS
64 * By default, access to unaligned memory is controlled by `memcpy()`, which is safe and portable.
65 * Unfortunately, on some target/compiler combinations, the generated assembly is sub-optimal.
66 * The below switch allow to select different access method for improved performance.
67 * Method 0 (default) : use `memcpy()`. Safe and portable.
68 * Method 1 : `__packed` statement. It depends on compiler extension (ie, not portable).
69 * This method is safe if your compiler supports it, and *generally* as fast or faster than `memcpy`.
70 * Method 2 : direct access. This method is portable but violate C standard.
71 * It can generate buggy code on targets which assembly generation depends on alignment.
72 * But in some circumstances, it's the only known way to get the most performance (ie GCC + ARMv6)
73 * See https://fastcompression.blogspot.fr/2015/08/accessing-unaligned-memory.html for details.
74 * Prefer these methods in priority order (0 > 1 > 2)
75 */
76#ifndef LZ4_FORCE_MEMORY_ACCESS /* can be defined externally */
77# if defined(__GNUC__) && \
78 ( defined(__ARM_ARCH_6__) || defined(__ARM_ARCH_6J__) || defined(__ARM_ARCH_6K__) \
79 || defined(__ARM_ARCH_6Z__) || defined(__ARM_ARCH_6ZK__) || defined(__ARM_ARCH_6T2__) )
80# define LZ4_FORCE_MEMORY_ACCESS 2
81# elif (defined(__INTEL_COMPILER) && !defined(_WIN32)) || defined(__GNUC__)

145/* LZ4_FORCE_O2 and LZ4_FORCE_INLINE
146 * gcc on ppc64le generates an unrolled SIMDized loop for LZ4_wildCopy8,
147 * together with a simple 8-byte copy loop as a fall-back path.
148 * However, this optimization hurts the decompression speed by >30%,
149 * because the execution does not go to the optimized loop
150 * for typical compressible data, and all of the preamble checks

386{

406/* customized variant of memcpy, which can overwrite up to 8 bytes beyond dstEnd */
407LZ4_FORCE_INLINE
408void LZ4_wildCopy8(void* dstPtr, const void* srcPtr, void* dstEnd)
409{
410 BYTE* d = (BYTE*)dstPtr;
411 const BYTE* s = (const BYTE*)srcPtr;
412 BYTE* const e = (BYTE*)dstEnd;
413
414 do { LZ4_memcpy(d,s,8); d+=8; s+=8; } while (d<e);
415}

438LZ4_FORCE_INLINE void
439LZ4_memcpy_using_offset_base(BYTE* dstPtr, const BYTE* srcPtr, BYTE* dstEnd, const size_t offset)
440{
441 assert(srcPtr + offset == dstPtr);
442 if (offset < 8) {
443 LZ4_write32(dstPtr, 0); /* silence an msan warning when offset==0 */
444 dstPtr[0] = srcPtr[0];
445 dstPtr[1] = srcPtr[1];
446 dstPtr[2] = srcPtr[2];
447 dstPtr[3] = srcPtr[3];
448 srcPtr += inc32table[offset];
449 LZ4_memcpy(dstPtr+4, srcPtr, 4);
450 srcPtr -= dec64table[offset];
451 dstPtr += 8;
452 } else {
453 LZ4_memcpy(dstPtr, srcPtr, 8);
454 dstPtr += 8;
455 srcPtr += 8;
456 }
457
458 LZ4_wildCopy8(dstPtr, srcPtr, dstEnd);
459}

461/* customized variant of memcpy, which can overwrite up to 32 bytes beyond dstEnd
462 * this version copies two times 16 bytes (instead of one time 32 bytes)
463 * because it must be compatible with offsets >= 16. */
464LZ4_FORCE_INLINE void
465LZ4_wildCopy32(void* dstPtr, const void* srcPtr, void* dstEnd)
466{
467 BYTE* d = (BYTE*)dstPtr;
468 const BYTE* s = (const BYTE*)srcPtr;
469 BYTE* const e = (BYTE*)dstEnd;
470
471 do { LZ4_memcpy(d,s,16); LZ4_memcpy(d+16,s+16,16); d+=32; s+=32; } while (d<e);
472}

474/* LZ4_memcpy_using_offset() presumes :
475 * - dstEnd >= dstPtr + MINMATCH
476 * - there is at least 8 bytes available to write after dstEnd */
477LZ4_FORCE_INLINE void
478LZ4_memcpy_using_offset(BYTE* dstPtr, const BYTE* srcPtr, BYTE* dstEnd, const size_t offset)
479{
480 BYTE v[8];
481
482 assert(dstEnd >= dstPtr + MINMATCH);
483
484 switch(offset) {
485 case 1:
486 MEM_INIT(v, *srcPtr, 8);
487 break;
488 case 2:
489 LZ4_memcpy(v, srcPtr, 2);
490 LZ4_memcpy(&v[2], srcPtr, 2);
491 LZ4_memcpy(&v[4], v, 4);
492 break;
493 case 4:
494 LZ4_memcpy(v, srcPtr, 4);
495 LZ4_memcpy(&v[4], srcPtr, 4);
496 break;
497 default:
498 LZ4_memcpy_using_offset_base(dstPtr, srcPtr, dstEnd, offset);
499 return;
500 }
501
502 LZ4_memcpy(dstPtr, v, 8);
503 dstPtr += 8;
504 while (dstPtr < dstEnd) {
505 LZ4_memcpy(dstPtr, v, 8);
506 dstPtr += 8;
507 }
508}

515static unsigned LZ4_NbCommonBytes (reg_t val)
516{
517 assert(val != 0);
518 if (LZ4_isLittleEndian()) {
519 if (sizeof(val) == 8) {
520# if defined(_MSC_VER) && (_MSC_VER >= 1800) && (defined(_M_AMD64) && !defined(_M_ARM64EC)) && !defined(LZ4_FORCE_SW_BITCOUNT)
521/*-*************************************************************************************************
522* ARM64EC is a Microsoft-designed ARM64 ABI compatible with AMD64 applications on ARM64 Windows 11.
523* The ARM64EC ABI does not support AVX/AVX2/AVX512 instructions, nor their relevant intrinsics
524* including _tzcnt_u64. Therefore, we need to neuter the _tzcnt_u64 code path for ARM64EC.
525****************************************************************************************************/
526# if defined(__clang__) && (__clang_major__ < 10)
527 /* Avoid undefined clang-cl intrinics issue.
528 * See https://github.com/lz4/lz4/pull/1017 for details. */
529 return (unsigned)__builtin_ia32_tzcnt_u64(val) >> 3;
530# else
531 /* x64 CPUS without BMI support interpret `TZCNT` as `REP BSF` */
532 return (unsigned)_tzcnt_u64(val) >> 3;
533# endif
534# elif defined(_MSC_VER) && defined(_WIN64) && !defined(LZ4_FORCE_SW_BITCOUNT)
535 unsigned long r = 0;
536 _BitScanForward64(&r, (U64)val);
537 return (unsigned)r >> 3;
538# elif (defined(__clang__) || (defined(__GNUC__) && ((__GNUC__ > 3) || \
539 ((__GNUC__ == 3) && (__GNUC_MINOR__ >= 4))))) && \
540 !defined(LZ4_FORCE_SW_BITCOUNT)
541 return (unsigned)__builtin_ctzll((U64)val) >> 3;
542# else
543 const U64 m = 0x0101010101010101ULL;
544 val ^= val - 1;
545 return (unsigned)(((U64)((val & (m - 1)) * m)) >> 56);
546# endif
547 } else /* 32 bits */ {
548# if defined(_MSC_VER) && (_MSC_VER >= 1400) && !defined(LZ4_FORCE_SW_BITCOUNT)
549 unsigned long r;
550 _BitScanForward(&r, (U32)val);
551 return (unsigned)r >> 3;
552# elif (defined(__clang__) || (defined(__GNUC__) && ((__GNUC__ > 3) || \
553 ((__GNUC__ == 3) && (__GNUC_MINOR__ >= 4))))) && \
554 !defined(__TINYC__) && !defined(LZ4_FORCE_SW_BITCOUNT)
555 return (unsigned)__builtin_ctz((U32)val) >> 3;
556# else
557 const U32 m = 0x01010101;
558 return (unsigned)((((val - 1) ^ val) & (m - 1)) * m) >> 24;
559# endif
560 }
561 } else /* Big Endian CPU */ {
562 if (sizeof(val)==8) {
563# if (defined(__clang__) || (defined(__GNUC__) && ((__GNUC__ > 3) || \
564 ((__GNUC__ == 3) && (__GNUC_MINOR__ >= 4))))) && \
565 !defined(__TINYC__) && !defined(LZ4_FORCE_SW_BITCOUNT)
566 return (unsigned)__builtin_clzll((U64)val) >> 3;
567# else
568#if 1
569 /* this method is probably faster,
570 * but adds a 128 bytes lookup table */
571 static const unsigned char ctz7_tab[128] = {
572 7, 0, 1, 0, 2, 0, 1, 0, 3, 0, 1, 0, 2, 0, 1, 0,
573 4, 0, 1, 0, 2, 0, 1, 0, 3, 0, 1, 0, 2, 0, 1, 0,
574 5, 0, 1, 0, 2, 0, 1, 0, 3, 0, 1, 0, 2, 0, 1, 0,
575 4, 0, 1, 0, 2, 0, 1, 0, 3, 0, 1, 0, 2, 0, 1, 0,
576 6, 0, 1, 0, 2, 0, 1, 0, 3, 0, 1, 0, 2, 0, 1, 0,
577 4, 0, 1, 0, 2, 0, 1, 0, 3, 0, 1, 0, 2, 0, 1, 0,
578 5, 0, 1, 0, 2, 0, 1, 0, 3, 0, 1, 0, 2, 0, 1, 0,
579 4, 0, 1, 0, 2, 0, 1, 0, 3, 0, 1, 0, 2, 0, 1, 0,
580 };
581 U64 const mask = 0x0101010101010101ULL;
582 U64 const t = (((val >> 8) - mask) | val) & mask;
583 return ctz7_tab[(t * 0x0080402010080402ULL) >> 57];
584#else
585 /* this method doesn't consume memory space like the previous one,
586 * but it contains several branches,
587 * that may end up slowing execution */
588 static const U32 by32 = sizeof(val)*4; /* 32 on 64 bits (goal), 16 on 32 bits.
589 Just to avoid some static analyzer complaining about shift by 32 on 32-bits target.
590 Note that this code path is never triggered in 32-bits mode. */
591 unsigned r;
592 if (!(val>>by32)) { r=4; } else { r=0; val>>=by32; }
593 if (!(val>>16)) { r+=2; val>>=8; } else { val>>=24; }
594 r += (!val);
595 return r;
596#endif
597# endif
598 } else /* 32 bits */ {
599# if (defined(__clang__) || (defined(__GNUC__) && ((__GNUC__ > 3) || \
600 ((__GNUC__ == 3) && (__GNUC_MINOR__ >= 4))))) && \
601 !defined(LZ4_FORCE_SW_BITCOUNT)
602 return (unsigned)__builtin_clz((U32)val) >> 3;
603# else
604 val >>= 8;
605 val = ((((val + 0x00FFFF00) | 0x00FFFFFF) + val) |
606 (val + 0x00FF0000)) >> 24;
607 return (unsigned)val ^ 3;
608# endif
609 }
610 }
611}

614#define STEPSIZE sizeof(reg_t)
615LZ4_FORCE_INLINE
616unsigned LZ4_count(const BYTE* pIn, const BYTE* pMatch, const BYTE* pInLimit)
617{
618 const BYTE* const pStart = pIn;
619
620 if (likely(pIn < pInLimit-(STEPSIZE-1))) {
621 reg_t const diff = LZ4_read_ARCH(pMatch) ^ LZ4_read_ARCH(pIn);
622 if (!diff) {
623 pIn+=STEPSIZE; pMatch+=STEPSIZE;
624 } else {
625 return LZ4_NbCommonBytes(diff);
626 } }
627
628 while (likely(pIn < pInLimit-(STEPSIZE-1))) {
629 reg_t const diff = LZ4_read_ARCH(pMatch) ^ LZ4_read_ARCH(pIn);
630 if (!diff) { pIn+=STEPSIZE; pMatch+=STEPSIZE; continue; }
631 pIn += LZ4_NbCommonBytes(diff);
632 return (unsigned)(pIn - pStart);
633 }
634
635 if ((STEPSIZE==8) && (pIn<(pInLimit-3)) && (LZ4_read32(pMatch) == LZ4_read32(pIn))) { pIn+=4; pMatch+=4; }
636 if ((pIn<(pInLimit-1)) && (LZ4_read16(pMatch) == LZ4_read16(pIn))) { pIn+=2; pMatch+=2; }
637 if ((pIn<pInLimit) && (*pMatch == *pIn)) pIn++;
638 return (unsigned)(pIn - pStart);
639}

711LZ4_FORCE_INLINE U32 LZ4_hash4(U32 sequence, tableType_t const tableType)
712{
713 if (tableType == byU16)
714 return ((sequence * 2654435761U) >> ((MINMATCH*8)-(LZ4_HASHLOG+1)));
715 else
716 return ((sequence * 2654435761U) >> ((MINMATCH*8)-LZ4_HASHLOG));
717}
718
719LZ4_FORCE_INLINE U32 LZ4_hash5(U64 sequence, tableType_t const tableType)
720{
721 const U32 hashLog = (tableType == byU16) ? LZ4_HASHLOG+1 : LZ4_HASHLOG;
722 if (LZ4_isLittleEndian()) {
723 const U64 prime5bytes = 889523592379ULL;
724 return (U32)(((sequence << 24) * prime5bytes) >> (64 - hashLog));
725 } else {
726 const U64 prime8bytes = 11400714785074694791ULL;
727 return (U32)(((sequence >> 24) * prime8bytes) >> (64 - hashLog));
728 }
729}

780/* LZ4_getIndexOnHash() :
781 * Index of match position registered in hash table.
782 * hash position must be calculated by using base+index, or dictBase+index.
783 * Assumption 1 : only valid if tableType == byU32 or byU16.
784 * Assumption 2 : h is presumed valid (within limits of hash table)
785 */
786LZ4_FORCE_INLINE U32 LZ4_getIndexOnHash(U32 h, const void* tableBase, tableType_t tableType)
787{
788 LZ4_STATIC_ASSERT(LZ4_MEMORY_USAGE > 2);
789 if (tableType == byU32) {
790 const U32* const hashTable = (const U32*) tableBase;
791 assert(h < (1U << (LZ4_MEMORY_USAGE-2)));
792 return hashTable[h];
793 }
794 if (tableType == byU16) {
795 const U16* const hashTable = (const U16*) tableBase;
796 assert(h < (1U << (LZ4_MEMORY_USAGE-1)));
797 return hashTable[h];
798 }
799 assert(0); return 0; /* forbidden case */
800}

807}

818LZ4_FORCE_INLINE void
819LZ4_prepareTable(LZ4_stream_t_internal* const cctx,
820 const int inputSize,
821 const tableType_t tableType) {
822 /* If the table hasn't been used, it's guaranteed to be zeroed out, and is
823 * therefore safe to use no matter what mode we're in. Otherwise, we figure
824 * out if it's safe to leave as is or whether it needs to be reset.
825 */
826 if ((tableType_t)cctx->tableType != clearedTable) {
827 assert(inputSize >= 0);
828 if ((tableType_t)cctx->tableType != tableType
829 || ((tableType == byU16) && cctx->currentOffset + (unsigned)inputSize >= 0xFFFFU)
830 || ((tableType == byU32) && cctx->currentOffset > 1 GB)
831 || tableType == byPtr
832 || inputSize >= 4 KB)
833 {
834 DEBUGLOG(4, "LZ4_prepareTable: Resetting table in %p", cctx);
835 MEM_INIT(cctx->hashTable, 0, LZ4_HASHTABLESIZE);
836 cctx->currentOffset = 0;
837 cctx->tableType = (U32)clearedTable;
838 } else {
839 DEBUGLOG(4, "LZ4_prepareTable: Re-use hash table (no reset)");
840 }
841 }
842
843 /* Adding a gap, so all previous entries are > LZ4_DISTANCE_MAX back,
844 * is faster than compressing without a gap.
845 * However, compressing with currentOffset == 0 is faster still,
846 * so we preserve that case.
847 */
848 if (cctx->currentOffset != 0 && tableType == byU32) {
849 DEBUGLOG(5, "LZ4_prepareTable: adding 64KB to currentOffset");
850 cctx->currentOffset += 64 KB;
851 }
852
853 /* Finally, clear history */
854 cctx->dictCtx = NULL;
855 cctx->dictionary = NULL;
856 cctx->dictSize = 0;
857}

865LZ4_FORCE_INLINE int LZ4_compress_generic_validated(
866 LZ4_stream_t_internal* const cctx,
867 const char* const source,
868 char* const dest,
869 const int inputSize,
870 int* inputConsumed, /* only written when outputDirective == fillOutput */
871 const int maxOutputSize,
872 const limitedOutput_directive outputDirective,
873 const tableType_t tableType,
874 const dict_directive dictDirective,
875 const dictIssue_directive dictIssue,
876 const int acceleration)
877{
878 int result;
879 const BYTE* ip = (const BYTE*) source;
880
881 U32 const startIndex = cctx->currentOffset;
882 const BYTE* base = (const BYTE*) source - startIndex;
883 const BYTE* lowLimit;
884
885 const LZ4_stream_t_internal* dictCtx = (const LZ4_stream_t_internal*) cctx->dictCtx;
886 const BYTE* const dictionary =
887 dictDirective == usingDictCtx ? dictCtx->dictionary : cctx->dictionary;
888 const U32 dictSize =
889 dictDirective == usingDictCtx ? dictCtx->dictSize : cctx->dictSize;
890 const U32 dictDelta = (dictDirective == usingDictCtx) ? startIndex - dictCtx->currentOffset : 0; /* make indexes in dictCtx comparable with index in current context */
891
892 int const maybe_extMem = (dictDirective == usingExtDict) || (dictDirective == usingDictCtx);
893 U32 const prefixIdxLimit = startIndex - dictSize; /* used when dictDirective == dictSmall */
894 const BYTE* const dictEnd = dictionary ? dictionary + dictSize : dictionary;
895 const BYTE* anchor = (const BYTE*) source;
896 const BYTE* const iend = ip + inputSize;
897 const BYTE* const mflimitPlusOne = iend - MFLIMIT + 1;
898 const BYTE* const matchlimit = iend - LASTLITERALS;
899
900 /* the dictCtx currentOffset is indexed on the start of the dictionary,
901 * while a dictionary in the current context precedes the currentOffset */
902 const BYTE* dictBase = (dictionary == NULL) ? NULL :
903 (dictDirective == usingDictCtx) ?
904 dictionary + dictSize - dictCtx->currentOffset :
905 dictionary + dictSize - startIndex;
906
907 BYTE* op = (BYTE*) dest;
908 BYTE* const olimit = op + maxOutputSize;
909
910 U32 offset = 0;
911 U32 forwardH;
912
913 DEBUGLOG(5, "LZ4_compress_generic_validated: srcSize=%i, tableType=%u", inputSize, tableType);
914 assert(ip != NULL);
915 /* If init conditions are not met, we don't have to mark stream
916 * as having dirty context, since no action was taken yet */
917 if (outputDirective == fillOutput && maxOutputSize < 1) { return 0; } /* Impossible to store anything */
918 if ((tableType == byU16) && (inputSize>=LZ4_64Klimit)) { return 0; } /* Size too large (not within 64K limit) */
919 if (tableType==byPtr) assert(dictDirective==noDict); /* only supported use case with byPtr */
920 assert(acceleration >= 1);
921
922 lowLimit = (const BYTE*)source - (dictDirective == withPrefix64k ? dictSize : 0);
923
924 /* Update context state */
925 if (dictDirective == usingDictCtx) {
926 /* Subsequent linked blocks can't use the dictionary. */
927 /* Instead, they use the block we just compressed. */
928 cctx->dictCtx = NULL;
929 cctx->dictSize = (U32)inputSize;
930 } else {
931 cctx->dictSize += (U32)inputSize;
932 }
933 cctx->currentOffset += (U32)inputSize;
934 cctx->tableType = (U32)tableType;
935
936 if (inputSize<LZ4_minLength) goto _last_literals; /* Input too small, no compression (all literals) */
937
938 /* First Byte */
939 LZ4_putPosition(ip, cctx->hashTable, tableType, base);
940 ip++; forwardH = LZ4_hashPosition(ip, tableType);
941
942 /* Main Loop */
943 for ( ; ; ) {
944 const BYTE* match;
945 BYTE* token;
946 const BYTE* filledIp;
947
948 /* Find a match */
949 if (tableType == byPtr) {
950 const BYTE* forwardIp = ip;
951 int step = 1;
952 int searchMatchNb = acceleration << LZ4_skipTrigger;
953 do {
954 U32 const h = forwardH;
955 ip = forwardIp;
956 forwardIp += step;
957 step = (searchMatchNb++ >> LZ4_skipTrigger);
958
959 if (unlikely(forwardIp > mflimitPlusOne)) goto _last_literals;
960 assert(ip < mflimitPlusOne);
961
962 match = LZ4_getPositionOnHash(h, cctx->hashTable, tableType, base);
963 forwardH = LZ4_hashPosition(forwardIp, tableType);
964 LZ4_putPositionOnHash(ip, h, cctx->hashTable, tableType, base);
965
966 } while ( (match+LZ4_DISTANCE_MAX < ip)
967 || (LZ4_read32(match) != LZ4_read32(ip)) );
968
969 } else { /* byU32, byU16 */
970
971 const BYTE* forwardIp = ip;
972 int step = 1;
973 int searchMatchNb = acceleration << LZ4_skipTrigger;
974 do {
975 U32 const h = forwardH;
976 U32 const current = (U32)(forwardIp - base);
977 U32 matchIndex = LZ4_getIndexOnHash(h, cctx->hashTable, tableType);
978 assert(matchIndex <= current);
979 assert(forwardIp - base < (ptrdiff_t)(2 GB - 1));
980 ip = forwardIp;
981 forwardIp += step;
982 step = (searchMatchNb++ >> LZ4_skipTrigger);
983
984 if (unlikely(forwardIp > mflimitPlusOne)) goto _last_literals;
985 assert(ip < mflimitPlusOne);
986
987 if (dictDirective == usingDictCtx) {
988 if (matchIndex < startIndex) {
989 /* there was no match, try the dictionary */
990 assert(tableType == byU32);
991 matchIndex = LZ4_getIndexOnHash(h, dictCtx->hashTable, byU32);
992 match = dictBase + matchIndex;
993 matchIndex += dictDelta; /* make dictCtx index comparable with current context */
994 lowLimit = dictionary;
995 } else {
996 match = base + matchIndex;
997 lowLimit = (const BYTE*)source;
998 }
999 } else if (dictDirective == usingExtDict) {
1000 if (matchIndex < startIndex) {
1001 DEBUGLOG(7, "extDict candidate: matchIndex=%5u < startIndex=%5u", matchIndex, startIndex);
1002 assert(startIndex - matchIndex >= MINMATCH);
1003 assert(dictBase);
1004 match = dictBase + matchIndex;
1005 lowLimit = dictionary;
1006 } else {
1007 match = base + matchIndex;
1008 lowLimit = (const BYTE*)source;
1009 }
1010 } else { /* single continuous memory segment */
1011 match = base + matchIndex;
1012 }
1013 forwardH = LZ4_hashPosition(forwardIp, tableType);
1014 LZ4_putIndexOnHash(current, h, cctx->hashTable, tableType);
1015
1016 DEBUGLOG(7, "candidate at pos=%u (offset=%u \n", matchIndex, current - matchIndex);
1017 if ((dictIssue == dictSmall) && (matchIndex < prefixIdxLimit)) { continue; } /* match outside of valid area */
1018 assert(matchIndex < current);
1019 if ( ((tableType != byU16) || (LZ4_DISTANCE_MAX < LZ4_DISTANCE_ABSOLUTE_MAX))
1020 && (matchIndex+LZ4_DISTANCE_MAX < current)) {
1021 continue;
1022 } /* too far */
1023 assert((current - matchIndex) <= LZ4_DISTANCE_MAX); /* match now expected within distance */
1024
1025 if (LZ4_read32(match) == LZ4_read32(ip)) {
1026 if (maybe_extMem) offset = current - matchIndex;
1027 break; /* match found */
1028 }
1029
1030 } while(1);
1031 }
1032
1033 /* Catch up */
1034 filledIp = ip;
1035 while (((ip>anchor) & (match > lowLimit)) && (unlikely(ip[-1]==match[-1]))) { ip--; match--; }
1036
1037 /* Encode Literals */
1038 { unsigned const litLength = (unsigned)(ip - anchor);
1039 token = op++;
1040 if ((outputDirective == limitedOutput) && /* Check output buffer overflow */
1041 (unlikely(op + litLength + (2 + 1 + LASTLITERALS) + (litLength/255) > olimit)) ) {
1042 return 0; /* cannot compress within `dst` budget. Stored indexes in hash table are nonetheless fine */
1043 }
1044 if ((outputDirective == fillOutput) &&
1045 (unlikely(op + (litLength+240)/255 /* litlen */ + litLength /* literals */ + 2 /* offset */ + 1 /* token */ + MFLIMIT - MINMATCH /* min last literals so last match is <= end - MFLIMIT */ > olimit))) {
1046 op--;
1047 goto _last_literals;
1048 }
1049 if (litLength >= RUN_MASK) {
1050 int len = (int)(litLength - RUN_MASK);
1051 *token = (RUN_MASK<<ML_BITS);
1052 for(; len >= 255 ; len-=255) *op++ = 255;
1053 *op++ = (BYTE)len;
1054 }
1055 else *token = (BYTE)(litLength<<ML_BITS);
1056
1057 /* Copy Literals */
1058 LZ4_wildCopy8(op, anchor, op+litLength);
1059 op+=litLength;
1060 DEBUGLOG(6, "seq.start:%i, literals=%u, match.start:%i",
1061 (int)(anchor-(const BYTE*)source), litLength, (int)(ip-(const BYTE*)source));
1062 }
1063
1064_next_match:
1065 /* at this stage, the following variables must be correctly set :
1066 * - ip : at start of LZ operation
1067 * - match : at start of previous pattern occurrence; can be within current prefix, or within extDict
1068 * - offset : if maybe_ext_memSegment==1 (constant)
1069 * - lowLimit : must be == dictionary to mean "match is within extDict"; must be == source otherwise
1070 * - token and *token : position to write 4-bits for match length; higher 4-bits for literal length supposed already written
1071 */
1072
1073 if ((outputDirective == fillOutput) &&
1074 (op + 2 /* offset */ + 1 /* token */ + MFLIMIT - MINMATCH /* min last literals so last match is <= end - MFLIMIT */ > olimit)) {
1075 /* the match was too close to the end, rewind and go to last literals */
1076 op = token;
1077 goto _last_literals;
1078 }
1079
1080 /* Encode Offset */
1081 if (maybe_extMem) { /* static test */
1082 DEBUGLOG(6, " with offset=%u (ext if > %i)", offset, (int)(ip - (const BYTE*)source));
1083 assert(offset <= LZ4_DISTANCE_MAX && offset > 0);
1084 LZ4_writeLE16(op, (U16)offset); op+=2;
1085 } else {
1086 DEBUGLOG(6, " with offset=%u (same segment)", (U32)(ip - match));
1087 assert(ip-match <= LZ4_DISTANCE_MAX);
1088 LZ4_writeLE16(op, (U16)(ip - match)); op+=2;
1089 }
1090
1091 /* Encode MatchLength */
1092 { unsigned matchCode;
1093
1094 if ( (dictDirective==usingExtDict || dictDirective==usingDictCtx)
1095 && (lowLimit==dictionary) /* match within extDict */ ) {
1096 const BYTE* limit = ip + (dictEnd-match);
1097 assert(dictEnd > match);
1098 if (limit > matchlimit) limit = matchlimit;
1099 matchCode = LZ4_count(ip+MINMATCH, match+MINMATCH, limit);
1100 ip += (size_t)matchCode + MINMATCH;
1101 if (ip==limit) {
1102 unsigned const more = LZ4_count(limit, (const BYTE*)source, matchlimit);
1103 matchCode += more;
1104 ip += more;
1105 }
1106 DEBUGLOG(6, " with matchLength=%u starting in extDict", matchCode+MINMATCH);
1107 } else {
1108 matchCode = LZ4_count(ip+MINMATCH, match+MINMATCH, matchlimit);
1109 ip += (size_t)matchCode + MINMATCH;
1110 DEBUGLOG(6, " with matchLength=%u", matchCode+MINMATCH);
1111 }
1112
1113 if ((outputDirective) && /* Check output buffer overflow */
1114 (unlikely(op + (1 + LASTLITERALS) + (matchCode+240)/255 > olimit)) ) {
1115 if (outputDirective == fillOutput) {
1116 /* Match description too long : reduce it */
1117 U32 newMatchCode = 15 /* in token */ - 1 /* to avoid needing a zero byte */ + ((U32)(olimit - op) - 1 - LASTLITERALS) * 255;
1118 ip -= matchCode - newMatchCode;
1119 assert(newMatchCode < matchCode);
1120 matchCode = newMatchCode;
1121 if (unlikely(ip <= filledIp)) {
1122 /* We have already filled up to filledIp so if ip ends up less than filledIp
1123 * we have positions in the hash table beyond the current position. This is
1124 * a problem if we reuse the hash table. So we have to remove these positions
1125 * from the hash table.
1126 */
1127 const BYTE* ptr;
1128 DEBUGLOG(5, "Clearing %u positions", (U32)(filledIp - ip));
1129 for (ptr = ip; ptr <= filledIp; ++ptr) {
1130 U32 const h = LZ4_hashPosition(ptr, tableType);
1131 LZ4_clearHash(h, cctx->hashTable, tableType);
1132 }
1133 }
1134 } else {
1135 assert(outputDirective == limitedOutput);
1136 return 0; /* cannot compress within `dst` budget. Stored indexes in hash table are nonetheless fine */
1137 }
1138 }
1139 if (matchCode >= ML_MASK) {
1140 *token += ML_MASK;
1141 matchCode -= ML_MASK;
1142 LZ4_write32(op, 0xFFFFFFFF);
1143 while (matchCode >= 4*255) {
1144 op+=4;
1145 LZ4_write32(op, 0xFFFFFFFF);
1146 matchCode -= 4*255;
1147 }
1148 op += matchCode / 255;
1149 *op++ = (BYTE)(matchCode % 255);
1150 } else
1151 *token += (BYTE)(matchCode);
1152 }
1153 /* Ensure we have enough space for the last literals. */
1154 assert(!(outputDirective == fillOutput && op + 1 + LASTLITERALS > olimit));
1155
1156 anchor = ip;
1157
1158 /* Test end of chunk */
1159 if (ip >= mflimitPlusOne) break;
1160
1161 /* Fill table */
1162 LZ4_putPosition(ip-2, cctx->hashTable, tableType, base);
1163
1164 /* Test next position */
1165 if (tableType == byPtr) {
1166
1167 match = LZ4_getPosition(ip, cctx->hashTable, tableType, base);
1168 LZ4_putPosition(ip, cctx->hashTable, tableType, base);
1169 if ( (match+LZ4_DISTANCE_MAX >= ip)
1170 && (LZ4_read32(match) == LZ4_read32(ip)) )
1171 { token=op++; *token=0; goto _next_match; }
1172
1173 } else { /* byU32, byU16 */
1174
1175 U32 const h = LZ4_hashPosition(ip, tableType);
1176 U32 const current = (U32)(ip-base);
1177 U32 matchIndex = LZ4_getIndexOnHash(h, cctx->hashTable, tableType);
1178 assert(matchIndex < current);
1179 if (dictDirective == usingDictCtx) {
1180 if (matchIndex < startIndex) {
1181 /* there was no match, try the dictionary */
1182 matchIndex = LZ4_getIndexOnHash(h, dictCtx->hashTable, byU32);
1183 match = dictBase + matchIndex;
1184 lowLimit = dictionary; /* required for match length counter */
1185 matchIndex += dictDelta;
1186 } else {
1187 match = base + matchIndex;
1188 lowLimit = (const BYTE*)source; /* required for match length counter */
1189 }
1190 } else if (dictDirective==usingExtDict) {
1191 if (matchIndex < startIndex) {
1192 assert(dictBase);
1193 match = dictBase + matchIndex;
1194 lowLimit = dictionary; /* required for match length counter */
1195 } else {
1196 match = base + matchIndex;
1197 lowLimit = (const BYTE*)source; /* required for match length counter */
1198 }
1199 } else { /* single memory segment */
1200 match = base + matchIndex;
1201 }
1202 LZ4_putIndexOnHash(current, h, cctx->hashTable, tableType);
1203 assert(matchIndex < current);
1204 if ( ((dictIssue==dictSmall) ? (matchIndex >= prefixIdxLimit) : 1)
1205 && (((tableType==byU16) && (LZ4_DISTANCE_MAX == LZ4_DISTANCE_ABSOLUTE_MAX)) ? 1 : (matchIndex+LZ4_DISTANCE_MAX >= current))
1206 && (LZ4_read32(match) == LZ4_read32(ip)) ) {
1207 token=op++;
1208 *token=0;
1209 if (maybe_extMem) offset = current - matchIndex;
1210 DEBUGLOG(6, "seq.start:%i, literals=%u, match.start:%i",
1211 (int)(anchor-(const BYTE*)source), 0, (int)(ip-(const BYTE*)source));
1212 goto _next_match;
1213 }
1214 }
1215
1216 /* Prepare next loop */
1217 forwardH = LZ4_hashPosition(++ip, tableType);
1218
1219 }
1220
1221_last_literals:
1222 /* Encode Last Literals */
1223 { size_t lastRun = (size_t)(iend - anchor);
1224 if ( (outputDirective) && /* Check output buffer overflow */
1225 (op + lastRun + 1 + ((lastRun+255-RUN_MASK)/255) > olimit)) {
1226 if (outputDirective == fillOutput) {
1227 /* adapt lastRun to fill 'dst' */
1228 assert(olimit >= op);
1229 lastRun = (size_t)(olimit-op) - 1/*token*/;
1230 lastRun -= (lastRun + 256 - RUN_MASK) / 256; /*additional length tokens*/
1231 } else {
1232 assert(outputDirective == limitedOutput);
1233 return 0; /* cannot compress within `dst` budget. Stored indexes in hash table are nonetheless fine */
1234 }
1235 }
1236 DEBUGLOG(6, "Final literal run : %i literals", (int)lastRun);
1237 if (lastRun >= RUN_MASK) {
1238 size_t accumulator = lastRun - RUN_MASK;
1239 *op++ = RUN_MASK << ML_BITS;
1240 for(; accumulator >= 255 ; accumulator-=255) *op++ = 255;
1241 *op++ = (BYTE) accumulator;
1242 } else {
1243 *op++ = (BYTE)(lastRun<<ML_BITS);
1244 }
1245 LZ4_memcpy(op, anchor, lastRun);
1246 ip = anchor + lastRun;
1247 op += lastRun;
1248 }
1249
1250 if (outputDirective == fillOutput) {
1251 *inputConsumed = (int) (((const char*)ip)-source);
1252 }
1253 result = (int)(((char*)op) - dest);
1254 assert(result > 0);
1255 DEBUGLOG(5, "LZ4_compress_generic: compressed %i bytes into %i bytes", inputSize, result);
1256 return result;
1257}

1491#define HASH_UNIT sizeof(reg_t)
1492int LZ4_loadDict (LZ4_stream_t* LZ4_dict, const char* dictionary, int dictSize)
1493{
1494 LZ4_stream_t_internal* dict = &LZ4_dict->internal_donotuse;
1495 const tableType_t tableType = byU32;
1496 const BYTE* p = (const BYTE*)dictionary;
1497 const BYTE* const dictEnd = p + dictSize;
1498 const BYTE* base;
1499
1500 DEBUGLOG(4, "LZ4_loadDict (%i bytes from %p into %p)", dictSize, dictionary, LZ4_dict);
1501
1502 /* It's necessary to reset the context,
1503 * and not just continue it with prepareTable()
1504 * to avoid any risk of generating overflowing matchIndex
1505 * when compressing using this dictionary */
1506 LZ4_resetStream(LZ4_dict);
1507
1508 /* We always increment the offset by 64 KB, since, if the dict is longer,
1509 * we truncate it to the last 64k, and if it's shorter, we still want to
1510 * advance by a whole window length so we can provide the guarantee that
1511 * there are only valid offsets in the window, which allows an optimization
1512 * in LZ4_compress_fast_continue() where it uses noDictIssue even when the
1513 * dictionary isn't a full 64k. */
1514 dict->currentOffset += 64 KB;
1515
1516 if (dictSize < (int)HASH_UNIT) {
1517 return 0;
1518 }
1519
1520 if ((dictEnd - p) > 64 KB) p = dictEnd - 64 KB;
1521 base = dictEnd - dict->currentOffset;
1522 dict->dictionary = p;
1523 dict->dictSize = (U32)(dictEnd - p);
1524 dict->tableType = (U32)tableType;
1525
1526 while (p <= dictEnd-HASH_UNIT) {
1527 LZ4_putPosition(p, dict->hashTable, tableType, base);
1528 p+=3;
1529 }
1530
1531 return (int)dict->dictSize;
1532}

1563static void LZ4_renormDictT(LZ4_stream_t_internal* LZ4_dict, int nextSize)
1564{
1565 assert(nextSize >= 0);
1566 if (LZ4_dict->currentOffset + (unsigned)nextSize > 0x80000000) { /* potential ptrdiff_t overflow (32-bits mode) */
1567 /* rescale hash table */
1568 U32 const delta = LZ4_dict->currentOffset - 64 KB;
1569 const BYTE* dictEnd = LZ4_dict->dictionary + LZ4_dict->dictSize;
1570 int i;
1571 DEBUGLOG(4, "LZ4_renormDictT");
1572 for (i=0; i<LZ4_HASH_SIZE_U32; i++) {
1573 if (LZ4_dict->hashTable[i] < delta) LZ4_dict->hashTable[i]=0;
1574 else LZ4_dict->hashTable[i] -= delta;
1575 }
1576 LZ4_dict->currentOffset = 64 KB;
1577 if (LZ4_dict->dictSize > 64 KB) LZ4_dict->dictSize = 64 KB;
1578 LZ4_dict->dictionary = dictEnd - LZ4_dict->dictSize;
1579 }

1583int LZ4_compress_fast_continue (LZ4_stream_t* LZ4_stream,
1584 const char* source, char* dest,
1585 int inputSize, int maxOutputSize,
1586 int acceleration)
1587{
1588 const tableType_t tableType = byU32;
1589 LZ4_stream_t_internal* const streamPtr = &LZ4_stream->internal_donotuse;
1590 const char* dictEnd = streamPtr->dictSize ? (const char*)streamPtr->dictionary + streamPtr->dictSize : NULL;
1591
1592 DEBUGLOG(5, "LZ4_compress_fast_continue (inputSize=%i, dictSize=%u)", inputSize, streamPtr->dictSize);
1593
1594 LZ4_renormDictT(streamPtr, inputSize); /* fix index overflow */
1595 if (acceleration < 1) acceleration = LZ4_ACCELERATION_DEFAULT;
1596 if (acceleration > LZ4_ACCELERATION_MAX) acceleration = LZ4_ACCELERATION_MAX;
1597
1598 /* invalidate tiny dictionaries */
1599 if ( (streamPtr->dictSize < 4) /* tiny dictionary : not enough for a hash */
1600 && (dictEnd != source) /* prefix mode */
1601 && (inputSize > 0) /* tolerance : don't lose history, in case next invocation would use prefix mode */
1602 && (streamPtr->dictCtx == NULL) /* usingDictCtx */
1603 ) {
1604 DEBUGLOG(5, "LZ4_compress_fast_continue: dictSize(%u) at addr:%p is too small", streamPtr->dictSize, streamPtr->dictionary);
1605 /* remove dictionary existence from history, to employ faster prefix mode */
1606 streamPtr->dictSize = 0;
1607 streamPtr->dictionary = (const BYTE*)source;
1608 dictEnd = source;
1609 }
1610
1611 /* Check overlapping input/dictionary space */
1612 { const char* const sourceEnd = source + inputSize;
1613 if ((sourceEnd > (const char*)streamPtr->dictionary) && (sourceEnd < dictEnd)) {
1614 streamPtr->dictSize = (U32)(dictEnd - sourceEnd);
1615 if (streamPtr->dictSize > 64 KB) streamPtr->dictSize = 64 KB;
1616 if (streamPtr->dictSize < 4) streamPtr->dictSize = 0;
1617 streamPtr->dictionary = (const BYTE*)dictEnd - streamPtr->dictSize;
1618 }
1619 }
1620
1621 /* prefix mode : source data follows dictionary */
1622 if (dictEnd == source) {
1623 if ((streamPtr->dictSize < 64 KB) && (streamPtr->dictSize < streamPtr->currentOffset))
1624 return LZ4_compress_generic(streamPtr, source, dest, inputSize, NULL, maxOutputSize, limitedOutput, tableType, withPrefix64k, dictSmall, acceleration);
1625 else
1626 return LZ4_compress_generic(streamPtr, source, dest, inputSize, NULL, maxOutputSize, limitedOutput, tableType, withPrefix64k, noDictIssue, acceleration);
1627 }
1628
1629 /* external dictionary mode */
1630 { int result;
1631 if (streamPtr->dictCtx) {
1632 /* We depend here on the fact that dictCtx'es (produced by
1633 * LZ4_loadDict) guarantee that their tables contain no references
1634 * to offsets between dictCtx->currentOffset - 64 KB and
1635 * dictCtx->currentOffset - dictCtx->dictSize. This makes it safe
1636 * to use noDictIssue even when the dict isn't a full 64 KB.
1637 */
1638 if (inputSize > 4 KB) {
1639 /* For compressing large blobs, it is faster to pay the setup
1640 * cost to copy the dictionary's tables into the active context,
1641 * so that the compression loop is only looking into one table.
1642 */
1643 LZ4_memcpy(streamPtr, streamPtr->dictCtx, sizeof(*streamPtr));
1644 result = LZ4_compress_generic(streamPtr, source, dest, inputSize, NULL, maxOutputSize, limitedOutput, tableType, usingExtDict, noDictIssue, acceleration);
1645 } else {
1646 result = LZ4_compress_generic(streamPtr, source, dest, inputSize, NULL, maxOutputSize, limitedOutput, tableType, usingDictCtx, noDictIssue, acceleration);
1647 }
1648 } else { /* small data <= 4 KB */
1649 if ((streamPtr->dictSize < 64 KB) && (streamPtr->dictSize < streamPtr->currentOffset)) {
1650 result = LZ4_compress_generic(streamPtr, source, dest, inputSize, NULL, maxOutputSize, limitedOutput, tableType, usingExtDict, dictSmall, acceleration);
1651 } else {
1652 result = LZ4_compress_generic(streamPtr, source, dest, inputSize, NULL, maxOutputSize, limitedOutput, tableType, usingExtDict, noDictIssue, acceleration);
1653 }
1654 }
1655 streamPtr->dictionary = (const BYTE*)source;
1656 streamPtr->dictSize = (U32)inputSize;
1657 return result;
1658 }
1659}

1727 * lencheck - end ip. Return an error if ip advances >= lencheck.
1728 * loop_check - check ip >= lencheck in body of loop. Returns loop_error if so.
1729 * initial_check - check ip >= lencheck before start of loop. Returns initial_error if so.
1730 * error (output) - error code. Should be set to 0 before call.
1731 */
1732typedef enum { loop_error = -2, initial_error = -1, ok = 0 } variable_length_error;
1733LZ4_FORCE_INLINE unsigned
1734read_variable_length(const BYTE**ip, const BYTE* lencheck,
1735 int loop_check, int initial_check,
1736 variable_length_error* error)
1737{
1738 U32 length = 0;
1739 U32 s;
1740 if (initial_check && unlikely((*ip) >= lencheck)) { /* overflow detection */
1741 *error = initial_error;
1742 return length;
1743 }
1744 do {
1745 s = **ip;
1746 (*ip)++;
1747 length += s;
1748 if (loop_check && unlikely((*ip) >= lencheck)) { /* overflow detection */
1749 *error = loop_error;
1750 return length;
1751 }
1752 } while (s==255);
1753
1754 return length;
1755}

1763LZ4_FORCE_INLINE int
1764LZ4_decompress_generic(
1765 const char* const src,
1766 char* const dst,
1767 int srcSize,
1768 int outputSize, /* If endOnInput==endOnInputSize, this value is `dstCapacity` */
1769
1770 endCondition_directive endOnInput, /* endOnOutputSize, endOnInputSize */
1771 earlyEnd_directive partialDecoding, /* full, partial */
1772 dict_directive dict, /* noDict, withPrefix64k, usingExtDict */
1773 const BYTE* const lowPrefix, /* always <= dst, == dst when no prefix */
1774 const BYTE* const dictStart, /* only if dict==usingExtDict */
1775 const size_t dictSize /* note : = 0 if noDict */
1776 )
1777{
1778 if ((src == NULL) || (outputSize < 0)) { return -1; }
1779
1780 { const BYTE* ip = (const BYTE*) src;
1781 const BYTE* const iend = ip + srcSize;
1782
1783 BYTE* op = (BYTE*) dst;
1784 BYTE* const oend = op + outputSize;
1785 BYTE* cpy;
1786
1787 const BYTE* const dictEnd = (dictStart == NULL) ? NULL : dictStart + dictSize;
1788
1789 const int safeDecode = (endOnInput==endOnInputSize);
1790 const int checkOffset = ((safeDecode) && (dictSize < (int)(64 KB)));
1791
1792
1793 /* Set up the "end" pointers for the shortcut. */
1794 const BYTE* const shortiend = iend - (endOnInput ? 14 : 8) /*maxLL*/ - 2 /*offset*/;
1795 const BYTE* const shortoend = oend - (endOnInput ? 14 : 8) /*maxLL*/ - 18 /*maxML*/;
1796
1797 const BYTE* match;
1798 size_t offset;
1799 unsigned token;
1800 size_t length;
1801
1802
1803 DEBUGLOG(5, "LZ4_decompress_generic (srcSize:%i, dstSize:%i)", srcSize, outputSize);
1804
1805 /* Special cases */
1806 assert(lowPrefix <= op);
1807 if ((endOnInput) && (unlikely(outputSize==0))) {
1808 /* Empty output buffer */
1809 if (partialDecoding) return 0;
1810 return ((srcSize==1) && (*ip==0)) ? 0 : -1;
1811 }
1812 if ((!endOnInput) && (unlikely(outputSize==0))) { return (*ip==0 ? 1 : -1); }
1813 if ((endOnInput) && unlikely(srcSize==0)) { return -1; }
1814
1815 /* Currently the fast loop shows a regression on qualcomm arm chips. */
1816#if LZ4_FAST_DEC_LOOP
1817 if ((oend - op) < FASTLOOP_SAFE_DISTANCE) {
1818 DEBUGLOG(6, "skip fast decode loop");
1819 goto safe_decode;
1820 }
1821
1822 /* Fast loop : decode sequences as long as output < iend-FASTLOOP_SAFE_DISTANCE */
1823 while (1) {
1824 /* Main fastloop assertion: We can always wildcopy FASTLOOP_SAFE_DISTANCE */
1825 assert(oend - op >= FASTLOOP_SAFE_DISTANCE);
1826 if (endOnInput) { assert(ip < iend); }
1827 token = *ip++;
1828 length = token >> ML_BITS; /* literal length */
1829
1830 assert(!endOnInput || ip <= iend); /* ip < iend before the increment */
1831
1832 /* decode literal length */
1833 if (length == RUN_MASK) {
1834 variable_length_error error = ok;
1835 length += read_variable_length(&ip, iend-RUN_MASK, (int)endOnInput, (int)endOnInput, &error);
1836 if (error == initial_error) { goto _output_error; }
1837 if ((safeDecode) && unlikely((uptrval)(op)+length<(uptrval)(op))) { goto _output_error; } /* overflow detection */
1838 if ((safeDecode) && unlikely((uptrval)(ip)+length<(uptrval)(ip))) { goto _output_error; } /* overflow detection */
1839
1840 /* copy literals */
1841 cpy = op+length;
1842 LZ4_STATIC_ASSERT(MFLIMIT >= WILDCOPYLENGTH);
1843 if (endOnInput) { /* LZ4_decompress_safe() */
1844 if ((cpy>oend-32) || (ip+length>iend-32)) { goto safe_literal_copy; }
1845 LZ4_wildCopy32(op, ip, cpy);
1846 } else { /* LZ4_decompress_fast() */
1847 if (cpy>oend-8) { goto safe_literal_copy; }
1848 LZ4_wildCopy8(op, ip, cpy); /* LZ4_decompress_fast() cannot copy more than 8 bytes at a time :
1849 * it doesn't know input length, and only relies on end-of-block properties */
1850 }
1851 ip += length; op = cpy;
1852 } else {
1853 cpy = op+length;
1854 if (endOnInput) { /* LZ4_decompress_safe() */
1855 DEBUGLOG(7, "copy %u bytes in a 16-bytes stripe", (unsigned)length);
1856 /* We don't need to check oend, since we check it once for each loop below */
1857 if (ip > iend-(16 + 1/*max lit + offset + nextToken*/)) { goto safe_literal_copy; }
1858 /* Literals can only be 14, but hope compilers optimize if we copy by a register size */
1859 LZ4_memcpy(op, ip, 16);
1860 } else { /* LZ4_decompress_fast() */
1861 /* LZ4_decompress_fast() cannot copy more than 8 bytes at a time :
1862 * it doesn't know input length, and relies on end-of-block properties */
1863 LZ4_memcpy(op, ip, 8);
1864 if (length > 8) { LZ4_memcpy(op+8, ip+8, 8); }
1865 }
1866 ip += length; op = cpy;
1867 }
1868
1869 /* get offset */
1870 offset = LZ4_readLE16(ip); ip+=2;
1871 match = op - offset;
1872 assert(match <= op);
1873
1874 /* get matchlength */
1875 length = token & ML_MASK;
1876
1877 if (length == ML_MASK) {
1878 variable_length_error error = ok;
1879 if ((checkOffset) && (unlikely(match + dictSize < lowPrefix))) { goto _output_error; } /* Error : offset outside buffers */
1880 length += read_variable_length(&ip, iend - LASTLITERALS + 1, (int)endOnInput, 0, &error);
1881 if (error != ok) { goto _output_error; }
1882 if ((safeDecode) && unlikely((uptrval)(op)+length<(uptrval)op)) { goto _output_error; } /* overflow detection */
1883 length += MINMATCH;
1884 if (op + length >= oend - FASTLOOP_SAFE_DISTANCE) {
1885 goto safe_match_copy;
1886 }
1887 } else {
1888 length += MINMATCH;
1889 if (op + length >= oend - FASTLOOP_SAFE_DISTANCE) {
1890 goto safe_match_copy;
1891 }
1892
1893 /* Fastpath check: Avoids a branch in LZ4_wildCopy32 if true */
1894 if ((dict == withPrefix64k) || (match >= lowPrefix)) {
1895 if (offset >= 8) {
1896 assert(match >= lowPrefix);
1897 assert(match <= op);
1898 assert(op + 18 <= oend);
1899
1900 LZ4_memcpy(op, match, 8);
1901 LZ4_memcpy(op+8, match+8, 8);
1902 LZ4_memcpy(op+16, match+16, 2);
1903 op += length;
1904 continue;
1905 } } }
1906
1907 if (checkOffset && (unlikely(match + dictSize < lowPrefix))) { goto _output_error; } /* Error : offset outside buffers */
1908 /* match starting within external dictionary */
1909 if ((dict==usingExtDict) && (match < lowPrefix)) {
1910 if (unlikely(op+length > oend-LASTLITERALS)) {
1911 if (partialDecoding) {
1912 DEBUGLOG(7, "partialDecoding: dictionary match, close to dstEnd");
1913 length = MIN(length, (size_t)(oend-op));
1914 } else {
1915 goto _output_error; /* end-of-block condition violated */
1916 } }
1917
1918 if (length <= (size_t)(lowPrefix-match)) {
1919 /* match fits entirely within external dictionary : just copy */
1920 memmove(op, dictEnd - (lowPrefix-match), length);
1921 op += length;
1922 } else {
1923 /* match stretches into both external dictionary and current block */
1924 size_t const copySize = (size_t)(lowPrefix - match);
1925 size_t const restSize = length - copySize;
1926 LZ4_memcpy(op, dictEnd - copySize, copySize);
1927 op += copySize;
1928 if (restSize > (size_t)(op - lowPrefix)) { /* overlap copy */
1929 BYTE* const endOfMatch = op + restSize;
1930 const BYTE* copyFrom = lowPrefix;
1931 while (op < endOfMatch) { *op++ = *copyFrom++; }
1932 } else {
1933 LZ4_memcpy(op, lowPrefix, restSize);
1934 op += restSize;
1935 } }
1936 continue;
1937 }
1938
1939 /* copy match within block */
1940 cpy = op + length;
1941
1942 assert((op <= oend) && (oend-op >= 32));
1943 if (unlikely(offset<16)) {
1944 LZ4_memcpy_using_offset(op, match, cpy, offset);
1945 } else {
1946 LZ4_wildCopy32(op, match, cpy);
1947 }
1948
1949 op = cpy; /* wildcopy correction */
1950 }
1951 safe_decode:
1952#endif
1953
1954 /* Main Loop : decode remaining sequences where output < FASTLOOP_SAFE_DISTANCE */
1955 while (1) {
1956 token = *ip++;
1957 length = token >> ML_BITS; /* literal length */
1958
1959 assert(!endOnInput || ip <= iend); /* ip < iend before the increment */
1960
1961 /* A two-stage shortcut for the most common case:
1962 * 1) If the literal length is 0..14, and there is enough space,
1963 * enter the shortcut and copy 16 bytes on behalf of the literals
1964 * (in the fast mode, only 8 bytes can be safely copied this way).
1965 * 2) Further if the match length is 4..18, copy 18 bytes in a similar
1966 * manner; but we ensure that there's enough space in the output for
1967 * those 18 bytes earlier, upon entering the shortcut (in other words,
1968 * there is a combined check for both stages).
1969 */
1970 if ( (endOnInput ? length != RUN_MASK : length <= 8)
1971 /* strictly "less than" on input, to re-enter the loop with at least one byte */
1972 && likely((endOnInput ? ip < shortiend : 1) & (op <= shortoend)) ) {
1973 /* Copy the literals */
1974 LZ4_memcpy(op, ip, endOnInput ? 16 : 8);
1975 op += length; ip += length;
1976
1977 /* The second stage: prepare for match copying, decode full info.
1978 * If it doesn't work out, the info won't be wasted. */
1979 length = token & ML_MASK; /* match length */
1980 offset = LZ4_readLE16(ip); ip += 2;
1981 match = op - offset;
1982 assert(match <= op); /* check overflow */
1983
1984 /* Do not deal with overlapping matches. */
1985 if ( (length != ML_MASK)
1986 && (offset >= 8)
1987 && (dict==withPrefix64k || match >= lowPrefix) ) {
1988 /* Copy the match. */
1989 LZ4_memcpy(op + 0, match + 0, 8);
1990 LZ4_memcpy(op + 8, match + 8, 8);
1991 LZ4_memcpy(op +16, match +16, 2);
1992 op += length + MINMATCH;
1993 /* Both stages worked, load the next token. */
1994 continue;
1995 }
1996
1997 /* The second stage didn't work out, but the info is ready.
1998 * Propel it right to the point of match copying. */
1999 goto _copy_match;
2000 }
2001
2002 /* decode literal length */
2003 if (length == RUN_MASK) {
2004 variable_length_error error = ok;
2005 length += read_variable_length(&ip, iend-RUN_MASK, (int)endOnInput, (int)endOnInput, &error);
2006 if (error == initial_error) { goto _output_error; }
2007 if ((safeDecode) && unlikely((uptrval)(op)+length<(uptrval)(op))) { goto _output_error; } /* overflow detection */
2008 if ((safeDecode) && unlikely((uptrval)(ip)+length<(uptrval)(ip))) { goto _output_error; } /* overflow detection */
2009 }
2010
2011 /* copy literals */
2012 cpy = op+length;
2013#if LZ4_FAST_DEC_LOOP
2014 safe_literal_copy:
2015#endif
2016 LZ4_STATIC_ASSERT(MFLIMIT >= WILDCOPYLENGTH);
2017 if ( ((endOnInput) && ((cpy>oend-MFLIMIT) || (ip+length>iend-(2+1+LASTLITERALS))) )
2018 || ((!endOnInput) && (cpy>oend-WILDCOPYLENGTH)) )
2019 {
2020 /* We've either hit the input parsing restriction or the output parsing restriction.
2021 * In the normal scenario, decoding a full block, it must be the last sequence,
2022 * otherwise it's an error (invalid input or dimensions).
2023 * In partialDecoding scenario, it's necessary to ensure there is no buffer overflow.
2024 */
2025 if (partialDecoding) {
2026 /* Since we are partial decoding we may be in this block because of the output parsing
2027 * restriction, which is not valid since the output buffer is allowed to be undersized.
2028 */
2029 assert(endOnInput);
2030 DEBUGLOG(7, "partialDecoding: copying literals, close to input or output end")
2031 DEBUGLOG(7, "partialDecoding: literal length = %u", (unsigned)length);
2032 DEBUGLOG(7, "partialDecoding: remaining space in dstBuffer : %i", (int)(oend - op));
2033 DEBUGLOG(7, "partialDecoding: remaining space in srcBuffer : %i", (int)(iend - ip));
2034 /* Finishing in the middle of a literals segment,
2035 * due to lack of input.
2036 */
2037 if (ip+length > iend) {
2038 length = (size_t)(iend-ip);
2039 cpy = op + length;
2040 }
2041 /* Finishing in the middle of a literals segment,
2042 * due to lack of output space.
2043 */
2044 if (cpy > oend) {
2045 cpy = oend;
2046 assert(op<=oend);
2047 length = (size_t)(oend-op);
2048 }
2049 } else {
2050 /* We must be on the last sequence because of the parsing limitations so check
2051 * that we exactly regenerate the original size (must be exact when !endOnInput).
2052 */
2053 if ((!endOnInput) && (cpy != oend)) { goto _output_error; }
2054 /* We must be on the last sequence (or invalid) because of the parsing limitations
2055 * so check that we exactly consume the input and don't overrun the output buffer.
2056 */
2057 if ((endOnInput) && ((ip+length != iend) || (cpy > oend))) {
2058 DEBUGLOG(6, "should have been last run of literals")
2059 DEBUGLOG(6, "ip(%p) + length(%i) = %p != iend (%p)", ip, (int)length, ip+length, iend);
2060 DEBUGLOG(6, "or cpy(%p) > oend(%p)", cpy, oend);
2061 goto _output_error;
2062 }
2063 }
2064 memmove(op, ip, length); /* supports overlapping memory regions; only matters for in-place decompression scenarios */
2065 ip += length;
2066 op += length;
2067 /* Necessarily EOF when !partialDecoding.
2068 * When partialDecoding, it is EOF if we've either
2069 * filled the output buffer or
2070 * can't proceed with reading an offset for following match.
2071 */
2072 if (!partialDecoding || (cpy == oend) || (ip >= (iend-2))) {
2073 break;
2074 }
2075 } else {
2076 LZ4_wildCopy8(op, ip, cpy); /* may overwrite up to WILDCOPYLENGTH beyond cpy */
2077 ip += length; op = cpy;
2078 }
2079
2080 /* get offset */
2081 offset = LZ4_readLE16(ip); ip+=2;
2082 match = op - offset;
2083
2084 /* get matchlength */
2085 length = token & ML_MASK;
2086
2087 _copy_match:
2088 if (length == ML_MASK) {
2089 variable_length_error error = ok;
2090 length += read_variable_length(&ip, iend - LASTLITERALS + 1, (int)endOnInput, 0, &error);
2091 if (error != ok) goto _output_error;
2092 if ((safeDecode) && unlikely((uptrval)(op)+length<(uptrval)op)) goto _output_error; /* overflow detection */
2093 }
2094 length += MINMATCH;
2095
2096#if LZ4_FAST_DEC_LOOP
2097 safe_match_copy:
2098#endif
2099 if ((checkOffset) && (unlikely(match + dictSize < lowPrefix))) goto _output_error; /* Error : offset outside buffers */
2100 /* match starting within external dictionary */
2101 if ((dict==usingExtDict) && (match < lowPrefix)) {
2102 if (unlikely(op+length > oend-LASTLITERALS)) {
2103 if (partialDecoding) length = MIN(length, (size_t)(oend-op));
2104 else goto _output_error; /* doesn't respect parsing restriction */
2105 }
2106
2107 if (length <= (size_t)(lowPrefix-match)) {
2108 /* match fits entirely within external dictionary : just copy */
2109 memmove(op, dictEnd - (lowPrefix-match), length);
2110 op += length;
2111 } else {
2112 /* match stretches into both external dictionary and current block */
2113 size_t const copySize = (size_t)(lowPrefix - match);
2114 size_t const restSize = length - copySize;
2115 LZ4_memcpy(op, dictEnd - copySize, copySize);
2116 op += copySize;
2117 if (restSize > (size_t)(op - lowPrefix)) { /* overlap copy */
2118 BYTE* const endOfMatch = op + restSize;
2119 const BYTE* copyFrom = lowPrefix;
2120 while (op < endOfMatch) *op++ = *copyFrom++;
2121 } else {
2122 LZ4_memcpy(op, lowPrefix, restSize);
2123 op += restSize;
2124 } }
2125 continue;
2126 }
2127 assert(match >= lowPrefix);
2128
2129 /* copy match within block */
2130 cpy = op + length;
2131
2132 /* partialDecoding : may end anywhere within the block */
2133 assert(op<=oend);
2134 if (partialDecoding && (cpy > oend-MATCH_SAFEGUARD_DISTANCE)) {
2135 size_t const mlen = MIN(length, (size_t)(oend-op));
2136 const BYTE* const matchEnd = match + mlen;
2137 BYTE* const copyEnd = op + mlen;
2138 if (matchEnd > op) { /* overlap copy */
2139 while (op < copyEnd) { *op++ = *match++; }
2140 } else {
2141 LZ4_memcpy(op, match, mlen);
2142 }
2143 op = copyEnd;
2144 if (op == oend) { break; }
2145 continue;
2146 }
2147
2148 if (unlikely(offset<8)) {
2149 LZ4_write32(op, 0); /* silence msan warning when offset==0 */
2150 op[0] = match[0];
2151 op[1] = match[1];
2152 op[2] = match[2];
2153 op[3] = match[3];
2154 match += inc32table[offset];
2155 LZ4_memcpy(op+4, match, 4);
2156 match -= dec64table[offset];
2157 } else {
2158 LZ4_memcpy(op, match, 8);
2159 match += 8;
2160 }
2161 op += 8;
2162
2163 if (unlikely(cpy > oend-MATCH_SAFEGUARD_DISTANCE)) {
2164 BYTE* const oCopyLimit = oend - (WILDCOPYLENGTH-1);
2165 if (cpy > oend-LASTLITERALS) { goto _output_error; } /* Error : last LASTLITERALS bytes must be literals (uncompressed) */
2166 if (op < oCopyLimit) {
2167 LZ4_wildCopy8(op, match, oCopyLimit);
2168 match += oCopyLimit - op;
2169 op = oCopyLimit;
2170 }
2171 while (op < cpy) { *op++ = *match++; }
2172 } else {
2173 LZ4_memcpy(op, match, 8);
2174 if (length > 16) { LZ4_wildCopy8(op+8, match+8, cpy); }
2175 }
2176 op = cpy; /* wildcopy correction */
2177 }
2178
2179 /* end of decoding */
2180 if (endOnInput) {
2181 DEBUGLOG(5, "decoded %i bytes", (int) (((char*)op)-dst));
2182 return (int) (((char*)op)-dst); /* Nb of output bytes decoded */
2183 } else {
2184 return (int) (((const char*)ip)-src); /* Nb of input bytes read */
2185 }
2186
2187 /* Overflow error detected */
2188 _output_error:
2189 return (int) (-(((const char*)ip)-src))-1;
2190 }
2191}

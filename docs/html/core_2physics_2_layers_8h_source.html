27class ObjectLayerPairFilterImpl : public JPH::ObjectLayerPairFilter

30 virtual bool ShouldCollide(JPH::ObjectLayer inObject1, JPH::ObjectLayer inObject2) const override
31 {

35 case Layers::UNUSED2:
36 case Layers::UNUSED3:
37 case Layers::UNUSED4:
38 return false;
39 case Layers::NON_MOVING:

50 }

64void ReloadFonts(void)
65{
66 ImGuiIO& io = ImGui::GetIO();
67 unsigned char* pixels = nullptr;
68
69 int width;
70 int height;
71 io.Fonts->GetTexDataAsRGBA32(&pixels, &width, &height, nullptr);
72 Image image = GenImageColor(width, height, BLANK);
73 memcpy(image.data, pixels, width * height * 4);
74
75 Texture2D* fontTexture = (Texture2D*)io.Fonts->TexID;
76 if (fontTexture && fontTexture->id != 0)
77 {
78 UnloadTexture(*fontTexture);
79 MemFree(fontTexture);
80 }
81
82 fontTexture = (Texture2D*)MemAlloc(sizeof(Texture2D));
83 *fontTexture = LoadTextureFromImage(image);
84 UnloadImage(image);
85 io.Fonts->TexID = fontTexture;
86}

98static void ImGuiNewFrame(float deltaTime)
99{
100 ImGuiIO& io = ImGui::GetIO();
101
102 if (IsWindowFullscreen())
103 {
104 int monitor = GetCurrentMonitor();
105 io.DisplaySize.x = float(GetMonitorWidth(monitor));
106 io.DisplaySize.y = float(GetMonitorHeight(monitor));
107 }
108 else
109 {
110 io.DisplaySize.x = float(GetScreenWidth());
111 io.DisplaySize.y = float(GetScreenHeight());
112 }
113
114 Vector2 resolutionScale = GetWindowScaleDPI();
115
116#if !defined(__APPLE__)
117 if (!IsWindowState(FLAG_WINDOW_HIGHDPI))
118 resolutionScale = Vector2{ 1,1 };
119#endif
120
121 io.DisplayFramebufferScale = ImVec2(resolutionScale.x, resolutionScale.y);
122
123 io.DeltaTime = deltaTime;
124
125 if (io.WantSetMousePos)
126 {
127 SetMousePosition((int)io.MousePos.x, (int)io.MousePos.y);
128 }
129 else
130 {
131 io.AddMousePosEvent((float)GetMouseX(), (float)GetMouseY());
132 }
133
134 auto setMouseEvent = [&io](int rayMouse, int imGuiMouse)
135 {
136 if (IsMouseButtonPressed(rayMouse))
137 io.AddMouseButtonEvent(imGuiMouse, true);
138 else if (IsMouseButtonReleased(rayMouse))
139 io.AddMouseButtonEvent(imGuiMouse, false);
140 };
141
142 setMouseEvent(MOUSE_BUTTON_LEFT, ImGuiMouseButton_Left);
143 setMouseEvent(MOUSE_BUTTON_RIGHT, ImGuiMouseButton_Right);
144 setMouseEvent(MOUSE_BUTTON_MIDDLE, ImGuiMouseButton_Middle);
145 setMouseEvent(MOUSE_BUTTON_FORWARD, ImGuiMouseButton_Middle+1);
146 setMouseEvent(MOUSE_BUTTON_BACK, ImGuiMouseButton_Middle+2);
147
148 {
149 Vector2 mouseWheel = GetMouseWheelMoveV();
150 io.AddMouseWheelEvent(mouseWheel.x, mouseWheel.y);
151 }
152
153 if ((io.ConfigFlags & ImGuiConfigFlags_NoMouseCursorChange) == 0)
154 {
155 ImGuiMouseCursor imgui_cursor = ImGui::GetMouseCursor();
156 if (imgui_cursor != CurrentMouseCursor || io.MouseDrawCursor)
157 {
158 CurrentMouseCursor = imgui_cursor;
159 if (io.MouseDrawCursor || imgui_cursor == ImGuiMouseCursor_None)
160 {
161 HideCursor();
162 }
163 else
164 {
165 ShowCursor();
166
167 if (!(io.ConfigFlags & ImGuiConfigFlags_NoMouseCursorChange))
168 {
169 SetMouseCursor((imgui_cursor > -1 && imgui_cursor < ImGuiMouseCursor_COUNT) ? MouseCursorMap[imgui_cursor] : MOUSE_CURSOR_DEFAULT);
170 }
171 }
172 }
173 }
174}

185static void ImGuiRenderTriangles(unsigned int count, int indexStart, const ImVector<ImDrawIdx>& indexBuffer, const ImVector<ImDrawVert>& vertBuffer, void* texturePtr)
186{
187 if (count < 3)
188 return;
189
190 Texture* texture = (Texture*)texturePtr;
191
192 unsigned int textureId = (texture == nullptr) ? 0 : texture->id;
193
194 rlBegin(RL_TRIANGLES);
195 rlSetTexture(textureId);
196
197 for (unsigned int i = 0; i <= (count - 3); i += 3)
198 {
199 ImDrawIdx indexA = indexBuffer[indexStart + i];
200 ImDrawIdx indexB = indexBuffer[indexStart + i + 1];
201 ImDrawIdx indexC = indexBuffer[indexStart + i + 2];
202
203 ImDrawVert vertexA = vertBuffer[indexA];
204 ImDrawVert vertexB = vertBuffer[indexB];
205 ImDrawVert vertexC = vertBuffer[indexC];
206
207 ImGuiTriangleVert(vertexA);
208 ImGuiTriangleVert(vertexB);
209 ImGuiTriangleVert(vertexC);
210 }
211 rlEnd();
212}

538void rlImGuiImageRect(const Texture* image, int destWidth, int destHeight, Rectangle sourceRect)
539{
540 if (!image)
541 return;
542
543 if (GlobalContext)
544 ImGui::SetCurrentContext(GlobalContext);
545
546 ImVec2 uv0;
547 ImVec2 uv1;
548
549 if (sourceRect.width < 0)
550 {
551 uv0.x = -((float)sourceRect.x / image->width);
552 uv1.x = (uv0.x - (float)(fabs(sourceRect.width) / image->width));
553 }
554 else
555 {
556 uv0.x = (float)sourceRect.x / image->width;
557 uv1.x = uv0.x + (float)(sourceRect.width / image->width);
558 }
559
560 if (sourceRect.height < 0)
561 {
562 uv0.y = -((float)sourceRect.y / image->height);
563 uv1.y = (uv0.y - (float)(fabs(sourceRect.height) / image->height));
564 }
565 else
566 {
567 uv0.y = (float)sourceRect.y / image->height;
568 uv1.y = uv0.y + (float)(sourceRect.height / image->height);
569 }
570
571 ImGui::Image((ImTextureID)image, ImVec2(float(destWidth), float(destHeight)), uv0, uv1);
572}

585void rlImGuiImageRenderTextureFit(const RenderTexture* image, bool center)
586{
587 if (!image)
588 return;
589
590 if (GlobalContext)
591 ImGui::SetCurrentContext(GlobalContext);
592
593 ImVec2 area = ImGui::GetContentRegionAvail();
594
595 float scale = area.x / image->texture.width;
596
597 float y = image->texture.height * scale;
598 if (y > area.y)
599 {
600 scale = area.y / image->texture.height;
601 }
602
603 int sizeX = int(image->texture.width * scale);
604 int sizeY = int(image->texture.height * scale);
605
606 if (center)
607 {
608 ImGui::SetCursorPosX(0);
609 ImGui::SetCursorPosX(area.x/2 - sizeX/2);
610 ImGui::SetCursorPosY(ImGui::GetCursorPosY() + (area.y / 2 - sizeY / 2));
611 }
612
613 rlImGuiImageRect(&image->texture, sizeX, sizeY, Rectangle{ 0,0, float(image->texture.width), -float(image->texture.height) });
614}

654void ImGui_ImplRaylib_RenderDrawData(ImDrawData* draw_data)
655{
656 rlDrawRenderBatchActive();
657 rlDisableBackfaceCulling();
658
659 for (int l = 0; l < draw_data->CmdListsCount; ++l)
660 {
661 const ImDrawList* commandList = draw_data->CmdLists[l];
662
663 for (const auto& cmd : commandList->CmdBuffer)
664 {

701bool ImGui_ImplRaylib_ProcessEvents(void)
702{
703 ImGuiIO& io = ImGui::GetIO();
704
705 bool focused = IsWindowFocused();
706 if (focused != LastFrameFocused)
707 io.AddFocusEvent(focused);
708 LastFrameFocused = focused;
709
710 // handle the modifyer key events so that shortcuts work
711 bool ctrlDown = rlImGuiIsControlDown();
712 if (ctrlDown != LastControlPressed)
713 io.AddKeyEvent(ImGuiMod_Ctrl, ctrlDown);
714 LastControlPressed = ctrlDown;
715
716 bool shiftDown = rlImGuiIsShiftDown();
717 if (shiftDown != LastShiftPressed)
718 io.AddKeyEvent(ImGuiMod_Shift, shiftDown);
719 LastShiftPressed = shiftDown;
720
721 bool altDown = rlImGuiIsAltDown();
722 if (altDown != LastAltPressed)
723 io.AddKeyEvent(ImGuiMod_Alt, altDown);
724 LastAltPressed = altDown;
725
726 bool superDown = rlImGuiIsSuperDown();
727 if (superDown != LastSuperPressed)
728 io.AddKeyEvent(ImGuiMod_Super, superDown);
729 LastSuperPressed = superDown;
730
731 // get the pressed keys, they are in event order
732 int keyId = GetKeyPressed();
733 while (keyId != 0)
734 {
735 auto keyItr = RaylibKeyMap.find(KeyboardKey(keyId));
736 if (keyItr != RaylibKeyMap.end())
737 io.AddKeyEvent(keyItr->second, true);
738 keyId = GetKeyPressed();
739 }
740
741 // look for any keys that were down last frame and see if they were down and are released
742 for (const auto keyItr : RaylibKeyMap)
743 {
744 if (IsKeyReleased(keyItr.first))
745 io.AddKeyEvent(keyItr.second, false);
746 }
747
748 // add the text input in order
749 unsigned int pressed = GetCharPressed();
750 while (pressed != 0)
751 {
752 io.AddInputCharacter(pressed);
753 pressed = GetCharPressed();
754 }
755
756 if (io.ConfigFlags & ImGuiConfigFlags_NavEnableGamepad && IsGamepadAvailable(0))
757 {
758 HandleGamepadButtonEvent(io, GAMEPAD_BUTTON_LEFT_FACE_UP, ImGuiKey_GamepadDpadUp);

764 HandleGamepadButtonEvent(io, GAMEPAD_BUTTON_RIGHT_FACE_RIGHT, ImGuiKey_GamepadFaceLeft);

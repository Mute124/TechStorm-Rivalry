27#define OUTPUT_WAVE_FILE "Capture.wav"
28#define BUFFERSIZE 4410
29
30#pragma pack (push,1)
31typedef struct
32{
33 char szRIFF[4];
34 long lRIFFSize;
35 char szWave[4];
36 char szFmt[4];
37 long lFmtSize;
38 WAVEFORMATEX wfex;

44int main()
45{
46 ALCdevice *pDevice;
47 ALCcontext *pContext;
48 ALCdevice *pCaptureDevice;
49 const ALCchar *szDefaultCaptureDevice;
50 ALint iSamplesAvailable;
51 FILE *pFile;
52 ALchar Buffer[BUFFERSIZE];
53 WAVEHEADER sWaveHeader;
54 ALint iDataSize = 0;
55 ALint iSize;
56
57 // NOTE : This code does NOT setup the Wave Device's Audio Mixer to select a recording input
58 // or a recording level.
59
60 // Initialize Framework
61 ALFWInit();
62
63 ALFWprintf("Capture Application\n");
64
65 if (!ALFWInitOpenAL())
66 {
67 ALFWprintf("Failed to initialize OpenAL\n");
68 ALFWShutdown();
69 return 0;
70 }
71
72 // Check for Capture Extension support
73 pContext = alcGetCurrentContext();
74 pDevice = alcGetContextsDevice(pContext);
75 if (alcIsExtensionPresent(pDevice, "ALC_EXT_CAPTURE") == AL_FALSE)
76 {
77 ALFWprintf("Failed to detect Capture Extension\n");
78 ALFWShutdownOpenAL();
79 ALFWShutdown();
80 return 0;
81 }
82
83 // Get list of available Capture Devices
84 const ALchar *pDeviceList = alcGetString(NULL, ALC_CAPTURE_DEVICE_SPECIFIER);
85 if (pDeviceList)
86 {
87 ALFWprintf("\nAvailable Capture Devices are:-\n");
88
89 while (*pDeviceList)
90 {
91 ALFWprintf("%s\n", pDeviceList);
92 pDeviceList += strlen(pDeviceList) + 1;
93 }
94 }
95
96 // Get the name of the 'default' capture device
97 szDefaultCaptureDevice = alcGetString(NULL, ALC_CAPTURE_DEFAULT_DEVICE_SPECIFIER);
98 ALFWprintf("\nDefault Capture Device is '%s'\n\n", szDefaultCaptureDevice);
99
100 // Open the default Capture device to record a 22050Hz 16bit Mono Stream using an internal buffer
101 // of BUFFERSIZE Samples (== BUFFERSIZE * 2 bytes)
102 pCaptureDevice = alcCaptureOpenDevice(szDefaultCaptureDevice, 22050, AL_FORMAT_MONO16, BUFFERSIZE);
103 if (pCaptureDevice)
104 {
105 ALFWprintf("Opened '%s' Capture Device\n\n", alcGetString(pCaptureDevice, ALC_CAPTURE_DEVICE_SPECIFIER));
106
107 // Create / open a file for the captured data
108 pFile = fopen(OUTPUT_WAVE_FILE, "wb");
109
110 // Prepare a WAVE file header for the captured data
111 sprintf(sWaveHeader.szRIFF, "RIFF");
112 sWaveHeader.lRIFFSize = 0;
113 sprintf(sWaveHeader.szWave, "WAVE");
114 sprintf(sWaveHeader.szFmt, "fmt ");
115 sWaveHeader.lFmtSize = sizeof(WAVEFORMATEX);
116 sWaveHeader.wfex.nChannels = 1;
117 sWaveHeader.wfex.wBitsPerSample = 16;
118 sWaveHeader.wfex.wFormatTag = WAVE_FORMAT_PCM;
119 sWaveHeader.wfex.nSamplesPerSec = 22050;
120 sWaveHeader.wfex.nBlockAlign = sWaveHeader.wfex.nChannels * sWaveHeader.wfex.wBitsPerSample / 8;
121 sWaveHeader.wfex.nAvgBytesPerSec = sWaveHeader.wfex.nSamplesPerSec * sWaveHeader.wfex.nBlockAlign;
122 sWaveHeader.wfex.cbSize = 0;
123 sprintf(sWaveHeader.szData, "data");
124 sWaveHeader.lDataSize = 0;
125
126 fwrite(&sWaveHeader, sizeof(WAVEHEADER), 1, pFile);
127
128 // Start audio capture
129 alcCaptureStart(pCaptureDevice);
130
131 // Record for two seconds or until a key is pressed
132 DWORD dwStartTime = timeGetTime();
133 while (!ALFWKeyPress() && (timeGetTime() <= (dwStartTime + 2000)))
134 {
135 // Release some CPU time ...
136 Sleep(1);
137
138 // Find out how many samples have been captured
139 alcGetIntegerv(pCaptureDevice, ALC_CAPTURE_SAMPLES, 1, &iSamplesAvailable);
140
141 ALFWprintf("Samples available : %d\r", iSamplesAvailable);
142
143 // When we have enough data to fill our BUFFERSIZE byte buffer, grab the samples
144 if (iSamplesAvailable > (BUFFERSIZE / sWaveHeader.wfex.nBlockAlign))
145 {
146 // Consume Samples
147 alcCaptureSamples(pCaptureDevice, Buffer, BUFFERSIZE / sWaveHeader.wfex.nBlockAlign);
148
149 // Write the audio data to a file
150 fwrite(Buffer, BUFFERSIZE, 1, pFile);
151
152 // Record total amount of data recorded
153 iDataSize += BUFFERSIZE;
154 }
155 }
156
157 // Stop capture
158 alcCaptureStop(pCaptureDevice);
159
160 // Check if any Samples haven't been consumed yet
161 alcGetIntegerv(pCaptureDevice, ALC_CAPTURE_SAMPLES, 1, &iSamplesAvailable);
162 while (iSamplesAvailable)
163 {
164 if (iSamplesAvailable > (BUFFERSIZE / sWaveHeader.wfex.nBlockAlign))
165 {
166 alcCaptureSamples(pCaptureDevice, Buffer, BUFFERSIZE / sWaveHeader.wfex.nBlockAlign);
167 fwrite(Buffer, BUFFERSIZE, 1, pFile);
168 iSamplesAvailable -= (BUFFERSIZE / sWaveHeader.wfex.nBlockAlign);
169 iDataSize += BUFFERSIZE;
170 }
171 else
172 {
173 alcCaptureSamples(pCaptureDevice, Buffer, iSamplesAvailable);
174 fwrite(Buffer, iSamplesAvailable * sWaveHeader.wfex.nBlockAlign, 1, pFile);
175 iDataSize += iSamplesAvailable * sWaveHeader.wfex.nBlockAlign;
176 iSamplesAvailable = 0;
177 }
178 }
179
180 // Fill in Size information in Wave Header
181 fseek(pFile, 4, SEEK_SET);
182 iSize = iDataSize + sizeof(WAVEHEADER) - 8;
183 fwrite(&iSize, 4, 1, pFile);
184 fseek(pFile, 42, SEEK_SET);
185 fwrite(&iDataSize, 4, 1, pFile);
186
187 fclose(pFile);
188
189 ALFWprintf("\nSaved captured audio data to '%s'\n", OUTPUT_WAVE_FILE);
190
191 // Close the Capture Device
192 alcCaptureCloseDevice(pCaptureDevice);
193 }
194
195 // Close down OpenAL
196 ALFWShutdownOpenAL();
197
198 // Close down the Framework
199 ALFWShutdown();
200
201 return 0;
202}

177// Load raw data from rres resource
178void *LoadDataFromResource(rresResourceChunk chunk, unsigned int *size)
179{
180 void *rawData = NULL;
181
182 // Data can be provided in the resource or linked to an external file
183 if (rresGetDataType(chunk.info.type) == RRES_DATA_RAW) // Raw data
184 {
185 rawData = LoadDataFromResourceChunk(chunk, size);
186 }
187 else if (rresGetDataType(chunk.info.type) == RRES_DATA_LINK) // Link to external file
188 {
189 // Get raw data from external linked file
190 unsigned int dataSize = 0;
191 void *data = LoadDataFromResourceLink(chunk, &dataSize);
192
193 rawData = data;
194 *size = dataSize;
195 }
196
197 return rawData;
198}

262// Load Wave data from rres resource
263Wave LoadWaveFromResource(rresResourceChunk chunk)
264{
265 Wave wave = { 0 };
266
267 if (rresGetDataType(chunk.info.type) == RRES_DATA_WAVE) // Wave data
268 {
269 if ((chunk.info.compType == RRES_COMP_NONE) && (chunk.info.cipherType == RRES_CIPHER_NONE))
270 {
271 wave.frameCount = chunk.data.props[0];
272 wave.sampleRate = chunk.data.props[1];
273 wave.sampleSize = chunk.data.props[2];
274 wave.channels = chunk.data.props[3];
275
276 unsigned int size = wave.frameCount*wave.sampleSize/8;
277 wave.data = RL_CALLOC(size, 1);
278 memcpy(wave.data, chunk.data.raw, size);
279 }
280 RRES_LOG("RRES: %c%c%c%c: WARNING: Data must be decompressed/decrypted\n", chunk.info.type[0], chunk.info.type[1], chunk.info.type[2], chunk.info.type[3]);
281 }
282 else if (rresGetDataType(chunk.info.type) == RRES_DATA_RAW) // Raw wave file
283 {
284 unsigned int dataSize = 0;
285 unsigned char *data = LoadDataFromResourceChunk(chunk, &dataSize);
286
287 wave = LoadWaveFromMemory(GetExtensionFromProps(chunk.data.props[1], chunk.data.props[2]), data, dataSize);

306// Load Font data from rres resource
307Font LoadFontFromResource(rresResourceMulti multi)
308{
309 Font font = { 0 };
310
311 // Font resource consist of (2) chunks:
312 // - RRES_DATA_FONT_GLYPHS: Basic font and glyphs properties/data
313 // - RRES_DATA_IMAGE: Image atlas for the font characters
314 if (multi.count >= 2)
315 {
316 if (rresGetDataType(multi.chunks[0].info.type) == RRES_DATA_FONT_GLYPHS)
317 {
318 if ((multi.chunks[0].info.compType == RRES_COMP_NONE) && (multi.chunks[0].info.cipherType == RRES_CIPHER_NONE))
319 {
320 // Load font basic properties from chunk[0]
321 font.baseSize = multi.chunks[0].data.props[0]; // Base size (default chars height)
322 font.glyphCount = multi.chunks[0].data.props[1]; // Number of characters (glyphs)
323 font.glyphPadding = multi.chunks[0].data.props[2]; // Padding around the chars
324
325 font.recs = (Rectangle *)RL_CALLOC(font.glyphCount, sizeof(Rectangle));
326 font.glyphs = (GlyphInfo *)RL_CALLOC(font.glyphCount, sizeof(GlyphInfo));
327
328 for (int i = 0; i < font.glyphCount; i++)
329 {
330 // Font glyphs info comes as a data blob
331 font.recs[i].x = (float)((rresFontGlyphInfo *)multi.chunks[0].data.raw)[i].x;
332 font.recs[i].y = (float)((rresFontGlyphInfo *)multi.chunks[0].data.raw)[i].y;
333 font.recs[i].width = (float)((rresFontGlyphInfo *)multi.chunks[0].data.raw)[i].width;
334 font.recs[i].height = (float)((rresFontGlyphInfo *)multi.chunks[0].data.raw)[i].height;
335
336 font.glyphs[i].value = ((rresFontGlyphInfo *)multi.chunks[0].data.raw)[i].value;
337 font.glyphs[i].offsetX = ((rresFontGlyphInfo *)multi.chunks[0].data.raw)[i].offsetX;
338 font.glyphs[i].offsetY = ((rresFontGlyphInfo *)multi.chunks[0].data.raw)[i].offsetY;
339 font.glyphs[i].advanceX = ((rresFontGlyphInfo *)multi.chunks[0].data.raw)[i].advanceX;
340
341 // NOTE: font.glyphs[i].image is not loaded
342 }
343 }
344 else RRES_LOG("RRES: %s: WARNING: Data must be decompressed/decrypted\n", multi.chunks[0].info.type);
345 }
346
347 // Load font image chunk
348 if (rresGetDataType(multi.chunks[1].info.type) == RRES_DATA_IMAGE)
349 {
350 if ((multi.chunks[0].info.compType == RRES_COMP_NONE) && (multi.chunks[0].info.cipherType == RRES_CIPHER_NONE))
351 {
352 Image image = LoadImageFromResourceChunk(multi.chunks[1]);
353 font.texture = LoadTextureFromImage(image);
354 UnloadImage(image);
355 }
356 else RRES_LOG("RRES: %s: WARNING: Data must be decompressed/decrypted\n", multi.chunks[1].info.type);
357 }
358 }
359 else // One chunk of data: RRES_DATA_RAW or RRES_DATA_LINK?
360 {
361 if (rresGetDataType(multi.chunks[0].info.type) == RRES_DATA_RAW) // Raw font file
362 {
363 unsigned int dataSize = 0;
364 unsigned char *rawData = LoadDataFromResourceChunk(multi.chunks[0], &dataSize);
365
366 font = LoadFontFromMemory(GetExtensionFromProps(multi.chunks[0].data.props[1], multi.chunks[0].data.props[2]), rawData, dataSize, 32, NULL, 0);
367
368 RL_FREE(rawData);
369 }
370 if (rresGetDataType(multi.chunks[0].info.type) == RRES_DATA_LINK) // Link to external font file
371 {
372 // Get raw data from external linked file
373 unsigned int dataSize = 0;
374 void *rawData = LoadDataFromResourceLink(multi.chunks[0], &dataSize);
375
376 // Load image from linked file data
377 // NOTE 1: Loading font at 32px base size and default charset (95 glyphs)
378 // NOTE 2: Function checks internally if the file extension is supported to
379 // properly load the data, if it fails it logs the result and font.texture.id = 0
380 font = LoadFontFromMemory(GetFileExtension(multi.chunks[0].data.raw), rawData, dataSize, 32, NULL, 0);
381
382 RRES_FREE(rawData);
383 }
384 }
385
386 return font;
387}

389// Load Mesh data from rres resource
390// NOTE: We try to load vertex data following raylib structure constraints,
391// in case data does not fit raylib Mesh structure, it is not loaded
392Mesh LoadMeshFromResource(rresResourceMulti multi)
393{
394 Mesh mesh = { 0 };
395
396 // TODO: Support externally linked mesh resource?
397
398 // Mesh resource consist of (n) chunks:
399 for (unsigned int i = 0; i < multi.count; i++)
400 {
401 if ((multi.chunks[0].info.compType == RRES_COMP_NONE) && (multi.chunks[0].info.cipherType == RRES_CIPHER_NONE))
402 {
403 // NOTE: raylib only supports vertex arrays with same vertex count,
404 // rres.chunks[0] defined vertexCount will be the reference for the following chunks
405 // The only exception to vertexCount is the mesh.indices array
406 if (mesh.vertexCount == 0) mesh.vertexCount = multi.chunks[0].data.props[0];
407
408 // Verify chunk type and vertex count
409 if (rresGetDataType(multi.chunks[i].info.type) == RRES_DATA_VERTEX)
410 {
411 // In case vertex count do not match we skip that resource chunk
412 if ((multi.chunks[i].data.props[1] != RRES_VERTEX_ATTRIBUTE_INDEX) && (multi.chunks[i].data.props[0] != mesh.vertexCount)) continue;
413
414 // NOTE: We are only loading raylib supported rresVertexFormat and raylib expected components count
415 switch (multi.chunks[i].data.props[1]) // Check rresVertexAttribute value
416 {
417 case RRES_VERTEX_ATTRIBUTE_POSITION:
418 {
419 // raylib expects 3 components per vertex and float vertex format
420 if ((multi.chunks[i].data.props[2] == 3) && (multi.chunks[i].data.props[3] == RRES_VERTEX_FORMAT_FLOAT))
421 {
422 mesh.vertices = (float *)RL_CALLOC(mesh.vertexCount*3, sizeof(float));
423 memcpy(mesh.vertices, multi.chunks[i].data.raw, mesh.vertexCount*3*sizeof(float));
424 }
425 else RRES_LOG("RRES: WARNING: MESH: Vertex attribute position not valid, componentCount/vertexFormat do not fit\n");
426
427 } break;
428 case RRES_VERTEX_ATTRIBUTE_TEXCOORD1:
429 {
430 // raylib expects 2 components per vertex and float vertex format
431 if ((multi.chunks[i].data.props[2] == 2) && (multi.chunks[i].data.props[3] == RRES_VERTEX_FORMAT_FLOAT))
432 {
433 mesh.texcoords = (float *)RL_CALLOC(mesh.vertexCount*2, sizeof(float));
434 memcpy(mesh.texcoords, multi.chunks[i].data.raw, mesh.vertexCount*2*sizeof(float));
435 }
436 else RRES_LOG("RRES: WARNING: MESH: Vertex attribute texcoord1 not valid, componentCount/vertexFormat do not fit\n");
437
438 } break;
439 case RRES_VERTEX_ATTRIBUTE_TEXCOORD2:
440 {
441 // raylib expects 2 components per vertex and float vertex format
442 if ((multi.chunks[i].data.props[2] == 2) && (multi.chunks[i].data.props[3] == RRES_VERTEX_FORMAT_FLOAT))
443 {
444 mesh.texcoords2 = (float *)RL_CALLOC(mesh.vertexCount*2, sizeof(float));
445 memcpy(mesh.texcoords2, multi.chunks[i].data.raw, mesh.vertexCount*2*sizeof(float));
446 }
447 else RRES_LOG("RRES: WARNING: MESH: Vertex attribute texcoord2 not valid, componentCount/vertexFormat do not fit\n");
448
449 } break;
450 case RRES_VERTEX_ATTRIBUTE_TEXCOORD3:
451 {
452 RRES_LOG("RRES: WARNING: MESH: Vertex attribute texcoord3 not supported\n");
453
454 } break;
455 case RRES_VERTEX_ATTRIBUTE_TEXCOORD4:
456 {
457 RRES_LOG("RRES: WARNING: MESH: Vertex attribute texcoord4 not supported\n");
458
459 } break;
460 case RRES_VERTEX_ATTRIBUTE_NORMAL:
461 {
462 // raylib expects 3 components per vertex and float vertex format
463 if ((multi.chunks[i].data.props[2] == 3) && (multi.chunks[i].data.props[3] == RRES_VERTEX_FORMAT_FLOAT))
464 {
465 mesh.normals = (float *)RL_CALLOC(mesh.vertexCount*3, sizeof(float));
466 memcpy(mesh.normals, multi.chunks[i].data.raw, mesh.vertexCount*3*sizeof(float));
467 }
468 else RRES_LOG("RRES: WARNING: MESH: Vertex attribute normal not valid, componentCount/vertexFormat do not fit\n");
469
470 } break;
471 case RRES_VERTEX_ATTRIBUTE_TANGENT:
472 {
473 // raylib expects 4 components per vertex and float vertex format
474 if ((multi.chunks[i].data.props[2] == 4) && (multi.chunks[i].data.props[3] == RRES_VERTEX_FORMAT_FLOAT))
475 {
476 mesh.tangents = (float *)RL_CALLOC(mesh.vertexCount*4, sizeof(float));
477 memcpy(mesh.tangents, multi.chunks[i].data.raw, mesh.vertexCount*4*sizeof(float));
478 }
479 else RRES_LOG("RRES: WARNING: MESH: Vertex attribute tangent not valid, componentCount/vertexFormat do not fit\n");
480
481 } break;
482 case RRES_VERTEX_ATTRIBUTE_COLOR:
483 {
484 // raylib expects 4 components per vertex and unsigned char vertex format
485 if ((multi.chunks[i].data.props[2] == 4) && (multi.chunks[i].data.props[3] == RRES_VERTEX_FORMAT_UBYTE))
486 {
487 mesh.colors = (unsigned char *)RL_CALLOC(mesh.vertexCount*4, sizeof(unsigned char));
488 memcpy(mesh.colors, multi.chunks[i].data.raw, mesh.vertexCount*4*sizeof(unsigned char));
489 }
490 else RRES_LOG("RRES: WARNING: MESH: Vertex attribute color not valid, componentCount/vertexFormat do not fit\n");
491
492 } break;
493 case RRES_VERTEX_ATTRIBUTE_INDEX:
494 {
495 // raylib expects 1 components per index and unsigned short vertex format
496 if ((multi.chunks[i].data.props[2] == 1) && (multi.chunks[i].data.props[3] == RRES_VERTEX_FORMAT_USHORT))
497 {
498 mesh.indices = (unsigned short *)RL_CALLOC(multi.chunks[i].data.props[0], sizeof(unsigned short));
499 memcpy(mesh.indices, multi.chunks[i].data.raw, multi.chunks[i].data.props[0]*sizeof(unsigned short));
500 }
501 else RRES_LOG("RRES: WARNING: MESH: Vertex attribute index not valid, componentCount/vertexFormat do not fit\n");
502
503 } break;
504 default: break;
505 }
506 }
507 }
508 else RRES_LOG("RRES: WARNING: Vertex provided data must be decompressed/decrypted\n");
509 }
510
511 return mesh;
512}

514// Unpack compressed/encrypted data from resource chunk
515// In case data could not be processed by rres.h, it is just copied in chunk.data.raw for processing here
516// NOTE 1: Function return 0 on success or an error code on failure
517// NOTE 2: Data corruption CRC32 check has already been performed by rresLoadResourceMulti() on rres.h
518int UnpackResourceChunk(rresResourceChunk *chunk)
519{
520 int result = 0;
521 bool updateProps = false;
522
523 // Result error codes:
524 // 0 - No error, decompression/decryption successful
525 // 1 - Encryption algorithm not supported
526 // 2 - Invalid password on decryption
527 // 3 - Compression algorithm not supported
528 // 4 - Error on data decompression
529
530 // NOTE 1: If data is compressed/encrypted the properties are not loaded by rres.h because
531 // it's up to the user to process the data; *chunk must be properly updated by this function
532 // NOTE 2: rres-raylib should support the same algorithms and libraries used by rrespacker tool
533 void *unpackedData = NULL;
534
535 // STEP 1. Data decryption
536 //-------------------------------------------------------------------------------------
537 unsigned char *decryptedData = NULL;
538
539 switch (chunk->info.cipherType)
540 {
541 case RRES_CIPHER_NONE: decryptedData = chunk->data.raw; break;
542#if defined(RRES_SUPPORT_ENCRYPTION_AES)
543 case RRES_CIPHER_AES:
544 {
545 // WARNING: Implementation dependant!
546 // rrespacker tool appends (salt[16] + MD5[16]) to encrypted data for convenience,
547 // Actually, chunk->info.packedSize considers those additional elements
548
549 // Get some memory for the possible message output
550 decryptedData = (unsigned char *)RL_CALLOC(chunk->info.packedSize - 16 - 16, 1);
551 if (decryptedData != NULL) memcpy(decryptedData, chunk->data.raw, chunk->info.packedSize - 16 - 16);
552
553 // Required variables for key stretching
554 uint8_t key[32] = { 0 }; // Encryption key
555 uint8_t salt[16] = { 0 }; // Key stretching salt
556
557 // Retrieve salt from chunk packed data
558 // salt is stored at the end of packed data, before nonce and MAC: salt[16] + MD5[16]
559 memcpy(salt, ((unsigned char *)chunk->data.raw) + (chunk->info.packedSize - 16 - 16), 16);
560
561 // Key stretching configuration
562 crypto_argon2_config config = {
563 .algorithm = CRYPTO_ARGON2_I, // Algorithm: Argon2i
564 .nb_blocks = 16384, // Blocks: 16 MB
565 .nb_passes = 3, // Iterations
566 .nb_lanes = 1 // Single-threaded
567 };
568 crypto_argon2_inputs inputs = {
569 .pass = (const uint8_t *)rresGetCipherPassword(), // User password
570 .pass_size = 16, // Password length
571 .salt = salt, // Salt for the password
572 .salt_size = 16
573 };
574 crypto_argon2_extras extras = { 0 }; // Extra parameters unused
575
576 void *workArea = RL_MALLOC(config.nb_blocks*1024); // Key stretching work area
577
578 // Generate strong encryption key, generated from user password using Argon2i algorithm (256 bit)
579 crypto_argon2(key, 32, workArea, config, inputs, extras);
580
581 // Wipe key generation secrets, they are no longer needed
582 crypto_wipe(salt, 16);
583 RL_FREE(workArea);
584
585 // Required variables for decryption and message authentication
586 unsigned int md5[4] = { 0 }; // Message Authentication Code generated on encryption
587
588 // Retrieve MD5 from chunk packed data
589 // NOTE: MD5 is stored at the end of packed data, after salt: salt[16] + MD5[16]
590 memcpy(md5, ((unsigned char *)chunk->data.raw) + (chunk->info.packedSize - 16), 4*sizeof(unsigned int));
591
592 // Message decryption, requires key
593 struct AES_ctx ctx = { 0 };
594 AES_init_ctx(&ctx, key);
595 AES_CTR_xcrypt_buffer(&ctx, (uint8_t *)decryptedData, chunk->info.packedSize - 16 - 16); // AES Counter mode, stream cipher
596
597 // Verify MD5 to check if data decryption worked
598 unsigned int decryptMD5[4] = { 0 };
599 unsigned int *md5Ptr = ComputeMD5(decryptedData, chunk->info.packedSize - 16 - 16);
600 for (int i = 0; i < 4; i++) decryptMD5[i] = md5Ptr[i];
601
602 // Wipe secrets if they are no longer needed
603 crypto_wipe(key, 32);
604
605 if (memcmp(decryptMD5, md5, 4*sizeof(unsigned int)) == 0) // Decrypted successfully!
606 {
607 chunk->info.packedSize -= (16 + 16); // We remove additional data size from packed size (salt[16] + MD5[16])
608 RRES_LOG("RRES: %c%c%c%c: Data decrypted successfully (AES)\n", chunk->info.type[0], chunk->info.type[1], chunk->info.type[2], chunk->info.type[3]);
609 }
610 else
611 {
612 result = 2; // Data was not decrypted as expected, wrong password or message corrupted
613 RRES_LOG("RRES: WARNING: %c%c%c%c: Data decryption failed, wrong password or corrupted data\n", chunk->info.type[0], chunk->info.type[1], chunk->info.type[2], chunk->info.type[3]);
614 }
615
616 } break;
617#endif
618#if defined(RRES_SUPPORT_ENCRYPTION_XCHACHA20)
619 case RRES_CIPHER_XCHACHA20_POLY1305:
620 {
621 // WARNING: Implementation dependant!
622 // rrespacker tool appends (salt[16] + nonce[24] + MAC[16]) to encrypted data for convenience,
623 // Actually, chunk->info.packedSize considers those additional elements
624
625 // Get some memory for the possible message output
626 decryptedData = (unsigned char *)RL_CALLOC(chunk->info.packedSize - 16 - 24 - 16, 1);
627
628 // Required variables for key stretching
629 uint8_t key[32] = { 0 }; // Encryption key
630 uint8_t salt[16] = { 0 }; // Key stretching salt
631
632 // Retrieve salt from chunk packed data
633 // salt is stored at the end of packed data, before nonce and MAC: salt[16] + nonce[24] + MAC[16]
634 memcpy(salt, ((unsigned char *)chunk->data.raw) + (chunk->info.packedSize - 16 - 24 - 16), 16);
635
636 // Key stretching configuration
637 crypto_argon2_config config = {
638 .algorithm = CRYPTO_ARGON2_I, // Algorithm: Argon2i
639 .nb_blocks = 16384, // Blocks: 16 MB
640 .nb_passes = 3, // Iterations
641 .nb_lanes = 1 // Single-threaded
642 };
643 crypto_argon2_inputs inputs = {
644 .pass = (const uint8_t *)rresGetCipherPassword(), // User password
645 .pass_size = 16, // Password length
646 .salt = salt, // Salt for the password
647 .salt_size = 16
648 };
649 crypto_argon2_extras extras = { 0 }; // Extra parameters unused
650
651 void *workArea = RL_MALLOC(config.nb_blocks*1024); // Key stretching work area
652
653 // Generate strong encryption key, generated from user password using Argon2i algorithm (256 bit)
654 crypto_argon2(key, 32, workArea, config, inputs, extras);
655
656 // Wipe key generation secrets, they are no longer needed
657 crypto_wipe(salt, 16);
658 RL_FREE(workArea);
659
660 // Required variables for decryption and message authentication
661 uint8_t nonce[24] = { 0 }; // nonce used on encryption, unique to processed file
662 uint8_t mac[16] = { 0 }; // Message Authentication Code generated on encryption
663
664 // Retrieve nonce and MAC from chunk packed data
665 // nonce and MAC are stored at the end of packed data, after salt: salt[16] + nonce[24] + MAC[16]
666 memcpy(nonce, ((unsigned char *)chunk->data.raw) + (chunk->info.packedSize - 16 - 24), 24);
667 memcpy(mac, ((unsigned char *)chunk->data.raw) + (chunk->info.packedSize - 16), 16);
668
669 // Message decryption requires key, nonce and MAC
670 int decryptResult = crypto_aead_unlock(decryptedData, mac, key, nonce, NULL, 0, chunk->data.raw, (chunk->info.packedSize - 16 - 24 - 16));
671
672 // Wipe secrets if they are no longer needed
673 crypto_wipe(nonce, 24);
674 crypto_wipe(key, 32);
675
676 if (decryptResult == 0) // Decrypted successfully!
677 {
678 chunk->info.packedSize -= (16 + 24 + 16); // We remove additional data size from packed size
679 RRES_LOG("RRES: %c%c%c%c: Data decrypted successfully (XChaCha20)\n", chunk->info.type[0], chunk->info.type[1], chunk->info.type[2], chunk->info.type[3]);
680 }
681 else if (decryptResult == -1)
682 {
683 result = 2; // Wrong password or message corrupted
684 RRES_LOG("RRES: WARNING: %c%c%c%c: Data decryption failed, wrong password or corrupted data\n", chunk->info.type[0], chunk->info.type[1], chunk->info.type[2], chunk->info.type[3]);
685 }
686 } break;
687#endif
688 default:
689 {
690 result = 1; // Decryption algorithm not supported
691 RRES_LOG("RRES: WARNING: %c%c%c%c: Chunk data encryption algorithm not supported\n", chunk->info.type[0], chunk->info.type[1], chunk->info.type[2], chunk->info.type[3]);
692
693 } break;
694 }
695
696 if ((result == 0) && (chunk->info.cipherType != RRES_CIPHER_NONE))
697 {
698 // Data is not encrypted any more, register it
699 chunk->info.cipherType = RRES_CIPHER_NONE;
700 updateProps = true;
701 }
702
703 // STEP 2: Data decompression (if decryption was successful)
704 //-------------------------------------------------------------------------------------
705 unsigned char *uncompData = NULL;
706
707 if (result == 0)
708 {
709 switch (chunk->info.compType)
710 {
711 case RRES_COMP_NONE: unpackedData = decryptedData; break;
712 case RRES_COMP_DEFLATE:
713 {
714 int uncompDataSize = 0;
715
716 // TODO: WARNING: Possible issue with allocators: RL_CALLOC() vs RRES_CALLOC()
717 uncompData = DecompressData(decryptedData, chunk->info.packedSize, &uncompDataSize);
718
719 if ((uncompData != NULL) && (uncompDataSize > 0)) // Decompression successful
720 {
721 unpackedData = uncompData;
722 chunk->info.packedSize = uncompDataSize;
723 RRES_LOG("RRES: %c%c%c%c: Data decompressed successfully (DEFLATE)\n", chunk->info.type[0], chunk->info.type[1], chunk->info.type[2], chunk->info.type[3]);
724 }
725 else
726 {
727 result = 4; // Decompression process failed
728 RRES_LOG("RRES: WARNING: %c%c%c%c: Chunk data decompression failed\n", chunk->info.type[0], chunk->info.type[1], chunk->info.type[2], chunk->info.type[3]);
729 }
730
731 // Security check, uncompDataSize must match the provided chunk->baseSize
732 if (uncompDataSize != chunk->info.baseSize) RRES_LOG("RRES: WARNING: Decompressed data could be corrupted, unexpected size\n");
733 } break;
734#if defined(RRES_SUPPORT_COMPRESSION_LZ4)
735 case RRES_COMP_LZ4:
736 {
737 int uncompDataSize = 0;
738 uncompData = (unsigned char *)RRES_CALLOC(chunk->info.baseSize, 1);
739 uncompDataSize = LZ4_decompress_safe(decryptedData, uncompData, chunk->info.packedSize, chunk->info.baseSize);
740
741 if ((uncompData != NULL) && (uncompDataSize > 0)) // Decompression successful
742 {
743 unpackedData = uncompData;
744 chunk->info.packedSize = uncompDataSize;
745 RRES_LOG("RRES: %c%c%c%c: Data decompressed successfully (LZ4)\n", chunk->info.type[0], chunk->info.type[1], chunk->info.type[2], chunk->info.type[3]);
746 }
747 else
748 {
749 result = 4; // Decompression process failed
750 RRES_LOG("RRES: WARNING: %c%c%c%c: Chunk data decompression failed\n", chunk->info.type[0], chunk->info.type[1], chunk->info.type[2], chunk->info.type[3]);
751 }
752
753 // WARNING: Decompression could be successful but not the original message size returned
754 if (uncompDataSize != chunk->info.baseSize) RRES_LOG("RRES: WARNING: Decompressed data could be corrupted, unexpected size\n");
755 } break;
756#endif
757 case RRES_COMP_QOI:
758 {
759 int uncompDataSize = 0;
760 qoi_desc desc = { 0 };
761
762 // TODO: WARNING: Possible issue with allocators: QOI_MALLOC() vs RRES_MALLOC()
763 uncompData = qoi_decode(decryptedData, chunk->info.packedSize, &desc, 0);
764 uncompDataSize = (desc.width*desc.height*desc.channels) + 20; // Add the 20 bytes of (propCount + props[4])
765
766 if ((uncompData != NULL) && (uncompDataSize > 0)) // Decompression successful
767 {
768 unpackedData = uncompData;
769 chunk->info.packedSize = uncompDataSize;
770 RRES_LOG("RRES: %c%c%c%c: Data decompressed successfully (QOI)\n", chunk->info.type[0], chunk->info.type[1], chunk->info.type[2], chunk->info.type[3]);
771 }
772 else
773 {
774 result = 4; // Decompression process failed
775 RRES_LOG("RRES: WARNING: %c%c%c%c: Chunk data decompression failed\n", chunk->info.type[0], chunk->info.type[1], chunk->info.type[2], chunk->info.type[3]);
776 }
777
778 if (uncompDataSize != chunk->info.baseSize) RRES_LOG("RRES: WARNING: Decompressed data could be corrupted, unexpected size\n");
779 } break;
780 default:
781 {
782 result = 3;
783 RRES_LOG("RRES: WARNING: %c%c%c%c: Chunk data compression algorithm not supported\n", chunk->info.type[0], chunk->info.type[1], chunk->info.type[2], chunk->info.type[3]);
784 } break;
785 }
786 }
787
788 if ((result == 0) && (chunk->info.compType != RRES_COMP_NONE))
789 {
790 // Data is not encrypted any more, register it
791 chunk->info.compType = RRES_COMP_NONE;
792 updateProps = true;
793 }
794
795 // Update chunk->data.propCount and chunk->data.props if required
796 if (updateProps && (unpackedData != NULL))
797 {
798 // Data is decompressed/decrypted into chunk->data.raw but data.propCount and data.props[] are still empty,
799 // they must be filled with the just updated chunk->data.raw (that contains everything)
800 chunk->data.propCount = ((int *)unpackedData)[0];
801
802 if (chunk->data.propCount > 0)
803 {
804 chunk->data.props = (unsigned int *)RRES_CALLOC(chunk->data.propCount, sizeof(int));
805 for (unsigned int i = 0; i < chunk->data.propCount; i++) chunk->data.props[i] = ((int *)unpackedData)[1 + i];
806 }
807
808 // Move chunk->data.raw pointer (chunk->data.propCount*sizeof(int)) positions
809 void *raw = RRES_CALLOC(chunk->info.baseSize - 20, 1);
810 if (raw != NULL) memcpy(raw, ((unsigned char *)unpackedData) + 20, chunk->info.baseSize - 20);
811 RRES_FREE(chunk->data.raw);
812 chunk->data.raw = raw;
813 RL_FREE(unpackedData);
814 }
815
816 return result;
817}

978// Compute MD5 hash code, returns 4 integers array (static)
979static unsigned int *ComputeMD5(unsigned char *data, int size)
980{
981#define LEFTROTATE(x, c) (((x) << (c)) | ((x) >> (32 - (c))))
982
983 static unsigned int hash[4] = { 0 };
984
985 // NOTE: All variables are unsigned 32 bit and wrap modulo 2^32 when calculating
986
987 // r specifies the per-round shift amounts
988 unsigned int r[] = {
989 7, 12, 17, 22, 7, 12, 17, 22, 7, 12, 17, 22, 7, 12, 17, 22,
990 5, 9, 14, 20, 5, 9, 14, 20, 5, 9, 14, 20, 5, 9, 14, 20,
991 4, 11, 16, 23, 4, 11, 16, 23, 4, 11, 16, 23, 4, 11, 16, 23,
992 6, 10, 15, 21, 6, 10, 15, 21, 6, 10, 15, 21, 6, 10, 15, 21
993 };
994
995 // Use binary integer part of the sines of integers (in radians) as constants// Initialize variables:
996 unsigned int k[] = {
997 0xd76aa478, 0xe8c7b756, 0x242070db, 0xc1bdceee,
998 0xf57c0faf, 0x4787c62a, 0xa8304613, 0xfd469501,
999 0x698098d8, 0x8b44f7af, 0xffff5bb1, 0x895cd7be,
1000 0x6b901122, 0xfd987193, 0xa679438e, 0x49b40821,
1001 0xf61e2562, 0xc040b340, 0x265e5a51, 0xe9b6c7aa,
1002 0xd62f105d, 0x02441453, 0xd8a1e681, 0xe7d3fbc8,
1003 0x21e1cde6, 0xc33707d6, 0xf4d50d87, 0x455a14ed,
1004 0xa9e3e905, 0xfcefa3f8, 0x676f02d9, 0x8d2a4c8a,
1005 0xfffa3942, 0x8771f681, 0x6d9d6122, 0xfde5380c,
1006 0xa4beea44, 0x4bdecfa9, 0xf6bb4b60, 0xbebfbc70,
1007 0x289b7ec6, 0xeaa127fa, 0xd4ef3085, 0x04881d05,
1008 0xd9d4d039, 0xe6db99e5, 0x1fa27cf8, 0xc4ac5665,
1009 0xf4292244, 0x432aff97, 0xab9423a7, 0xfc93a039,
1010 0x655b59c3, 0x8f0ccc92, 0xffeff47d, 0x85845dd1,
1011 0x6fa87e4f, 0xfe2ce6e0, 0xa3014314, 0x4e0811a1,
1012 0xf7537e82, 0xbd3af235, 0x2ad7d2bb, 0xeb86d391
1013 };
1014
1015 hash[0] = 0x67452301;
1016 hash[1] = 0xefcdab89;
1017 hash[2] = 0x98badcfe;
1018 hash[3] = 0x10325476;
1019
1020 // Pre-processing: adding a single 1 bit
1021 // Append '1' bit to message
1022 // NOTE: The input bytes are considered as bits strings,
1023 // where the first bit is the most significant bit of the byte
1024
1025 // Pre-processing: padding with zeros
1026 // Append '0' bit until message length in bit 448 (mod 512)
1027 // Append length mod (2 pow 64) to message
1028
1029 int newDataSize = ((((size + 8)/64) + 1)*64) - 8;
1030
1031 unsigned char *msg = RL_CALLOC(newDataSize + 64, 1); // Also appends "0" bits (we alloc also 64 extra bytes...)
1032 memcpy(msg, data, size);
1033 msg[size] = 128; // Write the "1" bit
1034
1035 unsigned int bitsLen = 8*size;
1036 memcpy(msg + newDataSize, &bitsLen, 4); // We append the len in bits at the end of the buffer
1037
1038 // Process the message in successive 512-bit chunks for each 512-bit chunk of message
1039 for (int offset = 0; offset < newDataSize; offset += (512/8))
1040 {
1041 // Break chunk into sixteen 32-bit words w[j], 0 <= j <= 15
1042 unsigned int *w = (unsigned int *)(msg + offset);
1043
1044 // Initialize hash value for this chunk
1045 unsigned int a = hash[0];
1046 unsigned int b = hash[1];
1047 unsigned int c = hash[2];
1048 unsigned int d = hash[3];
1049
1050 for (int i = 0; i < 64; i++)
1051 {
1052 unsigned int f, g;
1053
1054 if (i < 16)
1055 {
1056 f = (b & c) | ((~b) & d);
1057 g = i;
1058 }
1059 else if (i < 32)
1060 {

335inline bool CheckCollisionRayBox(Ray ray, BoundingBox box, float* outDistance)
336{
337 // Compute the inverse of the ray direction
338 Vector3 invDir = { 1.0f / ray.direction.x, 1.0f / ray.direction.y,
339 1.0f / ray.direction.z };
340
341 // Compute the sign of the inverse direction
342 int signX = (invDir.x < 0.0f) ? 1 : 0;
343 int signY = (invDir.y < 0.0f) ? 1 : 0;
344 int signZ = (invDir.z < 0.0f) ? 1 : 0;
345
346 // Compute the min and max values for each axis
347 float tmin = (box.min.x - ray.position.x) * invDir.x;
348 float tmax = (box.max.x - ray.position.x) * invDir.x;
349 float tymin = (box.min.y - ray.position.y) * invDir.y;
350 float tymax = (box.max.y - ray.position.y) * invDir.y;
351
352 // Swap the min and max values if necessary
353 if (tmin > tmax)
354 {
355 float temp = tmin;
356 tmin = tmax;
357 tmax = temp;
358 }
359
360 if (tymin > tymax)
361 {
362 float temp = tymin;
363 tymin = tymax;
364 tymax = temp;
365 }
366
367 // Check for a miss
368 if ((tmin > tymax) || (tymin > tmax))
369 {
370 return false;
371 }
372
373 // Compute the final min and max values for each axis
374 if (tymin > tmin)
375 {
376 tmin = tymin;
377 }
378
379 if (tymax < tmax)
380 {
381 tmax = tymax;
382 }
383
384 float tzmin = (box.min.z - ray.position.z) * invDir.z;
385 float tzmax = (box.max.z - ray.position.z) * invDir.z;
386
387 // Swap the min and max values if necessary
388 if (tzmin > tzmax)
389 {
390 float temp = tzmin;
391 tzmin = tzmax;
392 tzmax = temp;
393 }
394
395 // Check for a miss
396 if ((tmin > tzmax) || (tzmin > tmax))
397 {
398 return false;
399 }
400
401 // Compute the final min and max values for each axis
402 if (tzmin > tmin)
403 {
404 tmin = tzmin;
405 }
406
407 if (tzmax < tmax)
408 {
409 tmax = tzmax;
410 }
411
412 // Store the distance to the box in the output variable
413 if (outDistance)
414 {
415 *outDistance = tmin;
416 }
417
418 // The ray intersects the box
419 return true;
420}

423inline Vector3 CameraToWorld(Camera3D camera)
424{
425 float distance;
426
427 // Calculate the ray from the camera to the mouse position
428 Vector2 mousePos = GetMousePosition();
429 Vector3 mousePos3 = { mousePos.x, mousePos.y, 0 };
430
431 // Get the direction vector of the ray
432 Vector3 rayDirection =
433 Vector3Normalize(Vector3Subtract(mousePos3, camera.position));
434
435 // Multiply the direction vector with a scalar value to increase the length of
436 // the ray
437 float rayLength = 40.0f;
438 rayDirection = Vector3Scale(rayDirection, rayLength);
439
440 // Create the ray with the modified direction
441 Ray ray = { camera.position, rayDirection };
442
443 // Calculate the distance from the camera to the ground plane
444 distance = -(camera.position.y / ray.position.y);
445
446 // Calculate the point where the ray intersects the ground plane
447 Vector3 groundPoint =
448 Vector3Add(camera.position, Vector3Scale(ray.position, distance));
449
450 return groundPoint;
451}

461inline Vector3 Vector3RandomControlled(Vector3 start, int minDeviation, int maxDeviation)
462{
463 Vector3 result = start;
464
465 int roll = GetRandomValue(1, 3);
466 int deviationRoll = GetRandomValue(0, 1);
467
468 if (roll == 1)
469 {
470 result.x += (float)GetRandomValue(minDeviation, maxDeviation) + minDeviation;
471 result.y += minDeviation;
472 result.z += minDeviation;
473
474 if (deviationRoll == 1)
475 {
476 result.x += minDeviation;
477 }
478 else
479 {
480 result.x -= minDeviation;
481 }
482 }
483 else if (roll == 2)
484 {
485 result.y += (float)GetRandomValue(minDeviation, maxDeviation);
486 result.z += minDeviation;
487
488 if (deviationRoll == 1)
489 {
490 result.y += minDeviation;
491 }
492 else
493 {
494 result.y -= minDeviation;
495 }
496 }
497 else
498 {
499 result.x += minDeviation;
500 result.y += minDeviation;
501 result.z += minDeviation;
502
503 if (deviationRoll == 1)
504 {
505 result.z += (float)GetRandomValue(minDeviation, maxDeviation);
506 }
507 else
508 {
509 result.z -= (float)GetRandomValue(minDeviation, maxDeviation);
510 }
511 }
512
513 return result;
514}

565{
566 GuiLayout* layout = (GuiLayout*)RL_CALLOC(1, sizeof(GuiLayout));

577 char anchorName[MAX_ANCHOR_NAME_LENGTH] = { 0 };
578
579 fgets(buffer, 256, rglFile);

582 {

586 {

615 }

292// Style loading function: terminal
293static void GuiLoadStyleTerminal(void)
294{
295 // Load style properties provided
296 // NOTE: Default properties are propagated
297 for (int i = 0; i < TERMINAL_STYLE_PROPS_COUNT; i++)
298 {
299 GuiSetStyle(terminalStyleProps[i].controlId, terminalStyleProps[i].propertyId, terminalStyleProps[i].propertyValue);
300 }

302 // Custom font loading
303 // NOTE: Compressed font image data (DEFLATE), it requires DecompressData() function
304 int terminalFontDataSize = 0;
305 unsigned char *data = DecompressData(terminalFontData, TERMINAL_COMPRESSED_DATA_SIZE, &terminalFontDataSize);
306 Image imFont = { data, 256, 256, 1, 2 };
307
308 Font font = { 0 };
309 font.baseSize = 16;
310 font.glyphCount = 95;
311
312 // Load texture from image
313 font.texture = LoadTextureFromImage(imFont);
314 UnloadImage(imFont); // Uncompressed data can be unloaded from memory
315
316 // Copy char recs data from global fontRecs
317 // NOTE: Required to avoid issues if trying to free font
318 font.recs = (Rectangle *)malloc(font.glyphCount*sizeof(Rectangle));
319 memcpy(font.recs, terminalFontRecs, font.glyphCount*sizeof(Rectangle));
320
321 // Copy font char info data from global fontChars
322 // NOTE: Required to avoid issues if trying to free font
323 font.glyphs = (GlyphInfo *)malloc(font.glyphCount*sizeof(GlyphInfo));
324 memcpy(font.glyphs, terminalFontChars, font.glyphCount*sizeof(GlyphInfo));

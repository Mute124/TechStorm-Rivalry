314static void GuiLoadStyleBluish(void)
315{
316 // Load style properties provided
317 // NOTE: Default properties are propagated
318 for (int i = 0; i < BLUISH_STYLE_PROPS_COUNT; i++)
319 {
320 GuiSetStyle(bluishStyleProps[i].controlId, bluishStyleProps[i].propertyId, bluishStyleProps[i].propertyValue);
321 }
322
323 // Custom font loading
324 // NOTE: Compressed font image data (DEFLATE), it requires DecompressData() function
325 int bluishFontDataSize = 0;
326 unsigned char *data = DecompressData(bluishFontData, BLUISH_COMPRESSED_DATA_SIZE, &bluishFontDataSize);
327 Image imFont = { data, 256, 256, 1, 2 };
328
329 Font font = { 0 };
330 font.baseSize = 10;
331 font.glyphCount = 95;
332
333 // Load texture from image
334 font.texture = LoadTextureFromImage(imFont);
335 UnloadImage(imFont); // Uncompressed data can be unloaded from memory
336
337 // Copy char recs data from global fontRecs
338 // NOTE: Required to avoid issues if trying to free font
339 font.recs = (Rectangle *)malloc(font.glyphCount*sizeof(Rectangle));
340 memcpy(font.recs, bluishFontRecs, font.glyphCount*sizeof(Rectangle));
341
342 // Copy font char info data from global fontChars
343 // NOTE: Required to avoid issues if trying to free font
344 font.glyphs = (GlyphInfo *)malloc(font.glyphCount*sizeof(GlyphInfo));
345 memcpy(font.glyphs, bluishFontChars, font.glyphCount*sizeof(GlyphInfo));
346
347 GuiSetFont(font);
348
349 // Setup a white rectangle on the font to be used on shapes drawing,
350 // this way we make sure all gui can be drawn on a single pass because no texture change is required
351 // NOTE: Setting up this rectangle is a manual process (for the moment)
352 Rectangle whiteChar = { 66, 5, 2, 2 };
353 SetShapesTexture(font.texture, whiteChar);
354}

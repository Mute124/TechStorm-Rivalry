194double GuiDMValueBox(Rectangle bounds, double value, double minValue, double maxValue, int precision, bool editMode) {
195 // FIXME: Hope all those `memmove()` functions are correctly used so we won't leak memory or overflow the buffer !!!
196 static int framesCounter = 0; // Required for blinking cursor
197 static int cursor = 0; // Required for tracking the cursor position (only for a single active valuebox)
198
199 enum {cursorTimer = 6, maxChars = 31, textPadding = 2};
200
201 GuiState state = GuiGetState();
202
203 // Make sure value is in range
204 if(maxValue != minValue){
205 if(value < minValue) value = minValue;
206 if(value > maxValue) value = maxValue;
207 }
208
209 char textValue[maxChars + 1] = "\0";
210 snprintf(textValue, maxChars, "%.*f", precision, value); // NOTE: use `snprintf` here so we don't overflow the buffer
211 int len = strlen(textValue);
212
213 bool valueHasChanged = false;
214
215 // Update control
216 //--------------------------------------------------------------------
217 if ((state != STATE_DISABLED) && !guiLocked)
218 {
219 if (editMode)
220 {
221 // Make sure cursor position is correct
222 if(cursor > len) cursor = len;
223 if(cursor < 0) cursor = 0;
224
225 state = STATE_PRESSED;
226 framesCounter++;
227
228 if(IsKeyPressed(KEY_RIGHT) || (IsKeyDown(KEY_RIGHT) && (framesCounter%cursorTimer == 0))) {
229 // MOVE CURSOR TO RIGHT
230 ++cursor;
231 framesCounter = 0;
232 } else if(IsKeyPressed(KEY_LEFT) || (IsKeyDown(KEY_LEFT) && (framesCounter%cursorTimer == 0))) {
233 // MOVE CURSOR TO LEFT
234 --cursor;
235 framesCounter = 0;
236 } else if (IsKeyPressed(KEY_BACKSPACE) || (IsKeyDown(KEY_BACKSPACE) && (framesCounter%cursorTimer) == 0)) {
237 // HANDLE BACKSPACE
238 if(cursor > 0) {
239 if(textValue[cursor-1] != '.') {
240 if(cursor < len )
241 memmove(&textValue[cursor-1], &textValue[cursor], len-cursor);
242 textValue[len - 1] = '\0';
243 valueHasChanged = true;
244 }
245 --cursor;
246 }
247 framesCounter = 0;
248 } else if (IsKeyPressed(KEY_DELETE) || (IsKeyDown(KEY_DELETE) && (framesCounter%cursorTimer) == 0)) {
249 // HANDLE DEL
250 if(len > 0 && cursor < len && textValue[cursor] != '.') {
251 memmove(&textValue[cursor], &textValue[cursor+1], len-cursor);
252 textValue[len] = '\0';
253 len -= 1;
254 valueHasChanged = true;
255 }
256 } else if (IsKeyPressed(KEY_HOME)) {
257 // MOVE CURSOR TO START
258 cursor = 0;
259 } else if (IsKeyPressed(KEY_END)) {
260 // MOVE CURSOR TO END
261 cursor = len;
262 } else if (IsKeyDown(KEY_LEFT_CONTROL) && IsKeyPressed(KEY_C)) {
263 // COPY
264 SetClipboardText(textValue);
265 } else if (IsKeyDown(KEY_LEFT_CONTROL) && IsKeyPressed(KEY_X)) {
266 // CUT
267 SetClipboardText(textValue);
268 textValue[0] = '\0';
269 cursor = len = 0;
270 value = 0.0; // set it to 0 and pretend the value didn't change
271 } else if (IsKeyDown(KEY_LEFT_CONTROL) && IsKeyPressed(KEY_V)) {
272 // PASTE
273 const char* clip = GetClipboardText();
274 int clipLen = strlen(clip);
275 clipLen = clipLen > maxChars ? maxChars : clipLen;
276 memcpy(textValue, clip, clipLen);
277 len = clipLen;
278 textValue[len] = '\0';
279 valueHasChanged = true;
280 }
281 else {
282 // HANDLE KEY PRESS
283 int key = GetKeyPressed();
284 if( ((len < maxChars) && (key >= 48) && (key <= 57)) || (key == 46) || (key == 45) ) // only allow 0..9, minus(-) and dot(.)
285 {
286 if(precision != 0 && cursor < len) { // when we have decimals we can't insert at the end
287 memmove(&textValue[cursor], &textValue[cursor-1], len+1-cursor);
288 textValue[len+1] = '\0';
289 textValue[cursor] = (char)key;
290 cursor++;
291 valueHasChanged = true;
292 }
293 else if(precision == 0) {
294 if(cursor < len) memmove(&textValue[cursor], &textValue[cursor-1], len+1-cursor);
295 len += 1;
296 textValue[len+1] = '\0';
297 textValue[cursor] = (char)key;
298 cursor++;
299 valueHasChanged = true;
300 }
301 }
302 }
303
304 // Make sure cursor position is correct
305 if(cursor > len) cursor = len;
306 if(cursor < 0) cursor = 0;
307 }
308 else
309 {
310 if (CheckCollisionPointRec(GetMousePosition(), bounds))
311 {
312 state = STATE_FOCUSED;
313 if (IsMouseButtonPressed(MOUSE_LEFT_BUTTON)) framesCounter = 0;
314 }
315 }
316 }
317 //--------------------------------------------------------------------
318
319 // Draw control
320 //--------------------------------------------------------------------
321 DrawRectangleLinesEx(bounds, GuiGetStyle(VALUEBOX, BORDER_WIDTH), Fade(GetColor(GuiGetStyle(VALUEBOX, BORDER + (state*3))), guiAlpha));
322
323 Rectangle textBounds = {bounds.x + GuiGetStyle(VALUEBOX, BORDER_WIDTH) + textPadding, bounds.y + GuiGetStyle(VALUEBOX, BORDER_WIDTH),
324 bounds.width - 2*(GuiGetStyle(VALUEBOX, BORDER_WIDTH) + textPadding), bounds.height - 2*GuiGetStyle(VALUEBOX, BORDER_WIDTH)};
325
326 int textWidth = GetTextWidth(textValue);
327 if(textWidth > textBounds.width) textBounds.width = textWidth;
328
329 if (state == STATE_PRESSED)
330 {
331 DrawRectangle(bounds.x + GuiGetStyle(VALUEBOX, BORDER_WIDTH), bounds.y + GuiGetStyle(VALUEBOX, BORDER_WIDTH), bounds.width - 2*GuiGetStyle(VALUEBOX, BORDER_WIDTH), bounds.height - 2*GuiGetStyle(VALUEBOX, BORDER_WIDTH), Fade(GetColor(GuiGetStyle(VALUEBOX, BASE_COLOR_PRESSED)), guiAlpha));
332
333 // Draw blinking cursor
334 // NOTE: ValueBox internal text is always centered
335 if (editMode && ((framesCounter/20)%2 == 0)) {
336 // Measure text until the cursor
337 int textWidthCursor = -2;
338 if(cursor > 0) {
339 char c = textValue[cursor];
340 textValue[cursor] = '\0';
341 textWidthCursor = GetTextWidth(textValue);
342 textValue[cursor] = c;
343 }
344 //DrawRectangle(bounds.x + textWidthCursor + textPadding + 2, bounds.y + 2*GuiGetStyle(VALUEBOX, BORDER_WIDTH), 1, bounds.height - 4*GuiGetStyle(VALUEBOX, BORDER_WIDTH), Fade(GetColor(GuiGetStyle(VALUEBOX, BORDER_COLOR_PRESSED)), guiAlpha));
345 DrawRectangle(bounds.x + textWidthCursor + (int)((bounds.width - textWidth - textPadding)/2.0f) + 2, bounds.y + 2*GuiGetStyle(VALUEBOX, BORDER_WIDTH), 1, bounds.height - 4*GuiGetStyle(VALUEBOX, BORDER_WIDTH), Fade(GetColor(GuiGetStyle(VALUEBOX, BORDER_COLOR_PRESSED)), guiAlpha));
346 }
347 }
348 else if (state == STATE_DISABLED)
349 {
350 DrawRectangle(bounds.x + GuiGetStyle(VALUEBOX, BORDER_WIDTH), bounds.y + GuiGetStyle(VALUEBOX, BORDER_WIDTH), bounds.width - 2*GuiGetStyle(VALUEBOX, BORDER_WIDTH), bounds.height - 2*GuiGetStyle(VALUEBOX, BORDER_WIDTH), Fade(GetColor(GuiGetStyle(VALUEBOX, BASE_COLOR_DISABLED)), guiAlpha));
351 }
352
353 GuiDrawText(textValue, textBounds, TEXT_ALIGN_CENTER, Fade(GetColor(GuiGetStyle(VALUEBOX, TEXT + (state*3))), guiAlpha));
354
355 value = valueHasChanged ? strtod(textValue, NULL) : value;
356
357 // Make sure value is in range
358 if(maxValue != minValue){
359 if(value < minValue) value = minValue;
360 if(value > maxValue) value = maxValue;
361 }
362
363 return value;
364}

381
382 // Check spinner state

386 else state = STATE_FOCUSED;
387 }
388 }

390
391
392 // Draw control
393 //--------------------------------------------------------------------
394 // Draw value selector custom buttons

400
401#if defined(RAYGUI_SUPPORT_RICONS)

403 if (GuiButton(rightButtonBound, GuiIconText(RICON_ARROW_RIGHT_FILL, NULL))) value += step;
404#else
405 if (GuiButton(leftButtonBound, "<")) value -= step;
406 if (GuiButton(rightButtonBound, ">")) value += step;
407#endif
408
409 GuiSetStyle(BUTTON, TEXT_ALIGNMENT, tempTextAlign);
410 GuiSetStyle(BUTTON, BORDER_WIDTH, tempBorderWidth);
411
412 value = GuiDMValueBox(spinner, value, minValue, maxValue, precision, editMode);
413
414 return value;
415}

419void GuiDMPropertyList(Rectangle bounds, GuiDMProperty* props, int count, int* focus, int* scrollIndex) {
420 #ifdef RAYGUI_SUPPORT_RICONS
421 #define PROPERTY_COLLAPSED_ICON "#120#"
422 #define PROPERTY_EXPANDED_ICON "#121#"
423 #else
424 #define PROPERTY_COLLAPSED_ICON "+"
425 #define PROPERTY_EXPANDED_ICON "-"
426 #endif
427
428 #define PROPERTY_PADDING 6
429 #define PROPERTY_ICON_SIZE 16
430 #define PROPERTY_DECIMAL_DIGITS 3 //how many digits to show (used only for the vector properties)
431
432 // NOTE: Using ListView style for everything !!
433 GuiState state = GuiGetState();
434 int propFocused = (focus == NULL)? -1 : *focus;
435 int scroll = *scrollIndex > 0 ? 0 : *scrollIndex; // NOTE: scroll should always be negative or 0
436
437 // Each property occupies a certain number of slots, highly synchronized with the properties enum (GUI_PROP_BOOL ... GUI_PROP_SECTION)
438 // NOTE: If you add a custom property type make sure to add the number of slots it occupies here !!
439 const int propSlots[] = {1,1,1,2,1,3,4,5,5,5,1};
440
441 Rectangle absoluteBounds = {0}; // total bounds for all of the properties (unclipped)
442 // We need to loop over all the properties to get total height so we can see if we need a scrollbar or not
443 for(int p=0; p<count; ++p) {
444 // Calculate height of this property (properties can occupy 1 or more slots)
445 int height = GuiGetStyle(LISTVIEW, LIST_ITEMS_HEIGHT);
446 if(props[p].type < (sizeof(propSlots)/sizeof(propSlots[0]))) {
447 if(!PROP_CHECK_FLAG(&props[p], GUI_PFLAG_COLLAPSED)) height = propSlots[props[p].type]*GuiGetStyle(LISTVIEW, LIST_ITEMS_HEIGHT);
448 if(props[p].type == GUI_PROP_SECTION && (PROP_CHECK_FLAG(&props[p], GUI_PFLAG_COLLAPSED))) p += props[p].value.vsection; // skip slots for collapsed section
449 }
450 absoluteBounds.height += height+1;
451 }
452
453 // Check if we need a scrollbar and adjust bounds when necesary
454 bool useScrollBar = absoluteBounds.height > bounds.height - 2*GuiGetStyle(DEFAULT, BORDER_WIDTH) ? true : false;
455 if(!useScrollBar && scroll != 0) scroll = 0; // make sure scroll is 0 when there's no scrollbar
456
457 Rectangle scrollBarBounds = {bounds.x + GuiGetStyle(LISTVIEW, BORDER_WIDTH), bounds.y + GuiGetStyle(LISTVIEW, BORDER_WIDTH),
458 GuiGetStyle(LISTVIEW, SCROLLBAR_WIDTH), bounds.height - 2*GuiGetStyle(DEFAULT, BORDER_WIDTH)};
459
460 absoluteBounds.x = bounds.x + GuiGetStyle(LISTVIEW, LIST_ITEMS_SPACING) + GuiGetStyle(DEFAULT, BORDER_WIDTH);
461 absoluteBounds.y = bounds.y + GuiGetStyle(LISTVIEW, LIST_ITEMS_SPACING) + GuiGetStyle(DEFAULT, BORDER_WIDTH) + scroll;
462 absoluteBounds.width = bounds.width - 2*(GuiGetStyle(LISTVIEW, LIST_ITEMS_SPACING) + GuiGetStyle(DEFAULT, BORDER_WIDTH));
463
464 if(useScrollBar) {
465 if(GuiGetStyle(LISTVIEW, SCROLLBAR_SIDE) == SCROLLBAR_LEFT_SIDE)
466 absoluteBounds.x += GuiGetStyle(LISTVIEW, SCROLLBAR_WIDTH); // scrollbar is on the LEFT, adjust bounds
467 else
468 scrollBarBounds.x = bounds.x + bounds.width - GuiGetStyle(LISTVIEW, BORDER_WIDTH) - GuiGetStyle(LISTVIEW, SCROLLBAR_WIDTH); // scrollbar is on the RIGHT
469 absoluteBounds.width -= GuiGetStyle(LISTVIEW, SCROLLBAR_WIDTH); // adjust width to fit the scrollbar
470 }
471
472 int maxScroll = absoluteBounds.height + 2*(GuiGetStyle(LISTVIEW, LIST_ITEMS_SPACING) + GuiGetStyle(DEFAULT, BORDER_WIDTH))-bounds.height;
473
474 // Update control
475 //--------------------------------------------------------------------
476 Vector2 mousePos = GetMousePosition();
477 // NOTE: most of the update code is actually done in the draw control section
478 if ((state != STATE_DISABLED) && !guiLocked) {
479 if(!CheckCollisionPointRec(mousePos, bounds)) {
480 propFocused = -1;
481 }
482
483 if (useScrollBar)
484 {
485 int wheelMove = GetMouseWheelMove();
486 scroll += wheelMove*count;
487 if(-scroll > maxScroll) scroll = -maxScroll;
488 }
489 }
490 //--------------------------------------------------------------------
491
492
493 // Draw control
494 //--------------------------------------------------------------------
495 DrawRectangleRec(bounds, Fade(GetColor(GuiGetStyle(DEFAULT, BACKGROUND_COLOR)), guiAlpha) ); // Draw background
496 DrawRectangleLinesEx(bounds, GuiGetStyle(DEFAULT, BORDER_WIDTH), Fade(GetColor(GuiGetStyle(LISTVIEW, BORDER + state*3)), guiAlpha)); // Draw border
497
498 BeginScissorMode(absoluteBounds.x, bounds.y + GuiGetStyle(DEFAULT, BORDER_WIDTH), absoluteBounds.width, bounds.height - 2*GuiGetStyle(DEFAULT, BORDER_WIDTH));
499 int currentHeight = 0;
500 for(int p=0; p<count; ++p)
501 {
502 int height = GuiGetStyle(LISTVIEW, LIST_ITEMS_HEIGHT);
503 if(props[p].type < (sizeof(propSlots)/sizeof(propSlots[0])) && !PROP_CHECK_FLAG(&props[p], GUI_PFLAG_COLLAPSED) )
504 height = propSlots[props[p].type]*GuiGetStyle(LISTVIEW, LIST_ITEMS_HEIGHT); // get property height based on how many slots it occupies
505
506 // Even with scissor mode on, draw only properties that we can see (comment out both BeginScissorMode() / EndScissorMode() to see this)
507 if(absoluteBounds.y + currentHeight + height >= bounds.y && absoluteBounds.y + currentHeight <= bounds.y + bounds.height)
508 {
509 Rectangle propBounds = {absoluteBounds.x, absoluteBounds.y + currentHeight, absoluteBounds.width, height};
510 Color textColor = Fade(GetColor(GuiGetStyle(LISTVIEW, TEXT_COLOR_NORMAL)), guiAlpha);
511 int propState = STATE_NORMAL;
512
513 // Get the state of this property and do some initial drawing
514 if(PROP_CHECK_FLAG(&props[p], GUI_PFLAG_DISABLED)) {
515 propState = STATE_DISABLED;
516 propBounds.height += 1;

799bool GuiDMSaveProperties(const char* file, GuiDMProperty* props, int count) {
800 if(file == NULL || props == NULL) return false;
801 if(count == 0) return true;
802
803 FILE* f = fopen(file, "w");
804 if(f == NULL) return false;
805
806 // write header
807 fprintf(f, "#\n# Property types:\n"
808 "# b <name> <flags> <value> // Bool\n"
809 "# i <name> <flags> <value> <min> <max> <step> // Int\n"
810 "# f <name> <flags> <value> <min> <max> <step> <precision> // Float\n"
811 "# t <name> <flags> <value> <edit_length> // Text\n"
812 "# l <name> <flags> <value> <active> // Select\n"
813 "# g <name> <flags> <value> // Section (Group)\n"
814 "# v2 <name> <flags> <x> <y> // Vector 2D\n"
815 "# v3 <name> <flags> <x> <y> <z> // Vector 3D\n"
816 "# v4 <name> <flags> <x> <y> <z> <w> // Vector 4D\n"
817 "# r <name> <flags> <x> <y> <width> <height> // Rectangle\n"
818 "# c <name> <flags> <r> <g> <b> <a> // Color\n"
819 "#\n\n");

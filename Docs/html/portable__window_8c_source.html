43

49 // Main game loop

51 {
52 // Update
53 //----------------------------------------------------------------------------------
54 mousePosition = GetMousePosition();
55
56 if (IsMouseButtonPressed(MOUSE_LEFT_BUTTON) && !dragWindow)
57 {
58 if (CheckCollisionPointRec(mousePosition, (Rectangle){ 0, 0, screenWidth, 20 }))
59 {
60 dragWindow = true;
61 panOffset = mousePosition;
62 }
63 }
64
65 if (dragWindow)
66 {
67 windowPosition.x += (mousePosition.x - panOffset.x);
68 windowPosition.y += (mousePosition.y - panOffset.y);
69
70 SetWindowPosition((int)windowPosition.x, (int)windowPosition.y);
71
72 if (IsMouseButtonReleased(MOUSE_LEFT_BUTTON)) dragWindow = false;
73 }
74 //----------------------------------------------------------------------------------
75
76 // Draw
77 //----------------------------------------------------------------------------------
78 BeginDrawing();
79
80 ClearBackground(RAYWHITE);
81
82 exitWindow = GuiWindowBox((Rectangle){ 0, 0, screenWidth, screenHeight }, "#198# PORTABLE WINDOW");
83
84 DrawText(TextFormat("Mouse Position: [ %.0f, %.0f ]", mousePosition.x, mousePosition.y), 10, 40, 10, DARKGRAY);
85 DrawText(TextFormat("Window Position: [ %.0f, %.0f ]", windowPosition.x, windowPosition.y), 10, 60, 10, DARKGRAY);
86
87 EndDrawing();
88 //----------------------------------------------------------------------------------
89 }
90
91 // De-Initialization
92 //--------------------------------------------------------------------------------------

94 //--------------------------------------------------------------------------------------
95

97}

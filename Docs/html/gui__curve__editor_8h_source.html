144static int CompareGuiCurveEditPointPtr(const void *a, const void *b)
145{
146 float fa = (*(GuiCurveEditorPoint**)a)->position.x;
147 float fb = (*(GuiCurveEditorPoint**)b)->position.x;
148
149 return ((fa > fb) - (fa < fb));
150}

152float GuiCurveEval(GuiCurveEditorState *state, float t)
153{
154 // Sort points
155 GuiCurveEditorPoint* sortedPoints[GUI_CURVE_EDITOR_MAX_POINTS];
156
157 for (int i=0; i < state->numPoints; i++) sortedPoints[i] = &state->points[i];
158
159 qsort(sortedPoints, state->numPoints, sizeof(GuiCurveEditorPoint*), CompareGuiCurveEditPointPtr);
160
161 if (state->numPoints == 0) return state->start;
162
163 // Constants part on edges
164 if (t <= sortedPoints[0]->position.x) return state->start + (state->end-state->start)*sortedPoints[0]->position.y;
165 if (t >= sortedPoints[state->numPoints-1]->position.x) return state->start + (state->end-state->start)*sortedPoints[state->numPoints-1]->position.y;
166
167 // Find curve portion
168 for (int i=0; i < state->numPoints-1; i++)
169 {
170 const GuiCurveEditorPoint *p1 = sortedPoints[i];
171 const GuiCurveEditorPoint *p2 = sortedPoints[i+1];
172
173 // Skip this range
174 if (!((t >= p1->position.x) && (t < p2->position.x)) || (p1->position.x == p2->position.x)) continue;
175
176 float scale = (p2->position.x-p1->position.x);
177 float T = (t-p1->position.x)/scale;
178 float startTangent = scale*p1->tangents.y;
179 float endTangent = scale*p2->tangents.x;
180 float T2 = T*T;
181 float T3 = T*T*T;
182
183 return (state->start + (state->end-state->start)*((2*T3 - 3*T2 + 1)*p1->position.y + (T3 - 2*T2 + T)*startTangent + (3*T2 - 2*T3)*p2->position.y + (T3 - T2)*endTangent));
184 }
185
186 return state->start;
187}

73bool CheckCollisionRayBox(Ray ray, BoundingBox box, float* outDistance)
74{
75 // Compute the inverse of the ray direction
76 Vector3 invDir = { 1.0f / ray.direction.x, 1.0f / ray.direction.y,
77 1.0f / ray.direction.z };
78
79 // Compute the sign of the inverse direction
80 int signX = (invDir.x < 0.0f) ? 1 : 0;
81 int signY = (invDir.y < 0.0f) ? 1 : 0;
82 int signZ = (invDir.z < 0.0f) ? 1 : 0;
83
84 // Compute the min and max values for each axis
85 float tmin = (box.min.x - ray.position.x) * invDir.x;
86 float tmax = (box.max.x - ray.position.x) * invDir.x;
87 float tymin = (box.min.y - ray.position.y) * invDir.y;
88 float tymax = (box.max.y - ray.position.y) * invDir.y;
89
90 // Swap the min and max values if necessary
91 if (tmin > tmax)
92 {
93 float temp = tmin;
94 tmin = tmax;
95 tmax = temp;
96 }
97
98 if (tymin > tymax)
99 {
100 float temp = tymin;
101 tymin = tymax;
102 tymax = temp;
103 }
104
105 // Check for a miss
106 if ((tmin > tymax) || (tymin > tmax))
107 {
108 return false;
109 }
110
111 // Compute the final min and max values for each axis
112 if (tymin > tmin)
113 {
114 tmin = tymin;
115 }
116
117 if (tymax < tmax)
118 {
119 tmax = tymax;
120 }
121
122 float tzmin = (box.min.z - ray.position.z) * invDir.z;
123 float tzmax = (box.max.z - ray.position.z) * invDir.z;
124
125 // Swap the min and max values if necessary
126 if (tzmin > tzmax)
127 {
128 float temp = tzmin;
129 tzmin = tzmax;
130 tzmax = temp;
131 }
132
133 // Check for a miss
134 if ((tmin > tzmax) || (tzmin > tmax))
135 {
136 return false;
137 }
138
139 // Compute the final min and max values for each axis
140 if (tzmin > tmin)
141 {
142 tmin = tzmin;
143 }
144
145 if (tzmax < tmax)
146 {
147 tmax = tzmax;
148 }
149
150 // Store the distance to the box in the output variable
151 if (outDistance)
152 {
153 *outDistance = tmin;
154 }
155
156 // The ray intersects the box
157 return true;
158}

160Vector3 CameraToWorld(Camera3D camera)
161{
162 float distance;
163 // Calculate the ray from the camera to the mouse position
164 Vector2 mousePos = GetMousePosition();
165 Vector3 mousePos3 = { mousePos.x, mousePos.y, 0 };
166 // Get the direction vector of the ray
167 Vector3 rayDirection =
168 Vector3Normalize(Vector3Subtract(mousePos3, camera.position));
169
170 // Multiply the direction vector with a scalar value to increase the length of
171 // the ray
172 float rayLength = 40.0f;
173 rayDirection = Vector3Scale(rayDirection, rayLength);
174
175 // Create the ray with the modified direction
176 Ray ray = { camera.position, rayDirection };
177
178 // Calculate the distance from the camera to the ground plane
179 distance = -(camera.position.y / ray.position.y);
180
181 // Calculate the point where the ray intersects the ground plane
182 Vector3 groundPoint =
183 Vector3Add(camera.position, Vector3Scale(ray.position, distance));
184
185 return groundPoint;
186}

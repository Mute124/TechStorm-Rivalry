233// Returns the cameras forward vector (normalized)
234Vector3 GetCameraForward(Camera* camera)
235{
236 return Vector3Normalize(Vector3Subtract(camera->target, camera->position));
237}
238
239// Returns the cameras up vector (normalized)
240// Note: The up vector might not be perpendicular to the forward vector
241Vector3 GetCameraUp(Camera* camera)
242{
243 return Vector3Normalize(camera->up);
244}
245
246// Returns the cameras right vector (normalized)
247Vector3 GetCameraRight(Camera* camera)
248{
249 Vector3 forward = GetCameraForward(camera);
250 Vector3 up = GetCameraUp(camera);
251
252 return Vector3CrossProduct(forward, up);
253}

255// Moves the camera in its forward direction
256void CameraMoveForward(Camera* camera, float distance, bool moveInWorldPlane)
257{
258 Vector3 forward = GetCameraForward(camera);
259
260 if (moveInWorldPlane)
261 {
262 // Project vector onto world plane
263 forward.y = 0;
264 forward = Vector3Normalize(forward);
265 }
266
267 // Scale by distance
268 forward = Vector3Scale(forward, distance);
269
270 // Move position and target
271 camera->position = Vector3Add(camera->position, forward);
272 camera->target = Vector3Add(camera->target, forward);
273}
274
275// Moves the camera in its up direction
276void CameraMoveUp(Camera* camera, float distance)
277{
278 Vector3 up = GetCameraUp(camera);
279
280 // Scale by distance
281 up = Vector3Scale(up, distance);
282
283 // Move position and target
284 camera->position = Vector3Add(camera->position, up);
285 camera->target = Vector3Add(camera->target, up);
286}

288// Moves the camera target in its current right direction
289void CameraMoveRight(Camera* camera, float distance, bool moveInWorldPlane)
290{
291 Vector3 right = GetCameraRight(camera);
292
293 if (moveInWorldPlane)
294 {
295 // Project vector onto world plane
296 right.y = 0;
297 right = Vector3Normalize(right);
298 }
299
300 // Scale by distance
301 right = Vector3Scale(right, distance);
302
303 // Move position and target
304 camera->position = Vector3Add(camera->position, right);
305 camera->target = Vector3Add(camera->target, right);
306}

308// Moves the camera position closer/farther to/from the camera target
309void CameraMoveToTarget(Camera* camera, float delta)
310{
311 float distance = Vector3Distance(camera->position, camera->target);
312
313 // Apply delta
314 distance += delta;
315
316 // Distance must be greater than 0
317 if (distance <= 0) distance = 0.001f;
318
319 // Set new distance by moving the position along the forward vector
320 Vector3 forward = GetCameraForward(camera);
321 camera->position = Vector3Add(camera->target, Vector3Scale(forward, -distance));
322}

324// Rotates the camera around its up vector
325// Yaw is "looking left and right"
326// If rotateAroundTarget is false, the camera rotates around its position
327// Note: angle must be provided in radians
328void CameraYaw(Camera* camera, float angle, bool rotateAroundTarget)
329{
330 // Rotation axis
331 Vector3 up = GetCameraUp(camera);
332
333 // View vector
334 Vector3 targetPosition = Vector3Subtract(camera->target, camera->position);
335
336 // Rotate view vector around up axis
337 targetPosition = Vector3RotateByAxisAngle(targetPosition, up, angle);
338
339 if (rotateAroundTarget)
340 {
341 // Move position relative to target
342 camera->position = Vector3Subtract(camera->target, targetPosition);
343 }
344 else // rotate around camera.position
345 {
346 // Move target relative to position
347 camera->target = Vector3Add(camera->position, targetPosition);
348 }
349}
350
351// Rotates the camera around its right vector, pitch is "looking up and down"
352// - lockView prevents camera overrotation (aka "somersaults")
353// - rotateAroundTarget defines if rotation is around target or around its position
354// - rotateUp rotates the up direction as well (typically only usefull in CAMERA_FREE)
355// NOTE: angle must be provided in radians
356void CameraPitch(Camera* camera, float angle, bool lockView, bool rotateAroundTarget, bool rotateUp)
357{
358 // Up direction
359 Vector3 up = GetCameraUp(camera);
360
361 // View vector
362 Vector3 targetPosition = Vector3Subtract(camera->target, camera->position);
363
364 if (lockView)
365 {
366 // In these camera modes we clamp the Pitch angle
367 // to allow only viewing straight up or down.
368
369 // Clamp view up
370 float maxAngleUp = Vector3Angle(up, targetPosition);
371 maxAngleUp -= 0.001f; // avoid numerical errors
372 if (angle > maxAngleUp) angle = maxAngleUp;
373
374 // Clamp view down
375 float maxAngleDown = Vector3Angle(Vector3Negate(up), targetPosition);
376 maxAngleDown *= -1.0f; // downwards angle is negative
377 maxAngleDown += 0.001f; // avoid numerical errors
378 if (angle < maxAngleDown) angle = maxAngleDown;
379 }
380
381 // Rotation axis
382 Vector3 right = GetCameraRight(camera);
383
384 // Rotate view vector around right axis
385 targetPosition = Vector3RotateByAxisAngle(targetPosition, right, angle);
386
387 if (rotateAroundTarget)
388 {
389 // Move position relative to target
390 camera->position = Vector3Subtract(camera->target, targetPosition);
391 }
392 else // rotate around camera.position
393 {
394 // Move target relative to position
395 camera->target = Vector3Add(camera->position, targetPosition);
396 }
397
398 if (rotateUp)
399 {
400 // Rotate up direction around right axis
401 camera->up = Vector3RotateByAxisAngle(camera->up, right, angle);
402 }
403}
404
405// Rotates the camera around its forward vector
406// Roll is "turning your head sideways to the left or right"
407// Note: angle must be provided in radians
408void CameraRoll(Camera* camera, float angle)
409{
410 // Rotation axis
411 Vector3 forward = GetCameraForward(camera);
412
413 // Rotate up direction around forward axis
414 camera->up = Vector3RotateByAxisAngle(camera->up, forward, angle);
415}

417// Returns the camera view matrix
418Matrix GetCameraViewMatrix(Camera* camera)
419{
420 return MatrixLookAt(camera->position, camera->target, camera->up);
421}
422
423// Returns the camera projection matrix
424Matrix GetCameraProjectionMatrix(Camera* camera, float aspect)
425{
426 if (camera->projection == CAMERA_PERSPECTIVE)
427 {
428 return MatrixPerspective(camera->fovy * DEG2RAD, aspect, CAMERA_CULL_DISTANCE_NEAR, CAMERA_CULL_DISTANCE_FAR);
429 }
430 else if (camera->projection == CAMERA_ORTHOGRAPHIC)
431 {
432 double top = camera->fovy / 2.0;
433 double right = top * aspect;
434
435 return MatrixOrtho(-right, right, -top, top, CAMERA_CULL_DISTANCE_NEAR, CAMERA_CULL_DISTANCE_FAR);
436 }
437
438 return MatrixIdentity();
439}

442// Update camera position for selected mode
443// Camera mode: CAMERA_FREE, CAMERA_FIRST_PERSON, CAMERA_THIRD_PERSON, CAMERA_ORBITAL or CUSTOM
444void UpdateCamera(Camera* camera, int mode)
445{
446 Vector2 mousePositionDelta = GetMouseDelta();
447
448 bool moveInWorldPlane = ((mode == CAMERA_FIRST_PERSON) || (mode == CAMERA_THIRD_PERSON));
449 bool rotateAroundTarget = ((mode == CAMERA_THIRD_PERSON) || (mode == CAMERA_ORBITAL));
450 bool lockView = ((mode == CAMERA_FIRST_PERSON) || (mode == CAMERA_THIRD_PERSON) || (mode == CAMERA_ORBITAL));
451 bool rotateUp = false;
452
453 if (mode == CAMERA_ORBITAL)
454 {
455 // Orbital can just orbit
456 Matrix rotation = MatrixRotate(GetCameraUp(camera), CAMERA_ORBITAL_SPEED * GetFrameTime());
457 Vector3 view = Vector3Subtract(camera->position, camera->target);
458 view = Vector3Transform(view, rotation);
459 camera->position = Vector3Add(camera->target, view);
460 }
461 else
462 {
463 // Camera rotation
464 if (IsKeyDown(KEY_DOWN)) CameraPitch(camera, -CAMERA_ROTATION_SPEED, lockView, rotateAroundTarget, rotateUp);
465 if (IsKeyDown(KEY_UP)) CameraPitch(camera, CAMERA_ROTATION_SPEED, lockView, rotateAroundTarget, rotateUp);
466 if (IsKeyDown(KEY_RIGHT)) CameraYaw(camera, -CAMERA_ROTATION_SPEED, rotateAroundTarget);
467 if (IsKeyDown(KEY_LEFT)) CameraYaw(camera, CAMERA_ROTATION_SPEED, rotateAroundTarget);
468 if (IsKeyDown(KEY_Q)) CameraRoll(camera, -CAMERA_ROTATION_SPEED);
469 if (IsKeyDown(KEY_E)) CameraRoll(camera, CAMERA_ROTATION_SPEED);
470
471 // Camera movement
472 if (!IsGamepadAvailable(0))
473 {
474 // Camera pan (for CAMERA_FREE)
475 if ((mode == CAMERA_FREE) && (IsMouseButtonDown(MOUSE_BUTTON_MIDDLE)))
476 {
477 const Vector2 mouseDelta = GetMouseDelta();
478 if (mouseDelta.x > 0.0f) CameraMoveRight(camera, CAMERA_PAN_SPEED, moveInWorldPlane);
479 if (mouseDelta.x < 0.0f) CameraMoveRight(camera, -CAMERA_PAN_SPEED, moveInWorldPlane);
480 if (mouseDelta.y > 0.0f) CameraMoveUp(camera, -CAMERA_PAN_SPEED);
481 if (mouseDelta.y < 0.0f) CameraMoveUp(camera, CAMERA_PAN_SPEED);
482 }
483 else
484 {
485 // Mouse support
486 CameraYaw(camera, -mousePositionDelta.x * CAMERA_MOUSE_MOVE_SENSITIVITY, rotateAroundTarget);
487 CameraPitch(camera, -mousePositionDelta.y * CAMERA_MOUSE_MOVE_SENSITIVITY, lockView, rotateAroundTarget, rotateUp);
488 }
489
490 // Keyboard support
491 if (IsKeyDown(KEY_W)) CameraMoveForward(camera, CAMERA_MOVE_SPEED, moveInWorldPlane);
492 if (IsKeyDown(KEY_A)) CameraMoveRight(camera, -CAMERA_MOVE_SPEED, moveInWorldPlane);
493 if (IsKeyDown(KEY_S)) CameraMoveForward(camera, -CAMERA_MOVE_SPEED, moveInWorldPlane);
494 if (IsKeyDown(KEY_D)) CameraMoveRight(camera, CAMERA_MOVE_SPEED, moveInWorldPlane);
495 }
496 else
497 {
498 // Gamepad controller support
499 CameraYaw(camera, -(GetGamepadAxisMovement(0, GAMEPAD_AXIS_RIGHT_X) * 2) * CAMERA_MOUSE_MOVE_SENSITIVITY, rotateAroundTarget);
500 CameraPitch(camera, -(GetGamepadAxisMovement(0, GAMEPAD_AXIS_RIGHT_Y) * 2) * CAMERA_MOUSE_MOVE_SENSITIVITY, lockView, rotateAroundTarget, rotateUp);
501
502 if (GetGamepadAxisMovement(0, GAMEPAD_AXIS_LEFT_Y) <= -0.25f) CameraMoveForward(camera, CAMERA_MOVE_SPEED, moveInWorldPlane);
503 if (GetGamepadAxisMovement(0, GAMEPAD_AXIS_LEFT_X) <= -0.25f) CameraMoveRight(camera, -CAMERA_MOVE_SPEED, moveInWorldPlane);
504 if (GetGamepadAxisMovement(0, GAMEPAD_AXIS_LEFT_Y) >= 0.25f) CameraMoveForward(camera, -CAMERA_MOVE_SPEED, moveInWorldPlane);
505 if (GetGamepadAxisMovement(0, GAMEPAD_AXIS_LEFT_X) >= 0.25f) CameraMoveRight(camera, CAMERA_MOVE_SPEED, moveInWorldPlane);
506 }
507
508 if (mode == CAMERA_FREE)
509 {
510 if (IsKeyDown(KEY_SPACE)) CameraMoveUp(camera, CAMERA_MOVE_SPEED);
511 if (IsKeyDown(KEY_LEFT_CONTROL)) CameraMoveUp(camera, -CAMERA_MOVE_SPEED);
512 }
513 }
514
515 if ((mode == CAMERA_THIRD_PERSON) || (mode == CAMERA_ORBITAL) || (mode == CAMERA_FREE))
516 {
517 // Zoom target distance
518 CameraMoveToTarget(camera, -GetMouseWheelMove());
519 if (IsKeyPressed(KEY_KP_SUBTRACT)) CameraMoveToTarget(camera, 2.0f);
520 if (IsKeyPressed(KEY_KP_ADD)) CameraMoveToTarget(camera, -2.0f);
521 }
522}

525// Update camera movement, movement/rotation values should be provided by user
526void UpdateCameraPro(Camera* camera, Vector3 movement, Vector3 rotation, float zoom)
527{
528 // Required values
529 // movement.x - Move forward/backward
530 // movement.y - Move right/left
531 // movement.z - Move up/down
532 // rotation.x - yaw
533 // rotation.y - pitch
534 // rotation.z - roll
535 // zoom - Move towards target
536
537 bool lockView = true;
538 bool rotateAroundTarget = false;
539 bool rotateUp = false;
540 bool moveInWorldPlane = true;
541
542 // Camera rotation
543 CameraPitch(camera, -rotation.y * DEG2RAD, lockView, rotateAroundTarget, rotateUp);
544 CameraYaw(camera, -rotation.x * DEG2RAD, rotateAroundTarget);
545 CameraRoll(camera, rotation.z * DEG2RAD);
546
547 // Camera movement
548 CameraMoveForward(camera, movement.x, moveInWorldPlane);
549 CameraMoveRight(camera, movement.y, moveInWorldPlane);
550 CameraMoveUp(camera, movement.z);
551
552 // Zoom target distance
553 CameraMoveToTarget(camera, zoom);
554}

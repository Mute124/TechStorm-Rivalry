840 template <typename T, typename Table>
841 using for_each_value_ref = impl::copy_cvref<impl::wrap_node<impl::remove_cvref<impl::unwrap_node<T>>>, Table>;
842
843 template <typename Func, typename Table, typename T>
844 static constexpr bool can_for_each = std::is_invocable_v<Func, const key&, for_each_value_ref<T, Table>> //
845 || std::is_invocable_v<Func, for_each_value_ref<T, Table>>;
846
847 template <typename Func, typename Table, typename T>
848 static constexpr bool can_for_each_nothrow =
849 std::is_nothrow_invocable_v<Func, const key&, for_each_value_ref<T, Table>> //
850 || std::is_nothrow_invocable_v<Func, for_each_value_ref<T, Table>>;
851
852 template <typename Func, typename Table>
853 static constexpr bool can_for_each_any = can_for_each<Func, Table, table> //
854 || can_for_each<Func, Table, array> //
855 || can_for_each<Func, Table, std::string> //
856 || can_for_each<Func, Table, int64_t> //
857 || can_for_each<Func, Table, double> //
858 || can_for_each<Func, Table, bool> //
859 || can_for_each<Func, Table, date> //
860 || can_for_each<Func, Table, time> //
861 || can_for_each<Func, Table, date_time>;
862
863 template <typename Func, typename Table, typename T>
864 static constexpr bool for_each_is_nothrow_one = !can_for_each<Func, Table, T> //
865 || can_for_each_nothrow<Func, Table, T>;

870 template <typename Func, typename Table>
871 static constexpr bool for_each_is_nothrow = for_each_is_nothrow_one<Func, Table, table> //
872 && for_each_is_nothrow_one<Func, Table, array> //
873 && for_each_is_nothrow_one<Func, Table, std::string> //
874 && for_each_is_nothrow_one<Func, Table, int64_t> //
875 && for_each_is_nothrow_one<Func, Table, double> //
876 && for_each_is_nothrow_one<Func, Table, bool> //
877 && for_each_is_nothrow_one<Func, Table, date> //
878 && for_each_is_nothrow_one<Func, Table, time> //
879 && for_each_is_nothrow_one<Func, Table, date_time>;

883 template <typename Func, typename Table>
884 static void do_for_each(Func&& visitor, Table&& tbl) noexcept(for_each_is_nothrow<Func&&, Table&&>)
885 {
886 static_assert(can_for_each_any<Func&&, Table&&>,
887 "TOML table for_each visitors must be invocable for at least one of the toml::node "
888 "specializations:" TOML_SA_NODE_TYPE_LIST);
889
890 using kvp_type = impl::copy_cv<map_pair, std::remove_reference_t<Table>>;
891
892 for (kvp_type& kvp : tbl.map_)
893 {
894 using node_ref = impl::copy_cvref<toml::node, Table&&>;
895 static_assert(std::is_reference_v<node_ref>);
896
897 const auto keep_going =
898 static_cast<node_ref>(*kvp.second)
899 .visit(
900 [&](auto&& v)
901#if !TOML_MSVC // MSVC thinks this is invalid syntax O_o
902 noexcept(for_each_is_nothrow_one<Func&&, Table&&, decltype(v)>)
903#endif
904 {
905 using value_ref = for_each_value_ref<decltype(v), Table&&>;
906 static_assert(std::is_reference_v<value_ref>);
907
908 // func(key, val)
909 if constexpr (std::is_invocable_v<Func&&, const key&, value_ref>)
910 {
911 using return_type = decltype(static_cast<Func&&>(
912 visitor)(static_cast<const key&>(kvp.first), static_cast<value_ref>(v)));
913
914 if constexpr (impl::is_constructible_or_convertible<bool, return_type>)
915 {
916 return static_cast<bool>(static_cast<Func&&>(
917 visitor)(static_cast<const key&>(kvp.first), static_cast<value_ref>(v)));
918 }
919 else
920 {
921 static_cast<Func&&>(visitor)(static_cast<const key&>(kvp.first),
922 static_cast<value_ref>(v));
923 return true;
924 }
925 }
926
927 // func(val)
928 else if constexpr (std::is_invocable_v<Func&&, value_ref>)
929 {
930 using return_type =
931 decltype(static_cast<Func&&>(visitor)(static_cast<value_ref>(v)));
932
933 if constexpr (impl::is_constructible_or_convertible<bool, return_type>)
934 {
935 return static_cast<bool>(
936 static_cast<Func&&>(visitor)(static_cast<value_ref>(v)));
937 }
938 else
939 {
940 static_cast<Func&&>(visitor)(static_cast<value_ref>(v));
941 return true;
942 }
943 }
944
945 // visitor not compatible with this particular type
946 else
947 return true;
948 });
949
950 if (!keep_going)
951 return;
952 }
953 }

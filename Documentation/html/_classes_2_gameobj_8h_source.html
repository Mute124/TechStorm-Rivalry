27RenderTexture2D LoadRenderTextureDepthTex(int width, int height)
28{
29 RenderTexture2D target = { 0 };
30
31 target.id = rlLoadFramebuffer(width, height); // Load an empty framebuffer
32
33 if (target.id > 0)

35 rlEnableFramebuffer(target.id);
36
37 // Create color texture (default to RGBA)
38 target.texture.id = rlLoadTexture(0, width, height, PIXELFORMAT_UNCOMPRESSED_R8G8B8A8, 1);
39 target.texture.width = width;
40 target.texture.height = height;
41 target.texture.format = PIXELFORMAT_UNCOMPRESSED_R8G8B8A8;
42 target.texture.mipmaps = 1;
43
44 // Create depth texture buffer (instead of raylib default renderbuffer)
45 target.depth.id = rlLoadTextureDepth(width, height, false);
46 target.depth.width = width;
47 target.depth.height = height;
48 target.depth.format = 19; //DEPTH_COMPONENT_24BIT?
49 target.depth.mipmaps = 1;
50
51 // Attach color texture and depth texture to FBO
52 rlFramebufferAttach(target.id, target.texture.id, RL_ATTACHMENT_COLOR_CHANNEL0, RL_ATTACHMENT_TEXTURE2D, 0);
53 rlFramebufferAttach(target.id, target.depth.id, RL_ATTACHMENT_DEPTH, RL_ATTACHMENT_TEXTURE2D, 0);
54
55 // Check if fbo is complete with attachments (valid)
56 if (rlFramebufferComplete(target.id)) TRACELOG(LOG_INFO, "FBO: [ID %i] Framebuffer object created successfully", target.id);
57
58 rlDisableFramebuffer();
59 }
60 else TRACELOG(LOG_WARNING, "FBO: Framebuffer object can not be created");
61
62 return target;
63}

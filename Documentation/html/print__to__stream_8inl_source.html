72 void print_integer_to_stream(std::ostream & stream, T val, value_flags format = {}, size_t min_digits = 0)
73 {
74 if (!val)
75 {
76 if (!min_digits)
77 min_digits = 1;
78
79 for (size_t i = 0; i < min_digits; i++)
80 stream.put('0');
81
82 return;
83 }
84
85 static constexpr auto value_flags_mask =
86 value_flags::format_as_binary | value_flags::format_as_octal | value_flags::format_as_hexadecimal;
87 format &= value_flags_mask;
88
89 int base = 10;
90 if (format != value_flags::none && val > T{})
91 {
92 switch (format)
93 {
94 case value_flags::format_as_binary: base = 2; break;
95 case value_flags::format_as_octal: base = 8; break;
96 case value_flags::format_as_hexadecimal: base = 16; break;
97 default: break;
98 }
99 }
100
101#if TOML_INT_CHARCONV
102
103 char buf[(sizeof(T) * CHAR_BIT)];
104 const auto res = std::to_chars(buf, buf + sizeof(buf), val, base);
105 const auto len = static_cast<size_t>(res.ptr - buf);
106 for (size_t i = len; i < min_digits; i++)
107 stream.put('0');
108 if (base == 16)
109 {
110 for (size_t i = 0; i < len; i++)
111 if (buf[i] >= 'a')
112 buf[i] -= 32;
113 }
114 impl::print_to_stream(stream, buf, len);
115
116#else
117
118 using unsigned_type = std::conditional_t<(sizeof(T) > sizeof(unsigned)), std::make_unsigned_t<T>, unsigned>;
119 using cast_type = std::conditional_t<std::is_signed_v<T>, std::make_signed_t<unsigned_type>, unsigned_type>;
120
121 if (base == 2)
122 {
123 const auto len = sizeof(T) * CHAR_BIT;
124 for (size_t i = len; i < min_digits; i++)
125 stream.put('0');
126
127 bool found_one = false;
128 const auto v = static_cast<unsigned_type>(val);
129 unsigned_type mask = unsigned_type{ 1 } << (len - 1u);
130 for (size_t i = 0; i < len; i++)
131 {
132 if ((v & mask))
133 {
134 stream.put('1');
135 found_one = true;
136 }
137 else if (found_one)
138 stream.put('0');
139 mask >>= 1;
140 }
141 }
142 else
143 {
144 std::ostringstream ss;
145 ss.imbue(std::locale::classic());
146 ss << std::uppercase << std::setbase(base);
147 if (min_digits)
148 ss << std::setfill('0') << std::setw(static_cast<int>(min_digits));
149 ss << static_cast<cast_type>(val);
150 const auto str = std::move(ss).str();
151 impl::print_to_stream(stream, str);
152 }
153
154#endif
155 }

158 TOML_INTERNAL_LINKAGE
159 void print_floating_point_to_stream(std::ostream & stream,
160 T val,
161 value_flags format,
162 [[maybe_unused]] bool relaxed_precision)
163 {
164 switch (impl::fpclassify(val))
165 {
166 case impl::fp_class::neg_inf: impl::print_to_stream(stream, "-inf"sv); break;
167
168 case impl::fp_class::pos_inf: impl::print_to_stream(stream, "inf"sv); break;
169
170 case impl::fp_class::nan: impl::print_to_stream(stream, "nan"sv); break;
171
172 case impl::fp_class::ok:
173 {
174 static constexpr auto needs_decimal_point = [](auto&& s) noexcept
175 {
176 for (auto c : s)
177 if (c == '.' || c == 'E' || c == 'e')
178 return false;
179 return true;
180 };
181
182#if TOML_FLOAT_CHARCONV
183
184 const auto hex = !!(format & value_flags::format_as_hexadecimal);
185 char buf[charconv_buffer_length<T>];
186 auto res = hex ? std::to_chars(buf, buf + sizeof(buf), val, std::chars_format::hex)
187 : std::to_chars(buf, buf + sizeof(buf), val);
188 auto str = std::string_view{ buf, static_cast<size_t>(res.ptr - buf) };
189
190 char buf2[charconv_buffer_length<T>];
191 if (!hex && relaxed_precision)
192 {
193 res = std::to_chars(buf2, buf2 + sizeof(buf2), val, std::chars_format::general, 6);
194 const auto str2 = std::string_view{ buf2, static_cast<size_t>(res.ptr - buf2) };
195 if (str2.length() < str.length())
196 str = str2;
197 }
198
199 impl::print_to_stream(stream, str);
200 if (!hex && needs_decimal_point(str))
201 toml::impl::print_to_stream(stream, ".0"sv);
202
203#else
204
205 std::ostringstream ss;
206 ss.imbue(std::locale::classic());
207 if (!relaxed_precision)
208 ss.precision(std::numeric_limits<T>::max_digits10);
209 if (!!(format & value_flags::format_as_hexadecimal))
210 ss << std::hexfloat;
211 ss << val;
212 const auto str = std::move(ss).str();
213 impl::print_to_stream(stream, str);
214 if (!(format & value_flags::format_as_hexadecimal) && needs_decimal_point(str))
215 impl::print_to_stream(stream, ".0"sv);
216
217#endif
218 }
219 break;
220
221 default: TOML_UNREACHABLE;
222 }
223 }

371 void TOML_CALLCONV print_to_stream(std::ostream & stream, const toml::time_offset& val)
372 {
373 if (!val.minutes)
374 {
375 stream.put('Z');
376 return;
377 }
378
379 auto mins = static_cast<int>(val.minutes);
380 if (mins < 0)
381 {
382 stream.put('-');
383 mins = -mins;
384 }
385 else
386 stream.put('+');
387 const auto hours = mins / 60;
388 if (hours)
389 {
390 print_to_stream(stream, static_cast<unsigned int>(hours), {}, 2);
391 mins -= hours * 60;
392 }
393 else
394 print_to_stream(stream, "00"sv);
395 stream.put(':');
396 print_to_stream(stream, static_cast<unsigned int>(mins), {}, 2);
397 }

24 void yaml_formatter::print_yaml_string(const value<std::string>& str)
25 {
26 if (str->empty())
27 {
28 base::print(str);
29 return;
30 }
31
32 bool contains_newline = false;
33 for (auto c = str->c_str(), e = str->c_str() + str->length(); c < e && !contains_newline; c++)
34 contains_newline = *c == '\n';
35
36 if (contains_newline)
37 {
38 print_unformatted("|-"sv);
39
40 increase_indent();
41
42 auto line_end = str->c_str() - 1u;
43 const auto end = str->c_str() + str->length();
44 while (line_end != end)
45 {
46 auto line_start = line_end + 1u;
47 line_end = line_start;
48 for (; line_end != end && *line_end != '\n'; line_end++)
49 ;
50
51 if TOML_LIKELY(line_start != line_end || line_end != end)
52 {
53 print_newline();
54 print_indent();
55 print_unformatted(std::string_view{ line_start, static_cast<size_t>(line_end - line_start) });
56 }
57 }
58
59 decrease_indent();
60 }
61 else
62 print_string(*str, false, true);
63 }

66 void yaml_formatter::print(const toml::table& tbl, bool parent_is_array)
67 {
68 if (tbl.empty())
69 {
70 print_unformatted("{}"sv);
71 return;
72 }
73
74 increase_indent();
75
76 for (auto&& [k, v] : tbl)
77 {
78 if (!parent_is_array)
79 {
80 print_newline();
81 print_indent();
82 }
83 parent_is_array = false;
84
85 print_string(k.str(), false, true);
86 if (terse_kvps())
87 print_unformatted(":"sv);
88 else
89 print_unformatted(": "sv);
90
91 const auto type = v.type();
92 TOML_ASSUME(type != node_type::none);
93 switch (type)
94 {
95 case node_type::table: print(*reinterpret_cast<const table*>(&v)); break;
96 case node_type::array: print(*reinterpret_cast<const array*>(&v)); break;
97 case node_type::string: print_yaml_string(*reinterpret_cast<const value<std::string>*>(&v)); break;
98 default: print_value(v, type);
99 }
100 }
101
102 decrease_indent();
103 }
